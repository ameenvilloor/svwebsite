155 double scorecXYZ_aspectRatio2(dArray *xyz) {
156  double eLong, hShort2, vec1[3], vec2[3], vec3[3], ilen;
157  double alt2, length, dp12;
158  int i,j;
159  void *temp = 0;
160 
161  eLong = 0.0;
162  for (i = 0; i < 3; i++) {
163  for (j = i+1; j < 4; j++) {
164  scorecDiffVt(xyz[i], xyz[j], vec1);
165  length = scorecDotProd(vec1,vec1);
166  if (length > eLong) eLong = length;
167  }
168  }
169 
170  hShort2 = MTBIG;
171  for (i = 0; i < 4; i++) {
172  scorecDiffVt(xyz[(i+1)%4], xyz[(i+2)%4], vec1);
173  scorecDiffVt(xyz[(i+1)%4], xyz[(i+3)%4], vec2);
174  scorecCrossProd(vec1, vec2, vec3);
175  ilen = 1.0/scorecDotProd(vec3, vec3); /* reciprocal of vector length */
176 
177  scorecDiffVt(xyz[i], xyz[(i+1)%4], vec1);
178  dp12 = scorecDotProd(vec1, vec3);
179  alt2 = dp12*dp12*ilen; /* square of height */
180  if (hShort2 > alt2) hShort2 = alt2;
181  }
182 
183  return (eLong/hShort2);
184 
185 } /* scorecXYZ_aspectRatio2 */

200 void scorecXYZ_dihedral(dArray *xyz,double *sml,double *big) {
201  double cossml,cosbig,cosangs[6];
202  int e;
203 
204  /* get all 6 dihedral angles */
205  scorecXYZ_dhdAngs(xyz,cosangs);
206  cossml=cosangs[0];
207  cosbig=cosangs[0];
208  /* evaluate 5 other dihedral angles */
209  for (e=1; e<6; e++) {
210  /* dihedral angle evaluation */
211  if (cosangs[e]<cosbig)
212  cosbig = cosangs[e];
213  if (cosangs[e] > cossml)
214  cossml = cosangs[e];
215  }
216  if (cosbig<0.)
217  *big = -sqrt(-cosbig);
218  else
219  *big = sqrt(cosbig);
220  if (cossml<0.)
221  *sml = -sqrt(-cossml);
222  else
223  *sml = sqrt(cossml);
224 }

288 int scorecXYZ_shape(dArray *xyz,double *shape) {
289  double v31[3],v32[3],v30[3],v20[3],v21[3],cp[3],jcb,area ;
290 
291  scorecDiffVt(xyz[0],xyz[3],v30) ;
292  scorecDiffVt(xyz[2],xyz[3],v32) ;
293  scorecDiffVt(xyz[1],xyz[3],v31) ;
294  scorecCrossProd(v30,v32,cp) ;
295  jcb = scorecDotProd(cp,v31) ;
296  /* check if element is valid */
297  if ( jcb < 0. ) {
298  *shape = jcb ;
299  return(0) ;
300  }
301  /* compute the face areas */
302  area = scorecDotProd(cp,cp) ;
303  scorecCrossProd(v30,v31,cp) ;
304  area += scorecDotProd(cp,cp) ;
305  scorecCrossProd(v31,v32,cp) ;
306  area += scorecDotProd(cp,cp) ;
307 
308  scorecDiffVt(xyz[0],xyz[2],v20) ;
309  scorecDiffVt(xyz[1],xyz[2],v21) ;
310  scorecCrossProd(v20,v21,cp) ;
311  area += scorecDotProd(cp,cp) ;
312 
313  *shape = jcb / area ;
314  *shape = (*shape) * (*shape) * (*shape) *jcb ;
315  return(1) ;
316 }

346
