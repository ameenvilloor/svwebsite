130 int cvsolverIO::readHeader (const char* keyphrase,int* valueArray,
131  int nItems,const char* datatype,
132  const char* iotype) {
133 
134  int i,skip_size,integer_value;
135  int rewinded = 0;
136 
137  isBinary( iotype );
138 
139  LastHeaderKey_[0] = '\0';
140 
141  while (rewinded < 2) {
142  while (gzgets(filePointer_, Line_, 1024) != Z_NULL) {
143 
144  // ignore comment lines
145  if (Line_[0] == '#') {
146  //fprintf(stdout,"COMMENT: %s",Line_);
147  continue;
148  }
149 
150  // ignore blank lines
151 
152  //char* chkblank = StringStripper(Line_);
153  //fprintf(stdout,"chkblank: %i\n",strlen(chkblank));
154  if (strlen(Line_) <= 1) {
155  //fprintf(stdout,"Skipping blank line?\n");
156  continue;
157  }
158  //fprintf(stdout,"Line_: %s",Line_);
159  char* token = strtok ( Line_, ":" );
160  //fprintf(stdout,"token: %s\n",token);
161  if( cscompare( keyphrase , token ) ) {
162  LastHeaderKey_[0] = '\0';
163  sprintf(LastHeaderKey_,"%s",keyphrase);
164  token = strtok( NULL, " ,;<>" );
165  skip_size = 0;
166  skip_size = atoi( token );
167  for( i=0;i < nItems && ( token = strtok( NULL," ,;<>") );i++) {
168  valueArray[i] = atoi(token);
169  //fprintf(stdout,"integer (%i): %i\n",i,atoi(token));
170  }
171  if ( i < nItems ) {
172  fprintf(stderr,"Expected # of ints not recoverd from head\n");
173  fprintf(stderr,"when looking for : %s\n", keyphrase);
174  return CVSOLVER_IO_ERROR;
175  } else {
176  return CVSOLVER_IO_OK;
177  }
178  }
179  // this really belongs when you open the file!
180  if ( cscompare(token,"byteorder magic number") ) {
181  if ( binary_format_ ) {
182  gzread(filePointer_,&integer_value,sizeof(int));
183  char junk;
184  gzread(filePointer_,&junk,sizeof(char)); /* reading the new line */
185  } else {
186  gzgets(filePointer_,Line_,1024);
187  sscanf(Line_,"%i",&integer_value);
188  }
189  if ( FLOWSOLVER_MAGIC_NUMBER != integer_value ) {
190  //fprintf(stdout,"NOTE: Wrong_Endian found.\n");
191  Wrong_Endian_ = true;
192  }
193  continue;
194  }
195 
196  // skip to next header
197  token = strtok( NULL, " ,;<>" );
198  skip_size = atoi( token );
199  if ( binary_format_ ) {
200  gzseek(filePointer_,skip_size,SEEK_CUR);
201  } else {
202  for( int gama=0; gama < skip_size; gama++ ) {
203  gzgets(filePointer_,Line_,1024);
204  }
205  }
206 
207  } // end inner while
208 
209  gzrewind(filePointer_);
210  gzclearerr(filePointer_);
211  rewinded++;
212 
213  } // end outer while
214 
215  return CVSOLVER_IO_ERROR;
216 
217 }

220 int cvsolverIO::readDataBlock ( const char* keyphrase,
221  void* valueArray,
222  int nItems,
223  const char* datatype,
224  const char* iotype ) {
225 
226  int n;
227  int* valueArrayInt;
228  float* valueArrayFloat;
229  double* valueArrayDouble;
230  char junk;
231 
232  // check that the file has been opened
233  if (filePointer_ == Z_NULL) {
234  fprintf(stderr,"No file associated with Descriptor \n");
235  fprintf(stderr,"openfile_ function has to be called before \n");
236  fprintf(stderr,"acessing the file\n");
237  return CVSOLVER_IO_ERROR;
238  }
239 
240  if ( LastHeaderNotFound_ ) {
241  fprintf(stderr,"ERROR: last header was not found.\n");
242  return CVSOLVER_IO_ERROR;
243  }
244 
245  // error check..
246  // since we require that a consistant header always preceed the data block
247  // let us check to see that it is actually the case.
248 
249  if ( ! cscompare( LastHeaderKey_, keyphrase ) ) {
250  fprintf(stderr,"ERROR: header not consistant with data block\n");
251  fprintf(stderr,"Header: %s\n", LastHeaderKey_);
252  fprintf(stderr,"DataBlock: %s\n", keyphrase);
253  fprintf(stderr,"Please recheck read sequence\n");
254  return CVSOLVER_IO_ERROR;
255  }
256 
257  size_t type_size = typeSize( datatype );
258  isBinary( iotype );
259 
260  if ( binary_format_ ) {
261 
262  //fread(valueArray,type_size,nItems,filePointer_ );
263  gzread(filePointer_,valueArray,type_size*nItems);
264  //fread(&junk, sizeof(char), 1, filePointer_);
265  gzread(filePointer_,&junk,sizeof(char)); /* reading the new line */
266  if ( Wrong_Endian_ ) SwapArrayByteOrder( valueArray, type_size, nItems );
267 
268  } else {
269  char junk;
270  switch( type_of_data_ ) {
271  case INT:
272  valueArrayInt = static_cast<int*>( valueArray );
273  for( n=0; n < nItems ; n++ ) {
274  int integer_value;
275  Line_[0] = '\0';
276  gzgets(filePointer_,Line_,1024);
277  if (sscanf(Line_,"%i",&integer_value) != 1) {
278  return CVSOLVER_IO_ERROR;
279  }
280  valueArrayInt[n] = integer_value;
281  }
282  break;
283  case FLOAT:
284  valueArrayFloat = static_cast<float*>( valueArray );
285  for( n=0; n < nItems ; n++ ) {
286  float float_value;
287  Line_[0] = '\0';
288  gzgets(filePointer_,Line_,1024);
289  if (sscanf(Line_,"%f",&float_value) != 1) {
290  return CVSOLVER_IO_ERROR;
291  }
292  valueArrayFloat[n] = float_value;
293  }
294  break;
295  case DOUBLE:
296  valueArrayDouble = static_cast<double*>( valueArray );
297  for( n=0; n < nItems ; n++ ) {
298  double double_value;
299  Line_[0] = '\0';
300  gzgets(filePointer_,Line_,1024);
301  if (sscanf(Line_,"%lf",&double_value) != 1) {
302  return CVSOLVER_IO_ERROR;
303  }
304  valueArrayDouble[n] = double_value;
305  }
306  break;
307  }
308  }
309 
310  return CVSOLVER_IO_OK;
311 }

519 void cvsolverIO::SwapArrayByteOrder( void* array, int nbytes, int nItems ) {
520  /* This swaps the byte order for the array of nItems each
521  of size nbytes , This will be called only locally */
522  int i,j;
523  unsigned char ucTmp;
524  unsigned char* ucDst = (unsigned char*)array;
525 
526  for(i=0; i < nItems; i++) {
527  for(j=0; j < (nbytes/2); j++)
