61 void vtkvmtkVoronoiDiagram3D::ExtractUniqueEdges(vtkUnstructuredGrid* input, vtkCellArray* edgeArray) // could be a class
62 {
63  int isVisited;
64  vtkIdType i, j, k;
65  vtkIdType edgePts[2];
66  vtkIdType npts, *pts;
67  vtkIntArray* visited;
68  vtkIdList* pointCells;
69  vtkIdList* insertedLoopPoints;
70 
71  visited = vtkIntArray::New();
72  visited->SetNumberOfTuples(input->GetNumberOfPoints());
73  visited->FillComponent(0,0);
74 
75  pointCells = vtkIdList::New();
76  insertedLoopPoints = vtkIdList::New();
77 
78  for (i=0; i<input->GetNumberOfPoints(); i++)
79  {
80  visited->SetValue(i,1);
81  edgePts[0] = i;
82  pointCells->Initialize();
83  insertedLoopPoints->Initialize();
84  input->GetPointCells(i,pointCells);
85  for (j=0; j<pointCells->GetNumberOfIds(); j++)
86  {
87  input->GetCellPoints(pointCells->GetId(j),npts,pts);
88  for (k=0; k<npts; k++)
89  {
90  isVisited = visited->GetValue(pts[k]);
91  if (!isVisited)
92  if (insertedLoopPoints->IsId(pts[k])==-1)
93  {
94  edgePts[1] = pts[k];
95  insertedLoopPoints->InsertNextId(pts[k]);
96  edgeArray->InsertNextCell(2,edgePts);
97  }
98  }
99  }
100  }
101 
102  visited->Delete();
103  pointCells->Delete();
104  insertedLoopPoints->Delete();
105 
106 }

108 void vtkvmtkVoronoiDiagram3D::BuildVoronoiPolys(vtkUnstructuredGrid* input, vtkCellArray* voronoiPolys)
109 {
110  bool boundaryTetra;
111  vtkIdType npts, *pts;
112  pts = NULL;
113  vtkIdType neighborTetraId;
114  vtkIdType i, k, h;
115  vtkCellArray* edgeArray;
116  vtkIdList* polyIds;
117  vtkIdList* neighborCells;
118  vtkIdList* neighborTrianglePointIds;
119  vtkIdList* neighborNeighborCells;
120  vtkIdList* linePointIds;
121  vtkTetra* neighborTetra;
122 
123  edgeArray = vtkCellArray::New();
124  polyIds = vtkIdList::New();
125  neighborCells = vtkIdList::New();
126  neighborTrianglePointIds = vtkIdList::New();
127  neighborNeighborCells = vtkIdList::New();
128  linePointIds = vtkIdList::New();
129 
130  this->ExtractUniqueEdges(input,edgeArray);
131 
132  edgeArray->InitTraversal();
133  for (i=0; i<edgeArray->GetNumberOfCells(); i++)
134  {
135  edgeArray->GetNextCell(npts,pts);
136  linePointIds->Initialize();
137  linePointIds->InsertNextId(pts[0]);
138  linePointIds->InsertNextId(pts[1]);
139 
140  boundaryTetra = false;
141  neighborCells->Initialize();
142  neighborTrianglePointIds->Initialize();
143  polyIds->Initialize();
144  input->GetCellNeighbors(-1,linePointIds,neighborCells);
145 
146  neighborTrianglePointIds->InsertNextId(linePointIds->GetId(0));
147  neighborTrianglePointIds->InsertNextId(linePointIds->GetId(1));
148 
149  neighborTetraId = neighborCells->GetId(0);
150  neighborTetra = vtkTetra::SafeDownCast(input->GetCell(neighborTetraId));
151  polyIds->InsertNextId(neighborTetraId);
152  for (k=0; k<neighborCells->GetNumberOfIds(); k++)
153  {
154  neighborNeighborCells->Initialize();
155  for (h=0; h<neighborTetra->GetNumberOfPoints(); h++)
156  {
157  if (neighborTrianglePointIds->IsId(neighborTetra->GetPointId(h))==-1)
158  {
159  neighborTrianglePointIds->InsertId(2,neighborTetra->GetPointId(h));
160  input->GetCellNeighbors(neighborTetraId,neighborTrianglePointIds,neighborNeighborCells);
161  if (neighborNeighborCells->GetNumberOfIds()==0)
162  {
163  boundaryTetra = true;
164  break;
165  }
166  else if (neighborNeighborCells->GetNumberOfIds()==1)
167  {
168  neighborTetraId = neighborNeighborCells->GetId(0);
169  neighborTetra = vtkTetra::SafeDownCast(input->GetCell(neighborTetraId));
170  polyIds->InsertUniqueId(neighborTetraId);
171  break;
172  }
173  }
174  }
175  if (boundaryTetra)
176  break;
177  }
178  if (!boundaryTetra)
179  voronoiPolys->InsertNextCell(polyIds);
180  }
181 
182  edgeArray->Delete();
183  polyIds->Delete();
184  neighborCells->Delete();
185  neighborTrianglePointIds->Delete();
186  neighborNeighborCells->Delete();
187  linePointIds->Delete();
188 }

190 int vtkvmtkVoronoiDiagram3D::RequestData(
191  vtkInformation *vtkNotUsed(request),
192  vtkInformationVector **inputVector,
193  vtkInformationVector *outputVector)
194 {
195  vtkInformation *inInfo = inputVector[0]->GetInformationObject(0);
196  vtkInformation *outInfo = outputVector->GetInformationObject(0);
197 
198  vtkUnstructuredGrid* input = vtkUnstructuredGrid::SafeDownCast(
199  inInfo->Get(vtkDataObject::DATA_OBJECT()));
200  vtkPolyData *output = vtkPolyData::SafeDownCast(
201  outInfo->Get(vtkDataObject::DATA_OBJECT()));
202 
203  // Declare
204  double p0[3], p1[3], p2[3], p3[3];
205  double tetraCenter[3], tetraRadius;
206  double currentRadius;
207  vtkIdType i, j, id, poleId;
208  vtkPoints* newPoints;
209  vtkCellArray* newPolys;
210  vtkCellArray* newLines;
211  vtkDoubleArray* newScalars;
212  vtkDoubleArray* thicknessScalars;
213  vtkIdList* cellIds;
214  vtkTetra* tetra;
215 
216  poleId = -1;
217 
218  // Allocate
219  newPoints = vtkPoints::New();
220  newPoints->SetNumberOfPoints(input->GetNumberOfCells());
221  newScalars = vtkDoubleArray::New();
222  newScalars->SetNumberOfTuples(input->GetNumberOfCells());
223  newPolys = vtkCellArray::New();
224  newLines = vtkCellArray::New();
225  thicknessScalars = vtkDoubleArray::New();
226  thicknessScalars->SetNumberOfTuples(input->GetNumberOfPoints());
227  thicknessScalars->FillComponent(0,0.0);
228 
229  cellIds = vtkIdList::New();
230 
231  this->PoleIds->SetNumberOfIds(input->GetNumberOfPoints());
232 
233  // Execute
234 
235  for (i=0; i<input->GetNumberOfPoints(); i++)
236  {
237  this->PoleIds->SetId(i,-1);
238  }
239 
240  for (i=0; i<input->GetNumberOfCells(); i++)
241  {
242  tetra = (vtkTetra*) input->GetCell(i);
243  tetra->GetPoints()->GetPoint(0,p0);
244  tetra->GetPoints()->GetPoint(1,p1);
245  tetra->GetPoints()->GetPoint(2,p2);
246  tetra->GetPoints()->GetPoint(3,p3);
247  tetraRadius = sqrt(vtkTetra::Circumsphere(p0,p1,p2,p3,tetraCenter));
248 
249  newPoints->SetPoint(i,tetraCenter);
250  newScalars->SetValue(i,(double)tetraRadius);
251  }
252 
253  // compute poles
254  input->BuildLinks();
255  for (i=0; i<input->GetNumberOfPoints(); i++)
256  {
257  cellIds->Initialize();
258  input->GetPointCells(i,cellIds);
259  currentRadius = thicknessScalars->GetValue(i);
260  for (j=0; j<cellIds->GetNumberOfIds(); j++)
261  {
262  id = cellIds->GetId(j);
263  tetraRadius = newScalars->GetValue(id);
264  if (tetraRadius - currentRadius > VTK_VMTK_DOUBLE_TOL)
265  {
266  poleId = id;
267  currentRadius = tetraRadius;
268  }
269  }
270  this->PoleIds->SetId(i,poleId);
271  thicknessScalars->SetValue(i,currentRadius);
272  }
273 
274  this->BuildVoronoiPolys(input,newPolys);
275 
276  if (this->BuildLines)
277  {
278  this->BuildVoronoiLines();
279  }
280 
281  if (this->RadiusArrayName)
282  {
283  newScalars->SetName(this->RadiusArrayName);
284  }
285  else
286  {
287  newScalars->SetName("VoronoiRadius");
288  }
289 
290  output->SetPoints(newPoints);
291  output->GetPointData()->AddArray(newScalars);
292  output->GetPointData()->SetActiveScalars(newScalars->GetName());
293  output->SetPolys(newPolys);
294 
295  if (this->BuildLines)
296  {
297  output->SetLines(newLines);
298  }
299 
300  // Destroy
301  newPoints->Delete();
302  newPolys->Delete();
303  newLines->Delete();
304  newScalars->Delete();
305  thicknessScalars->Delete();
306  cellIds->Delete();
307 
308  return 1;
309 }
