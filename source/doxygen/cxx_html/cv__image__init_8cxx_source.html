282 int Image_DecodeCmd( ClientData clientData, Tcl_Interp *interp,
283  int argc, CONST84 char *argv[] )
284 {
285  char *usage;
286 
287  char *magname;
288  magname = NULL;
289  char *phasename;
290  char *result;
291  double venc,vencscale;
292  int mag_weight_flag = 0;
293 
294  int table_sz = 5;
295  ARG_Entry arg_table[] = {
296  { "-magImage", STRING_Type, &magname, NULL, GDSC_OPTIONAL, 0, { 0 } },
297  { "-phaseImage", STRING_Type, &phasename, NULL, REQUIRED, 0, { 0 } },
298  { "-result", STRING_Type, &result, NULL, REQUIRED, 0, { 0 } },
299  { "-venc", DOUBLE_Type, &venc, NULL, REQUIRED, 0, { 0 } },
300  { "-vencscale", DOUBLE_Type, &vencscale, NULL, REQUIRED, 0, { 0 } },
301  };
302 
303  usage = ARG_GenSyntaxStr( 1, argv, table_sz, arg_table );
304 
305  if ( argc == 1 ) {
306  Tcl_SetResult( interp, usage, TCL_VOLATILE );
307  return TCL_OK;
308  }
309 
310  if ( ARG_ParseTclStr( interp, argc, argv, 1,
311  table_sz, arg_table ) != TCL_OK ) {
312  Tcl_SetResult( interp, usage, TCL_VOLATILE );
313  return TCL_ERROR;
314  }
315 
316  // Do work of command
317 
318  RepositoryDataT type;
319  cvRepositoryData *img;
320  vtkStructuredPoints *vtkspMag, *vtkspPhase;
321 
322  if (magname != NULL) {
323  mag_weight_flag = 1;
324  // Look up given image object:
325  img = gRepository->GetObject( magname );
326  if ( img == NULL ) {
327  Tcl_AppendResult( interp, "couldn't find object ", magname, (char *)NULL );
328  return TCL_ERROR;
329  }
330 
331  // Make sure image is of type STRUCTURED_PTS_T:
332  type = img->GetType();
333  if ( type != STRUCTURED_PTS_T ) {
334  Tcl_AppendResult( interp, "error: object ", magname,
335  "not of type StructuredPts", (char *)NULL );
336  return TCL_ERROR;
337  }
338  // Retrive geometric information:
339  vtkspMag = ((cvStrPts*)img)->GetVtkStructuredPoints();
340  }
341 
342  // Look up given image object:
343  img = gRepository->GetObject( phasename );
344  if ( img == NULL ) {
345  Tcl_AppendResult( interp, "couldn't find object ", phasename, (char *)NULL );
346  return TCL_ERROR;
347  }
348 
349  // Make sure image is of type STRUCTURED_PTS_T:
350  type = img->GetType();
351  if ( type != STRUCTURED_PTS_T ) {
352  Tcl_AppendResult( interp, "error: object ", phasename,
353  "not of type StructuredPts", (char *)NULL );
354  return TCL_ERROR;
355  }
356 
357  // Retrive geometric information:
358  vtkspPhase = ((cvStrPts*)img)->GetVtkStructuredPoints();
359 
360  // Make sure the specified result object does not exist:
361  if ( gRepository->Exists( result ) ) {
362  Tcl_AppendResult( interp, "object ", result, " already exists",
363  (char *)NULL );
364  return TCL_ERROR;
365  }
366 
367  vtkStructuredPoints *obj;
368  int status;
369  if (mag_weight_flag == 0) {
370  status = mr_decode (vtkspPhase,venc,vencscale,&obj);
371  } else {
372  status = mr_decode_masked (vtkspMag,vtkspPhase,venc,vencscale,&obj);
373  }
374 
375  if ( status == CV_ERROR ) {
376  Tcl_AppendResult( interp, "Problem decoding ", magname, " and ",phasename,(char *)NULL );
377  return TCL_ERROR;
378  }
379 
380  cvStrPts *sp = new cvStrPts( obj );
381  sp->SetName( result );
382  if ( !( gRepository->Register( sp->GetName(), sp ) ) ) {
383  Tcl_AppendResult( interp, "error registering obj ", result,
384  " in repository", (char *)NULL );
385  delete sp;
386  return TCL_ERROR;
387  }
388 
389  Tcl_SetResult( interp, sp->GetName(), TCL_VOLATILE );
390  return TCL_OK;
391 
392 }

399 int Image_CalcCorrectionEqnCmd( ClientData clientData, Tcl_Interp *interp,
400  int argc, CONST84 char *argv[] )
401 {
402  char *usage;
403  int i;
404  ARG_List regionsArg;
405  ARG_List imagesArg;
406  int order = 0;
407  RepositoryDataT type;
408 
409  int table_sz = 3;
410  ARG_Entry arg_table[] = {
411  { "-regions", LIST_Type, &regionsArg, NULL, REQUIRED, 0, { 0 } },
412  { "-images", LIST_Type, &imagesArg, NULL, REQUIRED, 0, { 0 } },
413  { "-order", INT_Type, &order, NULL, REQUIRED, 0, { 0 } },
414  };
415 
416  usage = ARG_GenSyntaxStr( 1, argv, table_sz, arg_table );
417 
418  if ( argc == 1 ) {
419  Tcl_SetResult( interp, usage, TCL_VOLATILE );
420  return TCL_OK;
421  }
422 
423  if ( ARG_ParseTclStr( interp, argc, argv, 1,
424  table_sz, arg_table ) != TCL_OK ) {
425  Tcl_SetResult( interp, usage, TCL_VOLATILE );
426  return TCL_ERROR;
427  }
428 
429  // Do work of command
430 
431  // check for valid order
432  if (order < 0 || order > 3) {
433  Tcl_AppendResult( interp, "order must be 0,1, or 2", (char *)NULL );
434  ARG_FreeListArgvs( table_sz, arg_table );
435  return TCL_ERROR;
436  }
437 
438  int numRegions = regionsArg.argc;
439  int numImages = imagesArg.argc;
440 
441  if (numRegions == 0) {
442  Tcl_AppendResult( interp, "empty list of regions", (char *)NULL );
443  ARG_FreeListArgvs( table_sz, arg_table );
444  return TCL_ERROR;
445  }
446 
447  if (numImages == 0) {
448  Tcl_AppendResult( interp, "empty list of images", (char *)NULL );
449  ARG_FreeListArgvs( table_sz, arg_table );
450  return TCL_ERROR;
451  }
452 
453 
454  // find the corresponding repository objects to each name
455  cvRepositoryData *pd,*img;
456  vtkPolyData **listPd = new vtkPolyData* [numRegions];
457  vtkStructuredPoints **listImg = new vtkStructuredPoints* [numImages];
458 
459  for (i = 0; i < numRegions; i++) {
460  if (regionsArg.argv[i] != NULL) {
461  pd = gRepository->GetObject( regionsArg.argv[i] );
462  if ( pd == NULL ) {
463  Tcl_AppendResult( interp, "couldn't find object ", regionsArg.argv[i], (char *)NULL );
464  delete [] listPd;
465  delete [] listImg;
466  ARG_FreeListArgvs( table_sz, arg_table );
467  return TCL_ERROR;
468  }
469 
470  // Make sure region is of type POLYDATA_T:
471  type = pd->GetType();
472  if ( type != POLY_DATA_T ) {
473  Tcl_AppendResult( interp, "error: object ", regionsArg.argv[i],
474  "not of type cvPolyData", (char *)NULL );
475  delete [] listPd;
476  delete [] listImg;
477  ARG_FreeListArgvs( table_sz, arg_table );
478  return TCL_ERROR;
479  }
480  listPd[i] = ((cvPolyData*)pd)->GetVtkPolyData();
481  } else {
482  Tcl_AppendResult( interp, "NULL region pointer encountered ", (char *)NULL );
483  delete [] listPd;
484  delete [] listImg;
485  ARG_FreeListArgvs( table_sz, arg_table );
486  return TCL_ERROR;
487  }
488  }
489 
490  // find the corresponding repository objects to each name
491  for (i = 0; i < numImages; i++) {
492  if (imagesArg.argv[i] != NULL) {
493  img = gRepository->GetObject( imagesArg.argv[i] );
494  if ( img == NULL ) {
495  Tcl_AppendResult( interp, "couldn't find object ", imagesArg.argv[i], (char *)NULL );
496  delete [] listPd;
497  delete [] listImg;
498  ARG_FreeListArgvs ( table_sz, arg_table );
499  return TCL_ERROR;
500  }
501 
502  // Make sure image is of type POLYDATA_T:
503  type = img->GetType();
504  if ( type != STRUCTURED_PTS_T ) {
505  Tcl_AppendResult( interp, "error: object ", imagesArg.argv[i],
506  "not of type StructuredPts", (char *)NULL );
507  delete [] listPd;
508  delete [] listImg;
509  ARG_FreeListArgvs ( table_sz, arg_table );
510  return TCL_ERROR;
511  }
512  listImg[i] = ((cvStrPts*)img)->GetVtkStructuredPoints();
513 
514  } else {
515  Tcl_AppendResult( interp, "NULL image pointer encountered ", (char *)NULL );
516  delete [] listPd;
517  delete [] listImg;
518  ARG_FreeListArgvs ( table_sz, arg_table );
519  return TCL_ERROR;
520  }
521  }
522 
523  // debug info
524  for (i = 0; i < numRegions; i++) {
525  fprintf(stdout,"region %i: %s %i\n",i,regionsArg.argv[i],listPd[i]);
526  }
527  for (i = 0; i < numImages; i++) {
528  fprintf(stdout,"images %i: %s %i\n",i,imagesArg.argv[i],listImg[i]);
529  }
530 
531  // classify points and calculate correction equation
532  double results[6];
533  int status = img_calcCorrectionEqn(numRegions,listPd,numImages,listImg,order,results);
534 
535  // clean up
536  delete [] listPd;
537  delete [] listImg;
538  ARG_FreeListArgvs( table_sz, arg_table );
539 
540  if ( status == CV_ERROR ) {
541  Tcl_AppendResult( interp, "error finding correction equation ", (char *)NULL );
542  return TCL_ERROR;
543  }
544 
545  // return a string with the correction equation
546  char r[2048];
547  r[0] = '\0';
548  if (order == 0) {
549  sprintf(r,"%le",results[0]);
550  } else if (order == 1) {
551  sprintf(r,"%le %s %le %s %le %s",results[0],
552  " + ", results[1], "*$x + ", results[2], "*$y");
553  } else if (order == 2) {
554  sprintf(r,"%le %s %le %s %le %s %le %s %le %s %le %s",
555  results[0]," + ",results[1],"*$x + ",results[2],
556  "*$y + ",results[3],"*$x*$x + ",results[4],
557  "*$y*$y + ",results[5],"*$x*$y");
558  }
559  fprintf(stdout,r);
560  fprintf(stdout,"\n");
561 
562  Tcl_AppendResult( interp, r, (char *)NULL );
563  return TCL_OK;
564 
565 }
