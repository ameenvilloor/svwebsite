212 long cvAdaptCore::FindCubicRoots(const double coeff[4], double x[3])
213 {
214  double a1 = coeff[2] / coeff[3];
215  double a2 = coeff[1] / coeff[3];
216  double a3 = coeff[0] / coeff[3];
217 
218  if( ABS(a3)<1.0e-8 )
219  return FindQuadraticRoots(a1,a2,x);
220 
221  double Q = (a1 * a1 - 3 * a2) / 9.;
222  double R = (2. * a1 * a1 * a1 - 9. * a1 * a2 + 27. * a3) / 54.;
223  double Qcubed = Q * Q * Q;
224  double d = Qcubed - R * R;
225 
226  // printf ("d = %22.15e Q = %12.5E R = %12.5E Qcubed %12.5E\n",d,Q,R,Qcubed);
227 
229  if(Qcubed == 0.0 || fabs ( Qcubed - R * R ) < 1.e-8 * (fabs ( Qcubed) + fabs( R * R)) )
230  {
231  double theta;
232  if (Qcubed <= 0.0)theta = acos(1.0);
233  else if (R / sqrt(Qcubed) > 1.0)theta = acos(1.0);
234  else if (R / sqrt(Qcubed) < -1.0)theta = acos(-1.0);
235  else theta = acos(R / sqrt(Qcubed));
236  double sqrtQ = sqrt(Q);
237  // printf("sqrtQ = %12.5E teta=%12.5E a1=%12.5E\n",sqrt(Q),theta,a1);
238  x[0] = -2 * sqrtQ * cos( theta / 3) - a1 / 3;
239  x[1] = -2 * sqrtQ * cos((theta + 2 * M_PI) / 3) - a1 / 3;
240  x[2] = -2 * sqrtQ * cos((theta + 4 * M_PI) / 3) - a1 / 3;
241  return (3);
242  }
243 
244  /* Three real roots */
245  if (d >= 0.0) {
246  double theta = acos(R / sqrt(Qcubed));
247  double sqrtQ = sqrt(Q);
248  x[0] = -2 * sqrtQ * cos( theta / 3) - a1 / 3;
249  x[1] = -2 * sqrtQ * cos((theta + 2 * M_PI) / 3) - a1 / 3;
250  x[2] = -2 * sqrtQ * cos((theta + 4 * M_PI) / 3) - a1 / 3;
251  return (3);
252  }
253 
254  /* One real root */
255  else {
256  printf("IMPOSSIBLE !!!\n");
257 
258  double e = pow(sqrt(-d) + fabs(R), 1. / 3.);
259  if (R > 0)
260  e = -e;
261  x[0] = (e + Q / e) - a1 / 3.;
262  return (1);
263  }
264 }

267 long cvAdaptCore::NullSpace(const double *a, double *result, double eps, long n)
268 {
269  int r[MAXN], c[MAXN];
270  register long i, j, k;
271  int jj, kk, t;
272  double max, temp;
273  int ec;
274 
275  for (i = 0; i < n; i++)
276  r[i] = c[i] = -1; /* Reset row and column pivot indices */
277 
278  // copy the input matrix if not in place
279  if (result != a)
280  for (i = 0; i < n*n; i++)
281  result[i] = a[i];
282  // rest of algorithm is in place wrt result[]
283 
284  for (i = 0; i < n; i++) {
285  /* Find the biggest element in the remaining submatrix
286  * for the next full pivot.
287  */
288  max = 0.0;
289  for (k = 0; k < n; k++) {
290  if (r[k] < 0) {
291  for (j = 0; j < n; j++) {
292  if ((c[j] < 0) && ((temp = fabs(R(k, j))) > max)) {
293  kk = k;
294  jj = j;
295  max = temp;
296  }
297  }
298  }
299  }
300  if (max < eps)
301  break; /* Consider this and all subsequent pivots to be zero */
302 
303  c[jj] = kk; /* The row */
304  r[kk] = jj; /* and column of the next pivot */
305 
306  temp = 1.0 / R(kk, jj);
307  R(kk, jj) = 1.0;
308  for (j = 0; j < n; j++) /* Should this be for j != jj ? */
309  R(kk, j) *= temp; /* Row equilibration */
310 
311  for (k = 0; k < n; k++) { /* Row elimination */
312  if (k == kk)
313  continue; /* Don't do a thing to the pivot row */
314  temp = R(k, jj);
315  R(k, jj) = 0.0;
316  for (j = 0; j < n; j++) {
317  R(k, j) -= temp * R(kk, j); /* Subtract row kk from row k */
318  if (fabs(R(k, j)) < eps)
319  R(k, j) = 0.0; /* Flush to zero if too small */
320  }
321  }
322  }
323 
324  /* Sort into a truncated triangular matrix */
325  for (j = 0; j < n; j++) { /* For all columns... */
326  while ((c[j] >= 0) && (j != c[j])) {
327  for (k = 0; k < n; k++) {
328  if (r[k] < 0) {
329  /* Aha! a null column vector */
330  temp = R(k, j); /* Get it on top */
331  R(k, j) = R(k, c[j]);
332  R(k, c[j]) = temp;
333  }
334  }
335  t = c[j]; /* Twiddle until pivots are on the diagonal */
336  c[j] = c[t];
337  c[t] = t;
338  }
339  }
340 
341  /* Copy the null space vectors into the top of the A matrix */
342  ec = 0;
343  for (k = 0; k < n; k++) {
344  if (r[k] < 0) {
345  R(k, k) = 1.0; /* Set the pivot equal to 1 */
346  if (ec != k) {
347  for (j = 0; j < n; j++) {
348  R(ec, j) = R(k, j);
349  }
350  }
351  ec++;
352  }
353  }
354  /* The first ec rows of the matrix a are the vectors which are
355  * orthogonal to the columns of the matrix a.
356  */
357  return (ec);
358 }
