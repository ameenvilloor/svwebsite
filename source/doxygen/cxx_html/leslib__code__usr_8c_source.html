144  Real* pnt ; /* pointer */
145 
146 /*---------------------------------------------------------------------------
147  * Get the head of the memory
148  *---------------------------------------------------------------------------
149  */
150  if ( id == LES_RES_PNT ) {
151 
152  pnt = usrHd->resf ;
153  id = 0 ;
154 
155  } else if ( id == LES_SOL_PNT ) {
156 
157  pnt = usrHd->solinc ;
158  id = 0 ;
159 
160  } else if ( id < 0 ) {
161 
162  pnt = usrHd->aperm ;
163  id = id + usrHd->nPermDims ;
164 
165  } else {
166 
167  pnt = usrHd->atemp ;
168  id = id ;
169 
170  }
171 /*---------------------------------------------------------------------------
172  * Get the offset
173  *---------------------------------------------------------------------------
174  */
175  pnt = pnt + (id + offset) * usrHd->nNodes ;
176 
177 /*---------------------------------------------------------------------------
178  * Return the pointer
179  *---------------------------------------------------------------------------
180  */
181  return( pnt ) ;

286  Real* aperm,
287  Integer* nshg,
288  Integer* myrank,
289  Integer* lstep,
290  Integer* nPermDims ) {
291 
292  int nPrjs, PrjSrcId;
293  int nPresPrjs, PresPrjSrcId;
294  char filename[255];
295  int fileHandle=0;
296  int iarray[3];
297  int size, nitems;
298  double* projVec;
299  int i, j, count;
300 
301  sprintf( filename,"restart.%d.%d", *lstep, *myrank+1 );
302  openfile_( filename, "append", &fileHandle );
303 
304  nPrjs = (Integer) lesGetPar( lesArray[ *lesId ], LES_ACT_PRJS );
305  PrjSrcId = (Integer) lesGetPar( lesArray[ *lesId ], LES_PRJ_VEC_ID );
306 
307  if ( PrjSrcId < 0 ) PrjSrcId += *nPermDims;
308 
309  projVec = (double*)malloc( nPrjs * ( *nshg ) * sizeof( double ) );
310 
311  count = 0;
312  for( i = PrjSrcId; i < PrjSrcId+nPrjs; i ++ ) {
313  for( j = 0 ; j < *nshg; j++ ) {
314  projVec[ count++ ] = aperm[ (*nshg) * i + j ];
315  }
316  }
317 
318  iarray[ 0 ] = *nshg;
319  iarray[ 1 ] = nPrjs;
320  nitems = 2;
321  size = (*nshg)*nPrjs;
322 
323  writeheader_( &fileHandle, "projection vectors ", (void*)iarray,
324  &nitems, &size, "double", cvsolver_iotype );
325  nitems = size;
326  writedatablock_( &fileHandle, "projection vectors ", (void*)projVec,
327  &nitems, "double", cvsolver_iotype );
328  free(projVec);
329 
330  /*************************************************************************/
331 
332  nPresPrjs = (Integer) lesGetPar( lesArray[ *lesId ], LES_ACT_PRES_PRJS );
333  PresPrjSrcId =(Integer)lesGetPar( lesArray[ *lesId ], LES_PRES_PRJ_VEC_ID );
334  if ( PresPrjSrcId < 0 ) PresPrjSrcId += *nPermDims;
335 
336  projVec = (double*)malloc( nPresPrjs * ( *nshg ) * sizeof( double ) );
337 
338  count = 0;
339  for( i = PresPrjSrcId; i < (PresPrjSrcId + nPresPrjs) ; i ++ ) {
340  for( j = 0 ; j < *nshg; j++ ) {
341  projVec[ count++ ] = aperm[ (*nshg) * i + j ];
342  }
343  }
344 
345  iarray[ 0 ] = *nshg;
346  iarray[ 1 ] = nPresPrjs;
347  nitems = 2;
348  size = (*nshg)*nPresPrjs;
349 
350  writeheader_( &fileHandle, "pressure projection vectors ", (void*)iarray,
351  &nitems, &size, "double", cvsolver_iotype );
352  nitems = size;
353 
354  writedatablock_( &fileHandle, "pressure projection vectors ",
355  (void*)projVec, &nitems, "double", cvsolver_iotype );
356  free( projVec);
357 
358  closefile_( &fileHandle, "append" );
359 }
