160  // Hash table for band definition:
161  // ---
162  // Note that this hash table is implemented as an *array* of the
163  // template class SortedList from the Berkeley list package. As
164  // such, the nodeTable_ array pointer has no sorting associated with
165  // it... only the items of each array element (i.e. the items of
166  // each SortedList) are sorted.
167  // ---
168  // In choosing a hash function for the table, we need to consider
169  // the fact that we will need to search the table by (i,j,k) logical
170  // node indices. This search will be done several times for each
171  // sparse node as part of the cvLevelSetSparseGrid::ConstructNeighborInfo
172  // method. Some alternatives:
173  //
174  // 1. h = (i+j) % n; --> potentially many (i,j)'s per bucket
175  // 2. h = (j * iDim) + i; --> one (i,j) per bucket
176  //
177  // Once we've stored all the entries and are actually ready to
178  // search the table, if we use (1), we'll have a more compact table,
179  // but we'll have to do a more complicated search within the bucket
180  // (i.e. SortedList). If we use (2), we can do simple sorting /
181  // searching within buckets based on k, but there could be many many
182  // empty buckets in the table.
183  // ---

284 // -------------
285 // ComputeHashIx
286 // -------------
287 // Note that this hash fn provides the needed functionality that
288 // in-order traversal of (i,j,k)'s leads to a monotonic traversal of
289 // hash table buckets. This is the critical characteristic which
290 // enables node-number-maps to be used efficiently (i.e. forward
291 // mappings in O(log n) instead of O(n)).
292 
293 inline
294 int cvLevelSetSparseGrid::ComputeHashIx( int i, int j, int k )
295 {
296  return ( IJKToDenseIx( i, j, k ) % numBuckets_ );
297  return k;
298  return ( (j*I_) + i );
299 }
300 
301 
302 // ------------
303 // IJKToDenseIx
304 // ------------
305 
306 inline
307 int cvLevelSetSparseGrid::IJKToDenseIx( int i, int j, int k )
308 {
309  return ( (k*I_*J_) + (j*I_) + i );
310 }
