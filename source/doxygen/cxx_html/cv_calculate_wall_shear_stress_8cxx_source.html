102 int cvCalculateWallShearStress::CalcWallShearFromStresses() {
103 
104  int i;
105 
106  int numPts = surfaceMesh_->GetNumberOfPoints();
107  fprintf(stdout,"numPts: %i\n",numPts);
108  // get the normals
109  vtkDataArray* normals = surfaceMesh_->GetPointData()->GetNormals();
110  // get the stress tensors
111  vtkDataArray* stresses = tensors_->GetPointData()->GetTensors();
112 
113  // figure out which nodes are on surface
114  int* ptOnSurf = new int[numPts];
115 
116  for (i = 0; i < numPts; i++) {
117  ptOnSurf[i] = 0;
118  }
119 
120  vtkCellArray* polys = surfaceMesh_->GetPolys();
121  int numCells = polys->GetNumberOfCells();
122  fprintf(stdout,"numCells: %i\n",numCells);
123 
124  polys->InitTraversal();
125  vtkIdType celltype = 0;
126  vtkIdType *ids;
127  for (i = 0; i < numCells; i++) {
128  polys->GetNextCell(celltype,ids);
129  //fprintf(stdout,"checking cell: %i type: %i\n",i,celltype);
130  if (celltype != 3) {
131  return CV_ERROR;
132  }
133  ptOnSurf[ids[0]] = 1;
134  ptOnSurf[ids[1]] = 1;
135  ptOnSurf[ids[2]] = 1;
136  }
137 
138  int numPtsOnSurf = 0;
139  for (i = 0; i < numPts; i++) {
140  if (ptOnSurf[i] != 0) {
141  numPtsOnSurf++;
142  }
143  }
144  fprintf(stdout,"it appears %i pts are on the surface out of %i\n",numPtsOnSurf,numPts);
145 
146  // create return vtk vector
147  wallshear_ = vtkFloatingPointArrayType::New();
148  wallshear_->SetNumberOfComponents(3);
149  wallshear_->Allocate(numPts,10000);
150  wallshear_->Initialize();
151 
152  vtkFloatingPointType nrm[3];
153  vtkFloatingPointType stress[9];
154  double t[3];
155  double t_dot_n = 0.0;
156  double twall[3];
157 
158  if (stresses->GetNumberOfComponents() != 9) {
159  wallshear_->Delete();
160  return CV_ERROR;
161  }
162 
163  for (i = 0; i < numPts; i++) {
164  if (ptOnSurf[i] == 0) {
165  wallshear_->InsertNextTuple3(0.0,0.0,0.0);
166  } else {
167  // get outward normal
168  normals->GetTuple(i,nrm);
169  stresses->GetTuple(i,stress);
170  t[0] = stress[0]*nrm[0]+stress[1]*nrm[1]+stress[2]*nrm[2];
171  t[1] = stress[3]*nrm[0]+stress[4]*nrm[1]+stress[5]*nrm[2];
172  t[2] = stress[6]*nrm[0]+stress[7]*nrm[1]+stress[8]*nrm[2];
173  t_dot_n = t[0]*nrm[0]+t[1]*nrm[1]+t[2]*nrm[2];
174  twall[0] = t[0] - t_dot_n*nrm[0];
175  twall[1] = t[1] - t_dot_n*nrm[1];
176  twall[2] = t[2] - t_dot_n*nrm[2];
177  wallshear_->InsertNextTuple3(twall[0],twall[1],twall[2]);
178  }
179  }
180 
181  return CV_OK;
182 
183 }

279 cvPolyData* cvCalculateWallShearStress::CalcWallShearMean(int numPds, cvPolyData **shearPds) {
280 
281  int i = 0;
282  int j = 0;
283  vtkFloatingPointType shear[3];
284 
285  fprintf(stdout,"numPds: %i\n",numPds);
286 
287  // all of the shear pds must have the same num pts
288  int numPts = shearPds[0]->GetVtkPolyData()->GetNumberOfPoints();
289  fprintf(stdout,"numPts: %i\n",numPts);
290 
291  // create a list of the shear vectors
292  vtkDataArray **vectors = new vtkDataArray*[numPds];
293  for (i = 0; i < numPds; i++) {
294  vectors[i]=shearPds[i]->GetVtkPolyData()->GetPointData()->GetVectors();
295  }
296 
297  // create return vtk vector
298  vtkFloatingPointArrayType *shearmean = vtkFloatingPointArrayType::New();
299  shearmean->SetNumberOfComponents(1);
300  shearmean->Allocate(numPts,10000);
301  shearmean->Initialize();
302 
303  for (i = 0; i < numPts; i++) {
304  double v0 = 0.0;
305  double v1 = 0.0;
306  double v2 = 0.0;
307  for (j = 0; j < numPds; j++) {
308  vectors[j]->GetTuple(i,shear);
309  v0 += shear[0];v1 += shear[1];v2 += shear[2];
310  }
311  v0 = v0/numPds; v1 = v1/numPds; v2 = v2/numPds;
312  vtkFloatingPointType mag = sqrt(v0*v0+v1*v1+v2*v2);
313  shearmean->InsertNextTuple1(mag);
314  }
315 
316  // create cvPolyData object to return
317  vtkPolyData* pd = vtkPolyData::New();
318  if (surfaceMesh_ == NULL) {
319  pd->SetPoints(shearPds[0]->GetVtkPolyData()->GetPoints());
320  } else {
321  pd->CopyStructure(surfaceMesh_);
322  }
323  pd->GetPointData()->SetScalars(shearmean);
324  cvPolyData* reposobj = new cvPolyData(pd);
325  return reposobj;

377 
378  int i = 0;
379  vtkFloatingPointType mean = 0.0;
380  vtkFloatingPointType pulse = 0.0;
381 
382  // all of the shear pds must have the same num pts
383  int numPts = shearMean->GetVtkPolyData()->GetNumberOfPoints();
384  fprintf(stdout,"numPts: %i\n",numPts);
385 
386  // create a list of the shear vectors
387  vtkDataArray *meanscalars = shearMean->GetVtkPolyData()->GetPointData()->GetScalars();
388  vtkDataArray *pulsescalars = shearPulse->GetVtkPolyData()->GetPointData()->GetScalars();
389 
390  // create return vtk vector
391  vtkFloatingPointArrayType *osiScalars = vtkFloatingPointArrayType::New();
392  osiScalars->SetNumberOfComponents(1);
393  osiScalars->Allocate(numPts,10000);
394  osiScalars->Initialize();
395 
396  for (i = 0; i < numPts; i++) {
397  vtkFloatingPointType osi = 0.0;
398  mean = 0;
399  pulse = 0;
400  meanscalars->GetTuple(i,&mean);
401  pulsescalars->GetTuple(i,&pulse);
402  if (pulse <= 0.00001) {
403  //fprintf(stderr,"warning: pulse set to 0.\n");
404  osi = 0;
405  } else {
406  osi = 1.0/2.0*(1-mean/pulse);
407  }
408  osiScalars->InsertNextTuple1(osi);
409  }
410 
411  // create cvPolyData object to return
412  vtkPolyData* pd = vtkPolyData::New();
413  if (surfaceMesh_ == NULL) {
414  pd->SetPoints(shearMean->GetVtkPolyData()->GetPoints());
415  } else {
416  pd->CopyStructure(surfaceMesh_);
417  }
418  pd->GetPointData()->SetScalars(osiScalars);
419  cvPolyData* reposobj = new cvPolyData(pd);
420  return reposobj;

462  double v2 = 0.0;

474

477  delete scalars;
478

482  pd->SetPoints(inputPds[0]->GetVtkPolyData()->GetPoints());
483  } else {
484  pd->CopyStructure(surfaceMesh_);
485  }
486  pd->GetPointData()->SetScalars(avgscalar);
487  pd->GetPointData()->SetVectors(avgvec);
488  cvPolyData* reposobj = new cvPolyData(pd);
489  return reposobj;
