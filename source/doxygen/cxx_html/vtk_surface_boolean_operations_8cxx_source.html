75 void vtk2gts (vtkPolyData * input, GtsSurface * output)
76 {
77  GtsVertex ** vertices;
78 
79  g_return_if_fail(output != NULL);
80  if (g_hash_table_size(output->faces) != 0)
81  g_warning("overwriting surface");
82  input->BuildLinks();
83 
84  vertices = (GtsVertex **)
85  g_malloc(input->GetNumberOfPoints() * sizeof(GtsVertex *));
86  for (vtkIdType i = 0; i < input->GetNumberOfPoints(); ++i) {
87  vtkFloatingPointType x[3];
88  input->GetPoint(i, x);
89  vertices[i] = gts_vertex_new(output->vertex_class, x[0], x[1], x[2]);
90  }
91  for (vtkIdType i = 0; i < input->GetNumberOfCells(); ++i) {
92  vtkIdType npts, * pts;
93  input->GetCellPoints(i, npts, pts);
94  if (npts != 3) {
95  g_warning("skipping no triangle cell");
96  continue;
97  }
98  GtsEdge * newedges[3];
99  for (gint j = 0; j < 3; ++j) {
100  vtkIdType v1 = pts[j], v2 = pts[(j+1) % 3];
101  newedges[j] = gts_edge_new(output->edge_class,
102  vertices[v1], vertices[v2]);
103  if (gts_segment_is_duplicate(GTS_SEGMENT(newedges[j]))) {
104  GtsSegment * tmp = gts_segment_is_duplicate(GTS_SEGMENT(newedges[j]));
105  gts_object_destroy(GTS_OBJECT(newedges[j]));
106  newedges[j] = GTS_EDGE(tmp);
107  g_return_if_fail(gts_segment_is_duplicate(GTS_SEGMENT(newedges[j])) ==
108  NULL);
109  }
110  }
111  gts_surface_add_face(output,
112  gts_face_new(output->face_class,
113  newedges[0], newedges[1], newedges[2]));
114  }
115  g_free(vertices);
116 }

132 void gts2vtk (GtsSurface * input, vtkPolyData * output)
133 {
134  vtkNew(vtkPoints,points);
135  vtkNew(vtkCellArray,polys);
136  gts_surface_foreach_vertex(input, (GtsFunc) &gts2vtk_insert_point, points);
137  gts_surface_foreach_face(input, (GtsFunc) &gts2vtk_insert_triangle, polys);
138  gts_surface_foreach_vertex(input, (GtsFunc) gts_object_reset_reserved, NULL);
139  output->SetPoints(points);
