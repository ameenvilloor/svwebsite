172 static int itkls3d_NewCmd( CXX_TCL_STDARGS )
173 {
174 
175  CONST84 char *lsName;
176  cvITKLevelSetBase<ImageType> *ls;
177  Tcl_HashEntry *entryPtr;
178  int newEntry = 0;
179 
180 
181  // Check syntax:
182  if (argc != 2) {
183  Tcl_AppendResult( interp, "usage: ", argv[0], " <objName>",
184  (char *)NULL );
185  return TCL_ERROR;
186  }
187 
188  // Make sure this is a new object name:
189  lsName = argv[1];
190  if ( !NewName( lsName ) ) {
191  Tcl_AppendResult( interp, "ITKLevelSetCore object \"", lsName,
192  "\" already exists", (char *)NULL );
193  return TCL_ERROR;
194  }
195 
196  // Allocate new cvLevelSet object:
197  ls = new cvITKLevelSetBase<ImageType>;
198  if ( ls == NULL ) {
199  Tcl_AppendResult( interp, "error allocating object \"", lsName,
200  "\"", (char *)NULL );
201  return TCL_ERROR;
202  }
203 
204  strcpy( ls->tclName_, lsName );
205  entryPtr = Tcl_CreateHashEntry( &gLsetCoreTable, lsName, &newEntry );
206  if ( !newEntry ) {
207  Tcl_SetResult( interp, "error updating cvLevelSet hash table",
208  TCL_STATIC );
209  delete ls;
210  return TCL_ERROR;
211  }
212  Tcl_SetHashValue( entryPtr, (ClientData)ls );
213  char lsNameStr[2048];
214  lsNameStr[0]='\0';
215  sprintf(lsNameStr,"%s",lsName);
216  Tcl_SetResult( interp, lsNameStr, TCL_VOLATILE );
217  Tcl_CreateCommand( interp, Tcl_GetStringResult(interp), itkls3d_ObjectCmd,
218  (ClientData)ls, Deleteitkls3d );
219  return TCL_OK;
220 }
