82 // The following class defines a "list element" -- which is
83 // used to keep track of one item on a list. It is equivalent to a
84 // LISP cell, with a "car" ("next") pointing to the next element on the list,
85 // and a "cdr" ("item") pointing to the item on the list.
86 //
87 // This class is private to this module (and classes that inherit
88 // from this module). Made public for notational convenience.
89 
90 template <class T>
91 class cvLispListElement {
92  public:
93  cvLispListElement(T itm); // initialize a list element
94  cvLispListElement *next; // next element on list, NULL if this is last
95  T item; // item on the list
96 };

98 // The following class defines a "list" -- a singly linked list of
99 // list elements, each of which points to a single item on the list.
100 // The class has been tested only for primitive types (ints, pointers);
101 // no guarantees it will work in general. For instance, all types
102 // to be inserted into a list must have a "==" operator defined.
103 
104 template <class T>
105 class cvLispList {
106  public:
107  cvLispList(); // initialize the list
108  virtual ~cvLispList(); // de-allocate the list
109 
110  virtual void Prepend(T item);// Put item at the beginning of the list
111  virtual void Append(T item); // Put item at the end of the list
112 
113  T Front() { return first->item; }
114  // Return first item on list
115  // without removing it
116  T RemoveFront(); // Take item off the front of the list
117  void Remove(T item); // Remove specific item from list
118 
119  Bool IsInList(T item) const;// is the item in the list?
120 
121  unsigned int NumInList() { return numInList;};
122  // how many items in the list?
123  Bool IsEmpty() { return (numInList == 0); };
124  // is the list empty?
125 
126  void Apply(void (*f)(T)) const;
127  // apply function to all elements in list
128 
129  virtual void SanityCheck() const;
130  // has this list been corrupted?
131  void SelfTest(T *p, int numEntries);
132  // verify module is working
133 
134  protected:
135  cvLispListElement<T> *first; // Head of the list, NULL if list is empty
136  cvLispListElement<T> *last; // Last element of list
137  int numInList; // number of elements in list
138 
139 friend class cvLispListIterator<T>;
140 };

142 // The following class defines a "sorted list" -- a singly linked list of
143 // list elements, arranged so that "Remove" always returns the smallest
144 // element.
145 // All types to be inserted onto a sorted list must have a "Compare"
146 // function defined:
147 // int Compare(T x, T y)
148 // returns -1 if x < y
149 // returns 0 if x == y
150 // returns 1 if x > y
151 
152 template <class T>
153 class cvSortedList : public cvLispList<T> {
154  public:
155  cvSortedList(int (*comp)(T x, T y)) : cvLispList<T>() { compare = comp;};
156  ~cvSortedList() {}; // base class destructor called automatically
157 
158  void Insert(T item); // insert an item onto the list in sorted order
159 
160  void SanityCheck() const; // has this list been corrupted?
161  void SelfTest(T *p, int numEntries);
162  // verify module is working
163 
164  private:
165  int (*compare)(T x, T y); // function for sorting list elements
166 
167  void Prepend(T item) { Insert(item); } // *pre*pending has no meaning
168  // in a sorted list
169  void Append(T item) { Insert(item); } // neither does *ap*pend
170 
171 };

173 // The following class can be used to step through a list.
174 // Example code:
175 // cvLispListIterator<T> *iter(list);
176 //
177 // for (; !iter->IsDone(); iter->Next()) {
178 // Operation on iter->Item()
179 // }

181 template <class T>
182 class cvLispListIterator {
183  public:
184  cvLispListIterator(cvLispList<T> *list) { current = list->first; }
185  // initialize an iterator
186 
187  Bool IsDone() { return current == NULL; };
188  // return TRUE if we are at the end of the list
189 
190  T Item() { Assert(!IsDone()); return current->item; };
191  // return current element on list
192 
193  void Next() { current = current->next; };
194  // update iterator to point to next
195 
196  private:
197  cvLispListElement<T> *current; // where we are in the list
198 };
