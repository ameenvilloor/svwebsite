232  {
233  vtkIdType cellId0 = node0->Cells->GetId(id0);
234  int type0 = mesh0->GetCellType(cellId0);
235 
236  //Make sure the cell is a triangle
237  if (type0 == VTK_TRIANGLE)
238  {
239  vtkIdType npts0, *triPtIds0;
240  mesh0->GetCellPoints(cellId0, npts0, triPtIds0);
241  double triPts0[3][3];
242  for (vtkIdType id = 0; id < npts0; id++)
243  {
244  mesh0->GetPoint(triPtIds0[id], triPts0[id]);
245  }
246 
247  if (obbTree1->TriangleIntersectsNode
248  (node1, triPts0[0], triPts0[1], triPts0[2], transform))
249  {
250  int numCells1 = node1->Cells->GetNumberOfIds();
251  for (vtkIdType id1 = 0; id1 < numCells1; id1++)
252  {
253  vtkIdType cellId1 = node1->Cells->GetId(id1);
254  int type1 = mesh1->GetCellType(cellId1);
255  if (type1 == VTK_TRIANGLE)
256  {
257  // See if the two cells actually intersect. If they do,
258  // add an entry into the intersection maps and add an
259  // intersection line.
260  vtkIdType npts1, *triPtIds1;
261  mesh1->GetCellPoints(cellId1, npts1, triPtIds1);
262 
263  double triPts1[3][3];
264  for (vtkIdType id = 0; id < npts1; id++)
265  {
266  mesh1->GetPoint(triPtIds1[id], triPts1[id]);
267  }
268 
269  int coplanar = 0;
270  double outpt0[3], outpt1[3];
271  double surfaceid[2];
272  int intersects =
273  vtkIntersectionPolyDataFilter2::TriangleTriangleIntersection
274  (triPts0[0], triPts0[1], triPts0[2],
275  triPts1[0], triPts1[1], triPts1[2],
276  coplanar, outpt0, outpt1, surfaceid);
277 
278  if ( coplanar )
279  {
280  // Coplanar triangle intersection is not handled.
281  // This intersection will not be included in the output. TODO
282  //std::cout<<"Coplanar"<<endl;
283  intersects = 0;
284  continue;
285  }
286 
287  //If actual intersection, add point and cell to edge, line,
288  //and surface maps!
289  if ( intersects)
290  {
291  vtkIdType lineId = intersectionLines->GetNumberOfCells();
292 
293  vtkIdType ptId0, ptId1;
294  int unique[2];
295  vtkIdType check1,check2;
296  check1 = pointMerger->IsInsertedPoint(outpt0);
297  check2 = pointMerger->IsInsertedPoint(outpt1);
298  unique[0] = pointMerger->InsertUniquePoint(outpt0, ptId0);
299  unique[1] = pointMerger->InsertUniquePoint(outpt1, ptId1);
300 
301  int addline = 1;
302  if (ptId0 == ptId1)
303  addline = 0;
304 
305  //std::cout<<"Both Points are the same! "<<ptId0<<endl;
306 //std::cout<<"TELL ME IMPORTANT "<<surfaceid[0]<<" "<<surfaceid[1]<<endl;
307  if (ptId0 == ptId1 && surfaceid[0] != surfaceid[1])
308  intersectionSurfaceId->InsertValue(ptId0,3);
309 
310  else
311  {
312  //std::cout<<"Surface Id for pt "<<ptId0<<" is "<<surfaceid[0]<<endl;
313  if (unique[0])
314  intersectionSurfaceId->InsertValue(ptId0,surfaceid[0]);
315 
316  //std::cout<<"Surface Id for pt "<<ptId0<<" is "<<surfaceid[0]<<endl;
317  else
318  {
319  if (intersectionSurfaceId->GetValue(ptId0) != 3)
320  intersectionSurfaceId->InsertValue(ptId0,surfaceid[0]);
321 
322  }
323  //std::cout<<"Surface Id for pt "<<ptId1<<" is "<<surfaceid[1]<<endl;
324  if (unique[1])
325  intersectionSurfaceId->InsertValue(ptId1,surfaceid[1]);
326 
327  //std::cout<<"Surface Id for pt "<<ptId1<<" is "<<surfaceid[1]<<endl;
328  else
329  {
330  if (intersectionSurfaceId->GetValue(ptId1) != 3)
331  intersectionSurfaceId->InsertValue(ptId1,surfaceid[1]);
332  }
333  }
334 
335  info->IntersectionPtsMap[0]->
336  insert(std::make_pair(ptId0, cellId0));
337  info->IntersectionPtsMap[1]->
338  insert(std::make_pair(ptId0, cellId1));
339  info->IntersectionPtsMap[0]->
340  insert(std::make_pair(ptId1, cellId0));
341  info->IntersectionPtsMap[1]->
342  insert(std::make_pair(ptId1, cellId1));
343 
344  //Check to see if duplicate line. Line can only be a duplicate
345  //line if both points are not unique and they don't equal eachother
346  if (!unique[0] && !unique[1] && ptId0 != ptId1)
347  {
348  //std::cout<<"Both Points are not unique"<<endl;
349  vtkSmartPointer<vtkPolyData> lineTest =
350  vtkSmartPointer<vtkPolyData>::New();
351  lineTest->SetPoints(pointMerger->GetPoints());
352  lineTest->SetLines(intersectionLines);
353  lineTest->BuildLinks();
354  int newLine = info->CheckLine(lineTest,ptId0,ptId1);
355  //if (iCells1->GetNumberOfIds() > 1)
356  if (newLine == 0)
357  addline = 0;
358  }
359  if (addline)
360  {
361  //If the line is new and does not consist of two identical
362  //points, add the line to the intersection and update
363  //mapping information
364  intersectionLines->InsertNextCell(2);
365  intersectionLines->InsertCellPoint(ptId0);
366  intersectionLines->InsertCellPoint(ptId1);
367 
368  intersectionCellIds0->InsertNextValue(cellId0);
369  intersectionCellIds1->InsertNextValue(cellId1);
370 
371  info->PointCellIds[0]->InsertValue( ptId0, cellId0 );
372  info->PointCellIds[0]->InsertValue( ptId1, cellId0 );
373  info->PointCellIds[1]->InsertValue( ptId0, cellId1 );
374  info->PointCellIds[1]->InsertValue( ptId1, cellId1 );
375 
376  info->IntersectionMap[0]->
377  insert(std::make_pair(cellId0, lineId));
378  info->IntersectionMap[1]->
379  insert(std::make_pair(cellId1, lineId));
380 
381  // Check which edges of cellId0 and cellId1 outpt0 and
382  // outpt1 are on, if any.
383  int isOnEdge=0;
384  int m0p0=0,m0p1=0,m1p0=0,m1p1=0;
385  for (vtkIdType edgeId = 0; edgeId < 3; edgeId++)
386  {
387  isOnEdge = info->AddToPointEdgeMap(0, ptId0, outpt0,
388  mesh0, cellId0,edgeId, lineId, triPtIds0);
389  if (isOnEdge != -1)
390  m0p0++;
391  isOnEdge = info->AddToPointEdgeMap(0, ptId1, outpt1,
392  mesh0, cellId0,edgeId, lineId, triPtIds0);
393  if (isOnEdge != -1)
394  m0p1++;
395  isOnEdge = info->AddToPointEdgeMap(1, ptId0, outpt0,
396  mesh1, cellId1,edgeId, lineId, triPtIds1);
397  if (isOnEdge != -1)
398  m1p0++;
399  isOnEdge = info->AddToPointEdgeMap(1, ptId1, outpt1,
400  mesh1, cellId1,edgeId, lineId, triPtIds1);
401  if (isOnEdge != -1)
402  m1p1++;
403  }
404  //std::cout<<"Special CASE! check!"<<endl;
405  if (m0p0 > 0 && m1p0 > 0)
406  intersectionSurfaceId->InsertValue(ptId0,3);
407  if (m0p1 > 0 && m1p1 > 0)
408  intersectionSurfaceId->InsertValue(ptId1,3);
409  }
410  }
411  }
412  }
413  }
414  }
415  }
416  return 1;

421 int vtkIntersectionPolyDataFilter2::Impl
422 ::SplitMesh(int inputIndex, vtkPolyData *output, vtkPolyData *intersectionLines)
423 {
424  vtkSmartPointer<vtkIntArray> TouchedCellArray =
425  vtkSmartPointer<vtkIntArray>::New();
426  vtkSmartPointer<vtkIntArray> TouchedPointArray =
427  vtkSmartPointer<vtkIntArray>::New();
428  TouchedCellArray->SetNumberOfComponents(1);
429  TouchedPointArray->SetNumberOfComponents(1);
430 
431  vtkPolyData *input = this->Mesh[inputIndex];
432  IntersectionMapType *intersectionMap = this->IntersectionMap[inputIndex];
433  vtkCellData *inCD = input->GetCellData();
434  vtkCellData *outCD = output->GetCellData();
435  vtkIdType numCells = input->GetNumberOfCells();
436  vtkIdType cellIdX = 0;
437 
438  //
439  // Process points
440  //
441  vtkIdType inputNumPoints = input->GetPoints()->GetNumberOfPoints();
442  vtkSmartPointer< vtkPoints > points = vtkSmartPointer< vtkPoints >::New();
443  points->Allocate(100);
444  output->SetPoints(points);
445 
446  //
447  // Split intersection lines. The lines structure is constructed
448  // using a vtkPointLocator. However, some lines may have an endpoint
449  // on a cell edge that has no neighbor. We need to duplicate a line
450  // point in such a case and update the point ID in the line cell.
451  //
452  vtkSmartPointer< vtkPolyData > splitLines =
453  vtkSmartPointer <vtkPolyData >::New();
454  splitLines->DeepCopy(intersectionLines);
455 
456  vtkPointData *inPD = input->GetPointData();
457  vtkPointData *outPD = output->GetPointData();
458  outPD->CopyAllocate( inPD, input->GetNumberOfPoints() );
459 
460  // Copy over the point data from the input
461  for (vtkIdType ptId = 0; ptId < inputNumPoints; ptId++)
462  {
463  double pt[3];
464  input->GetPoints()->GetPoint(ptId, pt);
465  output->GetPoints()->InsertNextPoint(pt);
466  outPD->CopyData(inPD, ptId, ptId);
467  TouchedPointArray->InsertTuple1(ptId,-2);
468  BoundaryPoints[inputIndex]->InsertValue(ptId,0);
469  }
470 
471  // Copy the points from splitLines to the output, interpolating the
472  // data as we go.
473  //std::cout<<"NumberOfPoints "<<splitLines->GetNumberOfPoints()<<endl;
474  //std::cout<<"NumberOfLines "<<splitLines->GetNumberOfLines()<<endl;
475  for (vtkIdType id = 0; id < splitLines->GetNumberOfPoints(); id++)
476  {
477  double pt[3];
478  splitLines->GetPoint(id, pt);
479  vtkIdType newPtId = output->GetPoints()->InsertNextPoint(pt);
480 
481  // Retrieve the cell ID from splitLines
482  vtkIdType cellId = this->PointCellIds[inputIndex]->GetValue( id );
483 
484  double closestPt[3], pcoords[3], dist2, weights[3];
485  int subId;
486  vtkCell *cell = input->GetCell(cellId);
487  cell->EvaluatePosition(pt, closestPt, subId, pcoords, dist2, weights);
488  outPD->InterpolatePoint(input->GetPointData(), newPtId, cell->PointIds,
489  weights);
490  TouchedPointArray->InsertTuple1(newPtId,-2);
491  BoundaryPoints[inputIndex]->InsertValue(newPtId,0);
492  }
493 
494  //
495  // Process cells
496  //
497  outCD->CopyAllocate(inCD, numCells);
498 
499  if ( input->GetPolys()->GetNumberOfCells() > 0 )
500  {
501  vtkCellArray *cells = input->GetPolys();
502  vtkIdType newId = output->GetNumberOfCells();
503 
504  vtkSmartPointer< vtkCellArray > newPolys =
505  vtkSmartPointer< vtkCellArray >::New();
506 
507  newPolys->EstimateSize(cells->GetNumberOfCells(),3);
508  output->SetPolys(newPolys);
509 
510  vtkSmartPointer< vtkIdList > edgeNeighbors =
511  vtkSmartPointer< vtkIdList >::New();
512  vtkIdType nptsX = 0;
513  vtkIdType *pts = 0;
514  vtkSmartPointer< vtkIdList > cellsToCheck =
515  vtkSmartPointer< vtkIdList >::New();
516  for (cells->InitTraversal(); cells->GetNextCell(nptsX, pts); cellIdX++)
517  {
518  if ( nptsX != 3 )
519  {
520  vtkGenericWarningMacro( << "vtkIntersectionPolyDataFilter2 only works with "
521  << "triangle meshes." );
522  continue;
523  }
524 
525  cellsToCheck->Reset();
526  cellsToCheck->Allocate(nptsX+1);
527  cellsToCheck->InsertNextId(cellIdX);
528 
529  // Collect the cells relevant for splitting this cell. If the
530  // cell is in the intersection map, split. If not, one of its
531  // edges may be split by an intersection line that splits a
532  // neighbor cell. Mark the cell as needing a split if this is
533  // the case.
534  bool needsSplit = intersectionMap->find( cellIdX )
535  != intersectionMap->end();
536  for (vtkIdType ptId = 0; ptId < nptsX; ptId++)
537  {
538  vtkIdType pt0Id = pts[ptId];
539  vtkIdType pt1Id = pts[(ptId+1) % nptsX];
540  edgeNeighbors->Reset();
541  input->GetCellEdgeNeighbors(cellIdX, pt0Id, pt1Id, edgeNeighbors);
542  for (vtkIdType nbr = 0; nbr < edgeNeighbors->GetNumberOfIds(); nbr++)
543  {
544  vtkIdType nbrId = edgeNeighbors->GetId(nbr);
545  cellsToCheck->InsertNextId(nbrId);
546 
547  if ( intersectionMap->find( nbrId ) != intersectionMap->end() )
548  {
549  needsSplit = true;
550  }
551  } // for (vtkIdType nbr = 0; ...
552  } // for (vtkIdType pt = 0; ...
553 
554  // Splitting occurs here
555  if ( !needsSplit )
556  {
557  // Just insert the cell and copy the cell data
558  newId = newPolys->InsertNextCell(3, pts);
559  outCD->CopyData(inCD, cellIdX, newId);
560  TouchedCellArray->InsertTuple1(newId,-2);
561  }
562  else
563  {
564  //Total number of cells so that we know the id numbers of the new
565  //cells added and we can add it to the new cell id mapping
566  int numCurrCells = newPolys->GetNumberOfCells();
567  vtkCellArray *splitCells = this->SplitCell
568  (input, cellIdX, pts, intersectionMap, splitLines,
569  inputIndex,numCurrCells);
570 
571 
572  double pt0[3], pt1[3], pt2[3], normal[3];
573  points->GetPoint(pts[0], pt0);
574  points->GetPoint(pts[1], pt1);
575  points->GetPoint(pts[2], pt2);
576  vtkTriangle::ComputeNormal(pt0, pt1, pt2, normal);
577  vtkMath::Normalize(normal);
578 
579  vtkIdType npts, *ptIds, subCellId;
580  splitCells->InitTraversal();
581  for (subCellId = 0;splitCells->GetNextCell(npts, ptIds);subCellId++)
582  {
583  // Check for reversed cells. I'm not sure why, but in some
584  // cases, cells are reversed.
585  double subCellNormal[3];
586  points->GetPoint(ptIds[0], pt0);
587  points->GetPoint(ptIds[1], pt1);
588  points->GetPoint(ptIds[2], pt2);
589  vtkTriangle::ComputeNormal(pt0, pt1, pt2, subCellNormal);
590  vtkMath::Normalize(subCellNormal);
591 
592  if ( vtkMath::Dot(normal, subCellNormal) > 0 )
593  {
594  newId = newPolys->InsertNextCell(npts, ptIds);
595  }
596  else
597  {
598  newId = newPolys->InsertNextCell(npts);
599  for ( int i = 0; i < npts; i++)
600  {
601  newPolys->InsertCellPoint( ptIds[ npts-i-1 ] );
602  }
603  }
604 
605  outCD->CopyData(inCD, cellIdX, newId); // Duplicate cell data
606  TouchedCellArray->InsertTuple1(newId,1);
607  for (int i=0;i<npts;i++)
608  {
609  TouchedPointArray->InsertTuple1(ptIds[i],1);
610  }
611 
612  }

1339 //----------------------------------------------------------------------------
1340 int vtkIntersectionPolyDataFilter2::TriangleTriangleIntersection(double p1[3],
1341  double q1[3], double r1[3], double p2[3],
1342  double q2[3], double r2[3], int &coplanar,
1343  double pt1[3], double pt2[3], double surfaceid[2])
1344 {
1345  double n1[3], n2[3];
1346 
1347  // Compute supporting plane normals.
1348  vtkTriangle::ComputeNormal(p1, q1, r1, n1);
1349  vtkTriangle::ComputeNormal(p2, q2, r2, n2);
1350  double s1 = -vtkMath::Dot(n1, p1);
1351  double s2 = -vtkMath::Dot(n2, p2);
1352 
1353  // Compute signed distances of points p1, q1, r1 from supporting
1354  // plane of second triangle.
1355  double dist1[3];
1356  dist1[0] = vtkMath::Dot(n2, p1) + s2;
1357  dist1[1] = vtkMath::Dot(n2, q1) + s2;
1358  dist1[2] = vtkMath::Dot(n2, r1) + s2;
1359 
1360  // If signs of all points are the same, all the points lie on the
1361  // same side of the supporting plane, and we can exit early.
1362  if ((dist1[0]*dist1[1] > 0.0) && (dist1[0]*dist1[2] > 0.0))
1363  {
1364  //std::cout<<"Same side supporting plane 1!"<<endl;
1365  return 0;
1366  }
1367  // Do the same for p2, q2, r2 and supporting plane of first
1368  // triangle.
1369  double dist2[3];
1370  dist2[0] = vtkMath::Dot(n1, p2) + s1;
1371  dist2[1] = vtkMath::Dot(n1, q2) + s1;
1372  dist2[2] = vtkMath::Dot(n1, r2) + s1;
1373 
1374  // If signs of all points are the same, all the points lie on the
1375  // same side of the supporting plane, and we can exit early.
1376  if ((dist2[0]*dist2[1] > 0.0) && (dist2[0]*dist2[2] > 0.0))
1377  {
1378  //std::cout<<"Same side supporting plane 2!"<<endl;
1379  return 0;
1380  }
1381  // Check for coplanarity of the supporting planes.
1382  if ( fabs( n1[0] - n2[0] ) < 1e-9 &&
1383  fabs( n1[1] - n2[1] ) < 1e-9 &&
1384  fabs( n1[2] - n2[2] ) < 1e-9 &&
1385  fabs( s1 - s2 ) < 1e-9 )
1386  {
1387  coplanar = 1;
1388  //std::cout<<"Coplanar!"<<endl;
1389  return 0;
1390  }
1391 
1392  coplanar = 0;
1393 
1394  // There are more efficient ways to find the intersection line (if
1395  // it exists), but this is clear enough.
1396  double *pts1[3] = {p1, q1, r1}, *pts2[3] = {p2, q2, r2};
1397 
1398  // Find line of intersection (L = p + t*v) between two planes.
1399  double n1n2 = vtkMath::Dot(n1, n2);
1400  double a = (s1 - s2*n1n2) / (n1n2*n1n2 - 1.0);
1401  double b = (s2 - s1*n1n2) / (n1n2*n1n2 - 1.0);
1402  double p[3], v[3];
1403  p[0] = a*n1[0] + b*n2[0];
1404  p[1] = a*n1[1] + b*n2[1];
1405  p[2] = a*n1[2] + b*n2[2];
1406  vtkMath::Cross(n1, n2, v);
1407  vtkMath::Normalize( v );
1408 
1409  double tol = 1e-10;
1410  int index1 = 0, index2 = 0;
1411  double t1[3], t2[3];
1412  int ts1=50,ts2=50;
1413  for (int i = 0; i < 3; i++)
1414  {
1415  double t, x[3];
1416  int id1 = i, id2 = (i+1) % 3;
1417 
1418  // Find t coordinate on line of intersection between two planes.
1419  if (vtkPlane::IntersectWithLine( pts1[id1], pts1[id2], n2, p2, t, x ))
1420  {
1421  if (t < 1+tol && t > 1-tol)
1422  ts1 = index1;
1423 
1424  t1[index1++] = vtkMath::Dot(x, v) - vtkMath::Dot(p, v);
1425  }
1426 
1427  if (vtkPlane::IntersectWithLine( pts2[id1], pts2[id2], n1, p1, t, x ))
1428  {
1429  if (t < 1+tol && t > 1-tol)
1430  ts2 = index2;
1431 
1432  t2[index2++] = vtkMath::Dot(x, v) - vtkMath::Dot(p, v);
1433  }
1434  }
1435 
1436  //If the value of the index is greater than 2, the intersecting point
1437  //actually is intersected by all three edges. In this case, set the two
1438  //edges to the two edges where the intersecting point is not the end point
1439  if ( index1 > 2)
1440  {
1441  index1--;
1442  std::swap(t1[ts1], t1[2]);
1443  }
1444  if ( index2 > 2)
1445  {
1446  index2--;
1447  std::swap(t2[ts2], t2[2]);
1448  }
1449  // Check if only one edge or all edges intersect the supporting
1450  // planes intersection.
1451  if ( index1 != 2 || index2 != 2 )
1452  {
1453  //std::cout<<"Only one edge intersecting!"<<endl;
1454  return 0;
1455  }
1456 
1457  // Check for NaNs
1458  if ( vtkMath::IsNan( t1[0] ) || vtkMath::IsNan( t1[1] ) ||
1459  vtkMath::IsNan( t2[0] ) || vtkMath::IsNan( t2[1] ) )
1460  {
1461  std::cout<<"NaNs!"<<endl;
1462  return 0;
1463  }
1464 
1465  if ( t1[0] > t1[1] )
1466  {
1467  std::swap( t1[0], t1[1] );
1468  }
1469  if ( t2[0] > t2[1] )
1470  {
1471  std::swap( t2[0], t2[1] );
1472  }
1473  // Handle the different interval configuration cases.
1474  double tt1, tt2;
1475  if ( t1[1] < t2[0] || t2[1] < t1[0] )
1476  {
1477  // std::cout<<"No Overlap!"<<endl;
1478  return 0; // No overlap
1479  }
1480  else if ( t1[0] < t2[0] )
1481  {
1482  if ( t1[1] < t2[1] )
1483  {
1484  //First point on surface 2, second point on surface 1
1485  surfaceid[0] = 2;
1486  surfaceid[1] = 1;
1487  tt1 = t2[0];
1488  tt2 = t1[1];
1489  }
1490  else
1491  {
1492  //Both points belong to lines on surface 2
1493  surfaceid[0] = 2;
1494  surfaceid[1] = 2;
1495  tt1 = t2[0];
1496  tt2 = t2[1];
1497  }
1498  }
1499  else // t1[0] >= t2[0]
1500  {
1501  if ( t1[1] < t2[1] )
1502  {
1503  //Both points belong to lines on surface 1
1504  surfaceid[0] = 1;
1505  surfaceid[1] = 1;
1506  tt1 = t1[0];
1507  tt2 = t1[1];
1508  }
1509  else
1510  {
1511  //First point on surface 1, second point on surface 2
1512  surfaceid[0] = 1;
1513  surfaceid[1] = 2;
1514  tt1 = t1[0];
1515  tt2 = t2[1];
1516  }
1517  }
1518 
1519  // Create actual intersection points.
1520  pt1[0] = p[0] + tt1*v[0];
1521  pt1[1] = p[1] + tt1*v[1];
1522  pt1[2] = p[2] + tt1*v[2];
1523 
1524  pt2[0] = p[0] + tt2*v[0];
1525  pt2[1] = p[1] + tt2*v[1];
1526  pt2[2] = p[2] + tt2*v[2];
1527 
1528  return 1;
1529 }
