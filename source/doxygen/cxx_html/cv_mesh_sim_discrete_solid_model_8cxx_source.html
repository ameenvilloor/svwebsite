134 int cvMeshSimDiscreteSolidModel::FindNodesOnElementFace (pFace face, int* nodes, double *xyz) const {
135 
136  int i;
137  int num_nodes = 0;
138 
139  int pos = 0;
140 
141  pPList vert_list = F_vertices (face,MY_MESHSIM_VERTEX_ORDERING);
142  int numElemFaceVerts = PList_size (vert_list);
143 
144  // write out "linear" nodes
145  for (i = 0; i < numElemFaceVerts; i++) {
146  pVertex vertex = (pVertex)PList_item (vert_list, i);
147  pPoint point = V_point (vertex);
148  xyz[pos++]=P_x(point);
149  xyz[pos++]=P_y(point);
150  xyz[pos++]=P_z(point);
151  nodes[i] = P_id(point);
152  }
153 
154  num_nodes = numElemFaceVerts;
155 
156  // quad nodes
157  if (quadElem_ == 1) {
158  pPList edge_list = F_edges(face, 1, NULL);
159  int numEdges = PList_size (edge_list);
160 
161  for (i = 0; i < numEdges; i++) {
162  pEdge xedge = (pEdge)PList_item (edge_list, i);
163  if (E_numPoints(xedge) != 1) {
164  fprintf(stderr,"ERROR: no interior point!\n");
165  exit(-1);
166  }
167  nodes[numElemFaceVerts+i] = P_id(E_point(xedge,0));
168  xyz[pos++]=P_x(E_point(xedge,0));
169  xyz[pos++]=P_y(E_point(xedge,0));
170  xyz[pos++]=P_z(E_point(xedge,0));
171  }
172 
173  num_nodes+=numEdges;
174  }
175 
176  return num_nodes;
177 
178 }

181 cvPolyData* cvMeshSimDiscreteSolidModel::GetFacePolyData(int faceID, int useMaxDist, double max_dist) const {
182 
183  // useMaxDist and max_dist are ignored!
184 
185  // recall the node numbers start at 1 in the P_id,
186  // but in vtkPolyData file they start at 0.
187 
188  int n;
189  vtkPolyData *mycvPolyData;
190  cvPolyData *result;
191 
192  int foundIt = 0;
193 
194  GFIter myGFiter = GM_faceIter(geom_);
195  pGFace mygface;
196  while (mygface = GFIter_next(myGFiter)) {
197  if (faceID == GEN_tag(mygface)) {
198  foundIt++;
199  break;
200  }
201  }
202  GFIter_delete(myGFiter);
203 
204  if (!foundIt) {
205  fprintf(stdout,"Error: face id (%i) not found in model!\n",faceID);
206  return NULL;
207  }
208 
209  pMesh mesh = DM_getMesh(geom_);
210 
211  // vtk requires single precision
212  vtkFloatingPointType p[3];
213 
214  // create the vtk object
215  vtkPoints* myPoints = vtkPoints::New();
216  myPoints->Allocate(1000,1000);
217  mycvPolyData = vtkPolyData::New();
218  mycvPolyData->Allocate(1000,1000);
219  mycvPolyData->SetPoints(myPoints);
220 
221  vtkIdList* PointIds = vtkIdList::New();
222  PointIds->Allocate(10,10);
223 
224  // track the actual mesh node numbers in a scalar array
225  vtkFloatArray *myScalars = vtkFloatArray::New();
226  myScalars->Allocate(1000,1000);
227 
228  // node numbering in polydata object
229  int poly_nodes = 0;
230 
231  int num_faces = M_numFaces (mesh);
232 
233  FIter myFiter = M_faceIter(mesh);
234  pFace myface = NULL;
235  while (myface = FIter_next(myFiter)) {
236 
237  //if (!(loopfaces % 10000)) fprintf(stdout,"face %i\n",loopfaces);
238 
239  // Get the model tag for the face
240  int facetag=0;
241  pGEntity modelface = F_whatIn(myface);
242  if (modelface != NULL) {
243  facetag=GEN_tag(F_whatIn(myface));
244  }
245 
246  /* only output element face if it is on the model face */
247  if (facetag == faceID) {
248  int nodes[20];
249  double xyz[60];
250  int num_nodes = FindNodesOnElementFace(myface,nodes,xyz);
251 
252  int pos = 0;
253  for (n=0;n < num_nodes; n++) {
254  // vtk requires single precision
255  vtkFloatingPointType p[3];
256  p[0]=xyz[pos++];p[1]=xyz[pos++];p[2]=xyz[pos++];
257  myPoints->InsertNextPoint(p);
258  myScalars->InsertNextTuple1(nodes[n]);
259  }
260 
261  PointIds->Reset();
262  for (n = 0; n < num_nodes; n++) {
263  PointIds->InsertNextId(poly_nodes++);
264  }
265  mycvPolyData->InsertNextCell(VTK_POLYGON,PointIds);
266  }
267 
268 
269  }
270  FIter_delete(myFiter);
271 
272  // attach the scalars to the polydata
273  mycvPolyData->GetPointData()->SetScalars(myScalars);
274 
275  PointIds->Delete();
276  cvPolyData *tmpresult = new cvPolyData( mycvPolyData );
277  mycvPolyData->Delete(); // cvPolyData uses ref-counting
278 
279  // delete points and scalars here?
280  myPoints->Delete();
281  myScalars->Delete();
282 
283  // delete duplicate points
284  result = sys_geom_MergePts( tmpresult );
285 
286  delete tmpresult;
287 
288  return result;
289 
290 }
