169 inline
170 double cvLevelSetNode::ComputeDoubleDatum( GridScalarT t, double tol )
171 {
172  double ks[2];
173 
174  switch (t) {
175  case GS_CURVATURE_3Dk1:
176  if ( Compute3dks( K3dg_, K3dm_, tol, ks ) != CV_OK ) {
177  return 0.0;
178  }
179  return ks[0];
180  case GS_CURVATURE_3Dk2:
181  if ( Compute3dks( K3dg_, K3dm_, tol, ks ) != CV_OK ) {
182  return 0.0;
183  }
184  return ks[1];
185  default:
186  return 0.0;
187  }
188 }
