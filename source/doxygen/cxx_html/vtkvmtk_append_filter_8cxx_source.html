50 int vtkvmtkAppendFilter::RequestData(
51  vtkInformation *vtkNotUsed(request),
52  vtkInformationVector **inputVector,
53  vtkInformationVector *outputVector)
54 {
55  // get the output info object
56  vtkInformation *outInfo = outputVector->GetInformationObject(0);
57 
58  // get the ouptut
59  vtkUnstructuredGrid *output = vtkUnstructuredGrid::SafeDownCast(
60  outInfo->Get(vtkDataObject::DATA_OBJECT()));
61 
62  vtkIdType numPts, numCells;
63  int tenth, count, abort=0;
64  float decimal;
65  vtkPoints *newPts;
66  vtkPointData *pd;
67  vtkCellData *cd;
68  vtkIdList *ptIds, *newPtIds;
69  int i, idx;
70  vtkDataSet *ds;
71  vtkIdType ptId, cellId, newCellId;
72  vtkPointData *outputPD = output->GetPointData();
73  vtkCellData *outputCD = output->GetCellData();
74 
75  vtkDebugMacro(<<"Appending data together");
76 
77  double bounds[6], inputBounds[6];
78  vtkMath::UninitializeBounds(bounds);
79 
80  // Loop over all data sets, checking to see what data is common to
81  // all inputs. Note that data is common if 1) it is the same attribute
82  // type (scalar, vector, etc.), 2) it is the same native type (int,
83  // float, etc.), and 3) if a data array in a field, if it has the same name.
84  count = 0;
85  decimal = 0.0;
86 
87  numPts = 0;
88  numCells = 0;
89 
90  int numInputs = inputVector[0]->GetNumberOfInformationObjects();
91  vtkDataSetAttributes::FieldList ptList(numInputs);
92  vtkDataSetAttributes::FieldList cellList(numInputs);
93  int firstPD=1;
94  int firstCD=1;
95  vtkInformation *inInfo = 0;
96  for (idx = 0; idx < numInputs; ++idx)
97  {
98  inInfo = inputVector[0]->GetInformationObject(idx);
99  ds = 0;
100  if (inInfo)
101  {
102  ds = vtkDataSet::SafeDownCast(inInfo->Get(vtkDataObject::DATA_OBJECT()));
103  }
104  if (ds != NULL)
105  {
106  if ( ds->GetNumberOfPoints() <= 0 && ds->GetNumberOfCells() <= 0 )
107  {
108  continue; //no input, just skip
109  }
110 
111  numPts += ds->GetNumberOfPoints();
112  numCells += ds->GetNumberOfCells();
113 
114  ds->GetBounds(inputBounds);
115  if (vtkMath::AreBoundsInitialized(bounds))
116  {
117  int j;
118  for (j=0; j<6; j+=2)
119  {
120  if (inputBounds[j] < bounds[j])
121  {
122  bounds[j] = inputBounds[j];
123  }
124  if (inputBounds[j+1] > bounds[j+1])
125  {
126  bounds[j+1] = inputBounds[j+1];
127  }
128  }
129  }
130  else
131  {
132  int j;
133  for (j=0; j<6; j+=2)
134  {
135  bounds[j] = inputBounds[j];
136  bounds[j+1] = inputBounds[j+1];
137  }
138  }
139 
140  pd = ds->GetPointData();
141  if ( firstPD )
142  {
143  ptList.InitializeFieldList(pd);
144  firstPD = 0;
145  }
146  else
147  {
148  ptList.IntersectFieldList(pd);
149  }
150 
151  cd = ds->GetCellData();
152  if ( firstCD )
153  {
154  cellList.InitializeFieldList(cd);
155  firstCD = 0;
156  }
157  else
158  {
159  cellList.IntersectFieldList(cd);
160  }
161  }//if non-empty dataset
162  }//for all inputs
163 
164  if ( numPts < 1)
165  {
166  vtkDebugMacro(<<"No data to append!");
167  return 1;
168  }
169 
170  // Now can allocate memory
171  output->Allocate(numCells); //allocate storage for geometry/topology
172 // outputPD->CopyGlobalIdsOn();
173  outputPD->CopyAllocate(ptList,numPts);
174 // outputCD->CopyGlobalIdsOn();
175  outputCD->CopyAllocate(cellList,numCells);
176 
177  newPts = vtkPoints::New();
178  ptIds = vtkIdList::New(); ptIds->Allocate(VTK_CELL_SIZE);
179  newPtIds = vtkIdList::New(); newPtIds->Allocate(VTK_CELL_SIZE);
180 
181  vtkMergePoints* locator = vtkMergePoints::New();
182  if (this->MergeDuplicatePoints)
183  {
184  locator->InitPointInsertion(newPts,bounds);
185  }
186 
187  vtkIdList* ptIdMap = vtkIdList::New();
188 
189  vtkIdType newPtId;
190  double point[3];
191 
192  // Append each input dataset together
193  //
194  tenth = (numPts + numCells)/10 + 1;
195  int inputCount = 0; // Since empty inputs are not in the list.
196  //fprintf(stderr,"NumInputs: %d\n",numInputs);
197  for (idx = 0; idx < numInputs && !abort; ++idx)
198  {
199  inInfo = inputVector[0]->GetInformationObject(idx);
200  ds = 0;
201  if (inInfo)
202  {
203  ds = vtkDataSet::SafeDownCast(inInfo->Get(vtkDataObject::DATA_OBJECT()));
204  }
205  if ( ds != NULL &&
206  (ds->GetNumberOfPoints() > 0 || ds->GetNumberOfCells() > 0) )
207  {
208  numPts = ds->GetNumberOfPoints();
209  numCells = ds->GetNumberOfCells();
210  //fprintf(stderr,"NumPts: %d\n",numPts);
211  pd = ds->GetPointData();
212 
213  ptIdMap->Initialize();
214  ptIdMap->SetNumberOfIds(numPts);
215 
216  // copy points and point data
217  for (ptId=0; ptId < numPts && !abort; ptId++)
218  {
219  ds->GetPoint(ptId,point);
220  if (this->MergeDuplicatePoints)
221  {
222  if (locator->InsertUniquePoint(point,newPtId))
223  {
224  outputPD->CopyData(ptList,pd,inputCount,ptId,newPtId);
225  }
226  }
227  else
228  {
229  newPtId = newPts->InsertNextPoint(point);
230  outputPD->CopyData(ptList,pd,inputCount,ptId,newPtId);
231  }
232  ptIdMap->SetId(ptId,newPtId);
233 
234  // Update progress
235  count++;
236  if (!(count % tenth))
237  {
238  decimal += 0.1;
239  this->UpdateProgress(decimal);
240  abort = this->GetAbortExecute();
241  }
242  }
243 
244  cd = ds->GetCellData();
245  // copy cell and cell data
246  for (cellId=0; cellId<numCells && !abort; cellId++)
247  {
248  ds->GetCellPoints(cellId, ptIds);
249  newPtIds->Reset();
250  for (i=0; i<ptIds->GetNumberOfIds(); i++)
251  {
252  newPtIds->InsertId(i,ptIdMap->GetId(ptIds->GetId(i)));
253  }
254  newCellId = output->InsertNextCell(ds->GetCellType(cellId),newPtIds);
255  outputCD->CopyData(cellList,cd,inputCount,cellId,newCellId);
256 
257  // Update progress
258  count++;
259  if (!(count % tenth))
260  {
261  decimal += 0.1;
262  this->UpdateProgress(decimal);
263  abort = this->GetAbortExecute();
264  }
265  }
266  ++inputCount;
267  }
268  }
269 
270  // Update ourselves and release memory
271  //
272  output->SetPoints(newPts);
273  newPts->Delete();
274  ptIds->Delete();
275  newPtIds->Delete();
276  locator->Delete();
277 
278  return 1;
279 }
