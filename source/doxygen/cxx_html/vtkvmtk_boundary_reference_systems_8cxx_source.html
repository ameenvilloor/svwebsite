73 void vtkvmtkBoundaryReferenceSystems::ComputeBoundaryBarycenter(vtkPoints* points, double* barycenter)
74 {
75  int numberOfPoints;
76  double weightSum, weight;
77  vtkIdType i;
78 
79  // accounts for edge length
80 
81  numberOfPoints = points->GetNumberOfPoints();
82 
83  weightSum = 0.0;
84 
85  barycenter[0] = 0.0;
86  barycenter[1] = 0.0;
87  barycenter[2] = 0.0;
88 
89  double point0[3], point1[3];
90 
91  for (i=0; i<numberOfPoints; i++)
92  {
93  points->GetPoint(i,point0);
94  points->GetPoint((i+1)%numberOfPoints,point1);
95  weight = sqrt(vtkMath::Distance2BetweenPoints(point0,point1));
96  if (weight < VTK_VMTK_DOUBLE_TOL)
97  {
98  continue;
99  }
100  weightSum += weight;
101 
102  barycenter[0] += (point0[0] + point1[0])/2.0 * weight;
103  barycenter[1] += (point0[1] + point1[1])/2.0 * weight;
104  barycenter[2] += (point0[2] + point1[2])/2.0 * weight;
105  }
106 
107  barycenter[0] /= weightSum;
108  barycenter[1] /= weightSum;
109  barycenter[2] /= weightSum;
110 }

130 void vtkvmtkBoundaryReferenceSystems::ComputeBoundaryNormal(vtkPoints* points, double* barycenter, double* normal)
131 {
132  int numberOfPoints;
133  double point[3], vector1[3], vector0[3], cross[3], sumCross[3] = {0.0f,0.0f,0.0f};
134  vtkIdType i, j;
135 
136  // TODO: weight by edge length as for barycenter
137 
138  numberOfPoints = points->GetNumberOfPoints();
139 
140  points->GetPoint(numberOfPoints-1,point);
141  vector0[0] = point[0] - barycenter[0];
142  vector0[1] = point[1] - barycenter[1];
143  vector0[2] = point[2] - barycenter[2];
144 
145  for (i=0; i<numberOfPoints; i++)
146  {
147  points->GetPoint(i,point);
148  vector1[0] = point[0] - barycenter[0];
149  vector1[1] = point[1] - barycenter[1];
150  vector1[2] = point[2] - barycenter[2];
151 
152  vtkMath::Cross(vector1,vector0,cross);
153 
154  for (j=0; j<3; j++)
155  {
156  vector0[j] = vector1[j];
157  sumCross[j] += cross[j];
158  }
159  }
160 
161  vtkMath::Normalize(sumCross);
162 
163  for (j=0; j<3; j++)
164  {
165  normal[j] = sumCross[j];
166  }
167 }

169 void vtkvmtkBoundaryReferenceSystems::OrientBoundaryNormalOutwards(vtkPolyData* surface, vtkPolyData* boundaries, vtkIdType boundaryCellId, double normal[3], double outwardNormal[3])
170 {
171  vtkPolyLine* boundary = vtkPolyLine::SafeDownCast(boundaries->GetCell(boundaryCellId));
172 
173  int k=0;
174  if (!boundary)
175  {
176  for (k=0; k<3; k++)
177  {
178  outwardNormal[k] = normal[k];
179  }
180  return;
181  }
182 
183  surface->BuildCells();
184  surface->BuildLinks();
185 
186  unsigned short ncells;
187  vtkIdType *cells;
188  vtkIdType npts, *pts;
189 
190  double boundaryPoint[3], neighborPoint[3];
191 
192  int numberOfBoundaryPoints = boundary->GetNumberOfPoints();
193 
194  int j;
195  vtkIdList* boundaryIds = vtkIdList::New();
196  for (j=0; j<numberOfBoundaryPoints; j++)
197  {
198  boundaryIds->InsertNextId(static_cast<vtkIdType>(vtkMath::Round(boundaries->GetPointData()->GetScalars()->GetComponent(boundary->GetPointId(j),0))));
199  }
200 
201  double neighborsToBoundaryNormal[3];
202  neighborsToBoundaryNormal[0] = neighborsToBoundaryNormal[1] = neighborsToBoundaryNormal[2] = 0.0;
203  for (j=0; j<numberOfBoundaryPoints; j++)
204  {
205  surface->GetPointCells(boundaryIds->GetId(j),ncells,cells);
206  for (int c=0; c<ncells; c++)
207  {
208  surface->GetCellPoints(cells[c],npts,pts);
209  for (int p=0; p<npts; p++)
210  {
211  if (boundaryIds->IsId(pts[p]) == -1)
212  {
213  surface->GetPoint(boundaryIds->GetId(j),boundaryPoint);
214  surface->GetPoint(pts[p],neighborPoint);
215  for (k=0; k<3; k++)
216  {
217  neighborsToBoundaryNormal[k] += boundaryPoint[k] - neighborPoint[k];
218  }
219  }
220  }
221  }
222  }
223 
224  for (k=0; k<3; k++)
225  {
226  outwardNormal[k] = normal[k];
227  }
228 
229  vtkMath::Normalize(neighborsToBoundaryNormal);
230 
231  if (vtkMath::Dot(normal,neighborsToBoundaryNormal) < 0.0)
232  {
233  for (k=0; k<3; k++)
234  {
235  outwardNormal[k] *= -1.0;
236  }
237  }
238 
239  boundaryIds->Delete();
240 }

242 void vtkvmtkBoundaryReferenceSystems::ComputeReferenceSystemPoints(double origin[3], double normal[3], double radius, double point1[3], double point2[3])
243 {
244  point1[0] = point1[1] = point1[2] = 0.0;
245  point2[0] = point2[1] = point2[2] = 0.0;
246 
247  double x_p[3], y_p[3], x[3];
248 
249  x[0] = 1.0;
250  x[1] = 0.0;
251  x[2] = 0.0;
252 
253  vtkMath::Cross(normal,x,x_p);
254  vtkMath::Normalize(x_p);
255 
256  vtkMath::Cross(normal,x_p,y_p);
257  vtkMath::Normalize(y_p);
258 
259  for (int i=0; i<3; i++)
260  {
261  point1[i] = origin[i] + x_p[i] * radius;
262  point2[i] = origin[i] + y_p[i] * radius;
263  }
264 }

266 int vtkvmtkBoundaryReferenceSystems::RequestData(
267  vtkInformation *vtkNotUsed(request),
268  vtkInformationVector **inputVector,
269  vtkInformationVector *outputVector)

271  // get the info objects

278  vtkPolyData *output = vtkPolyData::SafeDownCast(
279  outInfo->Get(vtkDataObject::DATA_OBJECT()));
280 
281  if (!this->BoundaryRadiusArrayName)

283  vtkErrorMacro(<< "BoundaryRadiusArrayName not specified.");
284  return 1;
285  }
286 
287  if (!this->BoundaryNormalsArrayName)
288  {
289  vtkErrorMacro(<< "BoundaryNormalsArrayName not specified.");
290  return 1;
291  }
292 
293  if (!this->Point1ArrayName)
294  {
295  vtkErrorMacro(<< "Point1ArrayName not specified.");
296  return 1;
297  }
298 
299  if (!this->Point2ArrayName)
300  {
301  vtkErrorMacro(<< "Point2ArrayName not specified.");
302  return 1;
303  }
304 
305  vtkPoints* outputPoints = vtkPoints::New();
306  vtkCellArray* outputVerts = vtkCellArray::New();
307  vtkDoubleArray* outputRadius = vtkDoubleArray::New();
308  outputRadius->SetName(this->BoundaryRadiusArrayName);
309  vtkDoubleArray* outputNormals = vtkDoubleArray::New();
310  outputNormals->SetName(this->BoundaryNormalsArrayName);
311  outputNormals->SetNumberOfComponents(3);
312  vtkDoubleArray* point1Array = vtkDoubleArray::New();
313  point1Array->SetName(this->Point1ArrayName);
314  point1Array->SetNumberOfComponents(3);
315  vtkDoubleArray* point2Array = vtkDoubleArray::New();
316  point2Array->SetName(this->Point2ArrayName);
317  point2Array->SetNumberOfComponents(3);
318 
319  vtkvmtkPolyDataBoundaryExtractor* boundaryExtractor = vtkvmtkPolyDataBoundaryExtractor::New();
320  boundaryExtractor->SetInputData(input);
321  boundaryExtractor->Update();
322 
323  vtkPolyData* boundaries = boundaryExtractor->GetOutput();
324 
325  double barycenter[3], normal[3], outwardNormal[3], meanRadius;
326 
327  for (int i=0; i<boundaries->GetNumberOfCells(); i++)
328  {
329  vtkPolyLine* boundary = vtkPolyLine::SafeDownCast(boundaries->GetCell(i));
330 
331  if (!boundary)
332  {
333  vtkErrorMacro(<<"Boundary not a vtkPolyLine");
334  continue;
335  }
336 
337  this->ComputeBoundaryBarycenter(boundary->GetPoints(),barycenter);
338  meanRadius = this->ComputeBoundaryMeanRadius(boundary->GetPoints(),barycenter);
339  this->ComputeBoundaryNormal(boundary->GetPoints(),barycenter,normal);
340  this->OrientBoundaryNormalOutwards(input,boundaries,i,normal,outwardNormal);
341  double point1[3], point2[3];
342  this->ComputeReferenceSystemPoints(barycenter,normal,meanRadius,point1,point2);
343 
344  vtkIdType pointId = outputPoints->InsertNextPoint(barycenter);
345  outputVerts->InsertNextCell(1);
346  outputVerts->InsertCellPoint(pointId);
347  outputNormals->InsertNextTuple(outwardNormal);
348  outputRadius->InsertNextValue(meanRadius);
349  point1Array->InsertNextTuple(point1);
350  point2Array->InsertNextTuple(point2);
351  }
352 
353  output->SetPoints(outputPoints);
354  output->SetVerts(outputVerts);
355  output->GetPointData()->AddArray(outputNormals);
356  output->GetPointData()->SetActiveNormals(this->BoundaryNormalsArrayName);
357  output->GetPointData()->AddArray(outputRadius);
358  output->GetPointData()->SetActiveScalars(this->BoundaryRadiusArrayName);
359  output->GetPointData()->AddArray(point1Array);
360  output->GetPointData()->AddArray(point2Array);
361 
362  outputPoints->Delete();
363  outputVerts->Delete();
364  outputNormals->Delete();
365  outputRadius->Delete();
366  point1Array->Delete();
367  point2Array->Delete();
368  boundaryExtractor->Delete();
369 
370  return 1;
371 }
