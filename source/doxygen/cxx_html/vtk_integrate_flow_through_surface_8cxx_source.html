98 //-----------------------------------------------------------------------------
99 int vtkIntegrateFlowThroughSurface::RequestData(
100  vtkInformation *request,
101  vtkInformationVector **inputVector,
102  vtkInformationVector *outputVector)
103 {
104  // get the info objects
105  vtkInformation *inInfo = inputVector[0]->GetInformationObject(0);
106  vtkInformation *outInfo = outputVector->GetInformationObject(0);
107 
108  // get the input and output
109  vtkSmartPointer<vtkDataObject> input = inInfo->Get(vtkDataObject::DATA_OBJECT());
110 
111  vtkDataSet *dsInput = vtkDataSet::SafeDownCast(
112  inInfo->Get(vtkDataObject::DATA_OBJECT()));
113  vtkUnstructuredGrid *output = vtkUnstructuredGrid::SafeDownCast(
114  outInfo->Get(vtkDataObject::DATA_OBJECT()));
115 
116  vtkIntegrateAttributes* integrate = vtkIntegrateAttributes::New();
117  vtkCompositeDataSet *hdInput = vtkCompositeDataSet::SafeDownCast(
118  inInfo->Get(vtkDataObject::DATA_OBJECT()));
119  if (hdInput)
120  {
121  vtkMultiBlockDataSet* hds = vtkMultiBlockDataSet::New();
122  vtkCompositeDataIterator* iter = hdInput->NewIterator();
123  iter->GoToFirstItem();
124  while (!iter->IsDoneWithTraversal())
125  {
126  vtkDataSet* ds = vtkDataSet::SafeDownCast(iter->GetCurrentDataObject());
127  if (ds)
128  {
129  vtkDataSet* intermData = this->GenerateSurfaceVectors(ds);
130  if (intermData)
131  {
132  hds->SetBlock(hds->GetNumberOfBlocks(), intermData);
133  intermData->Delete();
134  }
135  }
136  iter->GoToNextItem();
137  }
138  iter->Delete();
139  inInfo->Set(vtkDataObject::DATA_OBJECT(), hds);
140  hds->Delete();
141  }
142  else if (dsInput)
143  {
144  vtkDataSet* intermData = this->GenerateSurfaceVectors(dsInput);
145  if (!intermData)
146  {
147  return 0;
148  }
149  inInfo->Set(vtkDataSet::DATA_OBJECT(), intermData);
150  intermData->Delete();
151  }
152  else
153  {
154  if (input)
155  {
156  vtkErrorMacro("This filter cannot handle input of type: "
157  << input->GetClassName());
158  }
159  return 0;
160  }
161 
162  integrate->ProcessRequest(request, inputVector, outputVector);
163 
164  if (hdInput)
165  {
166  inInfo->Set(vtkDataObject::DATA_OBJECT(), hdInput);
167  }
168  else if (dsInput)
169  {
170  inInfo->Set(vtkDataObject::DATA_OBJECT(), dsInput);
171  }
172 
173  vtkDataArray* flow = output->GetPointData()->GetArray("Perpendicular Scale");
174  if (flow)
175  {
176  flow->SetName("Surface Flow");
177  }
178 
179  integrate->Delete();
180  integrate = 0;
181 
182  return 1;
183 }
