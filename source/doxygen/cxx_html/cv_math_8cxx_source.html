77 int cvMath::linearInterpolate(double **orgPts, int numOrgPts, double t0,
78  double dt, int numOutPts, double ***rtnOutPts) {
79 
80  // This method takes an original set of points and returns a
81  // newly allocated set of interpolated points (where the requested
82  // number of points is numOutPts). Linear interpolation is used, and
83  // the values outside of the range of orgPts are fixed to the values
84  // at t_0 and t_n.
85 
86  int i,j;
87 
88  if (numOrgPts <= 0 || numOutPts <= 0) {
89  return CV_ERROR;
90  }
91 
92  double **outPts = createArray(numOutPts,2);
93  if (*outPts == NULL) {
94  return CV_ERROR;
95  }
96 
97  double t;
98 
99  for (i=0; i < numOutPts; i++) {
100 
101  t = t0 + dt*i;
102 
103  outPts[i][0] = t;
104 
105  // if time is outside of data range, fix to values at beginning
106  // and end of interval
107  if (t <= orgPts[0][0]) {
108  outPts[i][1] = orgPts[0][1];
109  continue;
110  } else if (t >= orgPts[numOrgPts-1][0]) {
111  outPts[i][1] = orgPts[numOrgPts-1][1];
112  continue;
113  }
114 
115  // interpolate
116  for (j = 1; j < numOrgPts; j++) {
117  if (t < orgPts[j][0]) {
118  double m = (orgPts[j][1]-orgPts[j-1][1])/(orgPts[j][0]-orgPts[j-1][0]);
119  outPts[i][1] = m*(t - orgPts[j-1][0]) + orgPts[j-1][1];
120  break;
121  }
122  }
123 
124  if (j == numOrgPts) {
125  fprintf(stdout,"Error interpolating point %i.\n",i);
126  deleteArray(outPts,numOutPts,2);
127  return CV_ERROR;
128  }
129  }
130 
131  // debug
132  //for (i = 0; i < numOutPts; i++) {
133  // fprintf(stdout,"%i: %8.3lf %8.3lf\n",i,outPts[i][0],outPts[i][1]);
134  //}
135 
136  *rtnOutPts = outPts;
137 
138  return CV_OK;

144 void cvMath::FFT(double data[],int nn,int isign) {
145 
146  int n,mmax,m,j,istep,i;
147  double wtemp,wr,wpr,wpi,wi,theta;
148  double tempr,tempi;
149 
150  n=nn << 1;
151  j=1;
152  for (i=1;i<n;i+=2) {
153  if (j > i) {
154  SWAP(data[j-1],data[i-1]);
155  SWAP(data[j+1-1],data[i+1-1]);
156  }
157  m=n >> 1;
158  while (m >= 2 && j > m) {
159  j -= m;
160  m >>= 1;
161  }
162  j += m;
163  }
164  mmax=2;
165  while (n > mmax) {
166  istep=2*mmax;
167  theta=6.28318530717959/(isign*mmax);
168  wtemp=sin(0.5*theta);
169  wpr = -2.0*wtemp*wtemp;
170  wpi=sin(theta);
171  wr=1.0;
172  wi=0.0;
173  for (m=1;m<mmax;m+=2) {
174  for (i=m;i<=n;i+=istep) {
175  j=i+mmax;
176  tempr=wr*data[j-1]-wi*data[j+1-1];
177  tempi=wr*data[j+1-1]+wi*data[j-1];
178  data[j-1]=data[i-1]-tempr;
179  data[j+1-1]=data[i+1-1]-tempi;
180  data[i-1] += tempr;
181  data[i+1-1] += tempi;
182  }
183  wr=(wtemp=wr)*wpr-wi*wpi+wr;
184  wi=wi*wpr+wtemp*wpi+wi;
185  }
186  mmax=istep;
187  }
188 }

297 double cvMath::compute_velocity (int k, double omega, double time, double radmax,
298  double alpha_k, double Y, double Qk[]) {
299 
300  double z1[2],z2[2],z3[2],z4[2],z5[2],z6[2],z7[2],z8[2],z9[2],z10[2],v_kwt;
301  double Pi = 3.1415926535;
302 
303 
304  ber_bei(0, Y*alpha_k, z1);
305  ber_bei(0, alpha_k, z2);
306  complex_div(z1,z2,z3);
307  z3[0] = 1 - z3[0];
308  z3[1] = -z3[1];
309  ber_bei(1, alpha_k, z4);
310  ber_bei(0, alpha_k, z5);
311  z6[0] = -(alpha_k*sqrt(2.0))/2.0;
312  z6[1] = (alpha_k*sqrt(2.0))/2.0;
313  complex_mult(z5,z6,z7);
314  complex_div(z4,z7,z8);
315  z8[0] = 1 - 2.0*z8[0];
316  z8[1] = -2.0*z8[1];
317  complex_div(z3,z8,z9);
318  complex_mult(Qk,z9,z10);
319  // note Charley's original code contains an error here, and
320  // the sign should be negative instead of positive as it
321  // was in his code
322  v_kwt = (z10[0]*cos(k*omega*time) - z10[1]*sin(k*omega*time))/(Pi*radmax*radmax);
323 
324  return v_kwt;

329 int cvMath::ber_bei (int order, double x, double z[]) {
330 
331  int k,kmax=30;
332  double toler=1.0e-4,z_old[2];
333  double Pi = 3.1415926535898;
334  double x_term, arg,ca,sa,mu,mod,phase,sqrt2;
335 
336  z[0] = 0.0;
337  z[1] = 0.0;
338 
339  if (fabs(x) < 15.0){
340  for (k=0;k<=kmax;k++) {
341  z_old[0] = z[0];
342  z_old[1] = z[1];
343  if (fabs(x) < toler && k==0) {
344  x_term = 1.0;
345  } else {
346  x_term = pow((x*x/4.0),k);
347  }
348  arg = (3.0*order/4.0 + k/2.0)*Pi;
349  ca = cos(arg);
350  sa = sin(arg);
351  z[0] = z[0] + (ca/factrl(k))*(x_term/factrl(order+k));
352  z[1] = z[1] + (sa/factrl(k))*(x_term/factrl(order+k));
353  }
354 
355  if (fabs((z_old[0]-z[0])/z_old[0])>toler || fabs((z_old[1]-z[1])/z_old[1])>toler) {
356 
357  printf(" ***************************** \n");
358  printf(" Error in ber_bei function \n");
359  printf(" value did not converge !!! \n");
360  printf(" ***************************** \n");
361 
362  }
363  if (fabs(x) < toler && order==0) {
364  /* do nothing */
365  } else {
366  z[0] = z[0]*pow((x/2.0),order);
367  z[1] = z[1]*pow((x/2.0),order);
368  }
369  } else {
370 
371  sqrt2 = sqrt(2.0);
372  mu = 4*order*order;
373  mod = 1.0 - (mu-1.0)/(8.0*sqrt2*x) + (mu-1)*(mu-1)/(256.0*x*x)
374  -(mu-1.0)*(mu*mu + 14.0*mu - 399.0)/(6144.0*sqrt2*x*x*x);
375  mod = mod*(exp(x/sqrt2)/(pow(2.0*Pi*x,0.5)));
376  phase = x/sqrt2 + (order/2.0 - 0.125)*Pi + (mu-1.0)/(8.0*sqrt2*x) + (mu-1)/(16.0*x*x)
377  -(mu-1.0)*(mu-25.0)/(384.0*sqrt2*x*x*x);
378  z[0] = mod*cos(phase);
379  z[1] = mod*sin(phase);
380  }
381 
382 
383 
384 
385  return 0;
386 }

441 
442  // This method takes an original set of points and returns the length
443  // of the line 2-D line.
444 
445  // If you specify closed == 1, the curve is assumed to be closed
446  // and the distance between the last point and the first is included
447  // in the value returned.
448 
449  if (numPts <= 1) {
450  *length = 0;
451  return CV_ERROR;
452  }

456  numSegments=numPts-1;

463  int j=i+1;

465  j = 0;

469  (pts[j][2]-pts[i][2])*(pts[j][2]-pts[i][2]) );
470  }
471 
472  *length = result;
473  return CV_OK;

478 int cvMath::linearInterpolateCurve(double **orgPts, int numOrgPts, int closed,
479  int numOutPts, double ***rtnOutPts) {
480 
481  // This method takes an original set of points and returns a
482  // newly allocated set of interpolated points (where the requested
483  // number of points is numOutPts). Linear interpolation between the points
484  // of the 3D curve is used.
485 
486  if (numOrgPts <= 1 || numOutPts <= 2) {
487  return CV_ERROR;
488  }
489 
490  // find the length of the curve
491  double length = 0;
492  curveLength(orgPts,numOrgPts,closed,&length);
493 
494  // now do linear interpolation of each coordinate
495  double **xin = createArray(numOrgPts+1,2);
496  double **yin = createArray(numOrgPts+1,2);
497  double **zin = createArray(numOrgPts+1,2);
498 
499  int i;
500  double t = 0;
501  for (i=0;i < numOrgPts;i++) {
502  xin[i][0]=t;xin[i][1]=orgPts[i][0];
503  yin[i][0]=t;yin[i][1]=orgPts[i][1];
504  zin[i][0]=t;zin[i][1]=orgPts[i][2];
505  int j = i+1;
506  if (j == numOrgPts) {
507  j = 0;
508  }
509  t += sqrt( (orgPts[j][0]-orgPts[i][0])*(orgPts[j][0]-orgPts[i][0]) +
510  (orgPts[j][1]-orgPts[i][1])*(orgPts[j][1]-orgPts[i][1]) +
511  (orgPts[j][2]-orgPts[i][2])*(orgPts[j][2]-orgPts[i][2]) );
512  }
513 
514  int numPts = numOrgPts;
515  double dt = length / (numOutPts-1);
516  if (closed != 0) {
517  xin[numOrgPts][0]=length;xin[numOrgPts][1]=orgPts[0][0];
518  yin[numOrgPts][0]=length;yin[numOrgPts][1]=orgPts[0][1];
519  zin[numOrgPts][0]=length;zin[numOrgPts][1]=orgPts[0][2];
520  numPts++;
521  dt = length / numOutPts;
522  }
523 
524  // now do linear interpolation on each coordinate
525  double **xout = createArray(numOutPts,2);
526  double **yout = createArray(numOutPts,2);
527  double **zout = createArray(numOutPts,2);
528 
529  if (linearInterpolate(xin, numPts, 0, dt, numOutPts, &xout) == CV_ERROR) {
530  deleteArray(xin,numOrgPts+1,2); deleteArray(xout,numOutPts,2);
531  deleteArray(yin,numOrgPts+1,2); deleteArray(yout,numOutPts,2);
532  deleteArray(zin,numOrgPts+1,2); deleteArray(zout,numOutPts,2);
533  return CV_ERROR;
534  }
535  if (linearInterpolate(yin, numPts, 0, dt, numOutPts, &yout) == CV_ERROR) {
536  deleteArray(xin,numOrgPts+1,2); deleteArray(xout,numOutPts,2);
537  deleteArray(yin,numOrgPts+1,2); deleteArray(yout,numOutPts,2);
538  deleteArray(zin,numOrgPts+1,2); deleteArray(zout,numOutPts,2);
539  return CV_ERROR;
540  }
541  if (linearInterpolate(zin, numPts, 0, dt, numOutPts, &zout) == CV_ERROR) {
542  deleteArray(xin,numOrgPts+1,2); deleteArray(xout,numOutPts,2);
543  deleteArray(yin,numOrgPts+1,2); deleteArray(yout,numOutPts,2);
544  deleteArray(zin,numOrgPts+1,2); deleteArray(zout,numOutPts,2);
545  return CV_ERROR;
546  }
547 
548  // put it all back together
549  double **outPts = createArray(numOutPts,3);
550  if (*outPts == NULL) {
551  deleteArray(xin,numOrgPts+1,2); deleteArray(xout,numOutPts,2);
552  deleteArray(yin,numOrgPts+1,2); deleteArray(yout,numOutPts,2);
553  deleteArray(zin,numOrgPts+1,2); deleteArray(zout,numOutPts,2);
554  return CV_ERROR;
555  }
556  for (i = 0; i < numOutPts;i++) {
557  outPts[i][0]=xout[i][1];
558  outPts[i][1]=yout[i][1];
559  outPts[i][2]=zout[i][1];
560  }
561 
562  // debug
563  //for (i = 0; i < numOutPts; i++) {
564  // fprintf(stdout,"%i: %8.3lf %8.3lf %8.3lf\n",i,outPts[i][0],outPts[i][1],outPts[i][2]);
565  //}
566 
567  // clean up
568  deleteArray(xin,numOrgPts+1,2); deleteArray(xout,numOutPts,2);
569  deleteArray(yin,numOrgPts+1,2); deleteArray(yout,numOutPts,2);
570  deleteArray(zin,numOrgPts+1,2); deleteArray(zout,numOutPts,2);
571 
572  *rtnOutPts = outPts;
573 
574  return CV_OK;
575 
576 }

593 int cvMath::smoothCurve(double **orgPts, int numOrgPts, int closed, int keepNumModes,

595 
596  // This method takes an original set of points and returns a
597  // newly allocated set of interpolated points (where the requested
