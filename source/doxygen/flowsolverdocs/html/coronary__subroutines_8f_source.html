140  subroutine corint(ctime,Plvist)
141 
142  use convolcorflow ! brings numCORSrfs, ValuePlvist
143  include "global.h"
144  include "common_blocks/nomodule.h"
145 
146  real*8 ctime, ptime, wr
147  integer nlast, nper, k, j
148  real*8 plvist(0:maxsurf)
149 c print *, 'I am in the CORint subroutine'
150 c print *, 'numCORSrfs',numCORSrfs
151  do k =1,numcorsrfs
152  nlast=numcort(k) ! number of time series to interpolate from
153  nper=ctime/valueplvist(nlast,1,k) ! number of periods completed to shift off
154  ptime = ctime-nper*valueplvist(nlast,1,k) ! now time in periodic domain
155 
156  do j=2,nlast !loop to find the interval that we are in
157  if(valueplvist(j,1,k).gt.ptime) then ! this is upper bound, j-1 is lower
158  wr=(ptime-valueplvist(j-1,1,k))
159  & / ( valueplvist(j,1,k)-valueplvist(j-1,1,k) )
160  plvist(k)= valueplvist(j-1,2,k)*(one-wr)
161  & + valueplvist(j,2,k)*wr
162  exit
163  endif
164  enddo
165 c print *, 'Plvist(',k,') = ', Plvist(k)
166  enddo
167  return
168  end

173  subroutine calccoric(y,srfIdList,numSrfs)
174 
175  use convolcorflow
176 
177  include "global.h"
178  include "common_blocks/timdat.h"
179  include "common_blocks/conpar.h"
180  include "common_blocks/inpdat.h"
181 
182  integer srfidlist(0:maxsurf), numsrfs, irankcoupled, k
183  real*8 y(nshg,4) !need velocity and pressure
184  real*8 couplearea(0:maxsurf)
185  real*8 velonly(nshg,3), ponly(nshg)
186  real*8 qini(0:maxsurf), pini(0:maxsurf)
187  real*8 plvistini(0:maxsurf)
188 
189  call calccor()
190  call calccoefcor()
191  ponly(:)= one ! one to get area
192  call integrscalar(couplearea,ponly,srfidlist,numsrfs) !get surf area
193  corarea(1:numsrfs)=couplearea(1:numsrfs)
194  call corint(delt(1)*lstep,plvistini)
195  coric = zero
196  if (lstep .eq. zero) then
197  velonly(:,1:3)=y(:,1:3)
198  call getflowq(qini,velonly,srfidlist,numsrfs) !get initial flow
199  qhistcor(1,1:numsrfs)=qini(1:numsrfs)
200  ponly(:)=y(:,4) ! pressure
201  call integrscalar(pini,ponly,srfidlist,numsrfs) !get initial pressure integral
202  phistcor(1,1:numsrfs) = pini(1:numsrfs)/corarea(1:numsrfs)
203  pini(1:numsrfs)=pini(1:numsrfs)/corarea(1:numsrfs)
204  plvhistcor(1,1:numsrfs)=plvistini(1:numsrfs)
205  elseif (lstep .gt. zero) then
206  qini(1:numsrfs) = qhistcor(1,1:numsrfs)
207  pini(1:numsrfs) = phistcor(1,1:numsrfs)
208  plvistini(1:numsrfs)=plvhistcor(1,1:numsrfs)
209  endif
210 
211  do k=1, numsrfs
212  coric(1,k) = (valuelistcor(6,k)*dpinidt(k)-
213  & valuelistcor(6,k)*cor(2,k)*pini(k)-
214  & valuelistcor(3,k)*dqinidt(k)-
215  & (valuelistcor(3,k)*cor(1,k)+
216  & valuelistcor(2,k))*qini(k)-
217  & valuelistcor(8,k)*plvistini(k))
218  & /detcor(k)
219  coric(2,k) = (valuelistcor(6,k)*dpinidt(k)-
220  & valuelistcor(6,k)*cor(1,k)*pini(k)-
221  & valuelistcor(3,k)*dqinidt(k)-
222  & (valuelistcor(3,k)*cor(2,k)+
223  & valuelistcor(2,k))*qini(k)-
224  & valuelistcor(8,k)*plvistini(k))
225  & /detcor(k)
226  enddo
227 
228  return
229  end

234  subroutine calccoefcor()
235 
236  use convolcorflow
237 
238  include "global.h"
239  include "common_blocks/nomodule.h"
240  include "common_blocks/inpdat.h"
241 
242  INTEGER k
243 
244  do k=1, numcorsrfs
245  coefcor(1,k)= (valuelistcor(3,k)*
246  & cor(1,k)*cor(1,k)
247  & +valuelistcor(2,k)*cor(1,k)+
248  & valuelistcor(1,k))/detcor(k)
249  coefcor(2,k)=(valuelistcor(3,k)*
250  & cor(2,k)*cor(2,k)
251  & +valuelistcor(2,k)*cor(2,k)+
252  & valuelistcor(1,k))/detcor(k)
253  coefcor(3,k)=(valuelistcor(8,k)*cor(1,k)+
254  & valuelistcor(7,k))
255  & /detcor(k)
256  coefcor(4,k)=(valuelistcor(8,k)*cor(2,k)+
257  & valuelistcor(7,k))
258  & /detcor(k)
259  coefcor(5,k)=valuelistcor(3,k)/valuelistcor(6,k)
260  enddo
261 
262  return
263  end
264 c
265 c... calculates dtCOR, the exponents if the exponentials for the Coronary BC
266 c
267  subroutine calccor()
268 
269  use convolcorflow ! brings ValueListCOR
270 
271  include "global.h"
272  include "common_blocks/nomodule.h"
273  include "common_blocks/inpdat.h"
274 
275  INTEGER k
276 
277  do k=1, numcorsrfs
278  detcor(k)=sqrt(valuelistcor(5,k)*valuelistcor(5,k)
279  & -four*valuelistcor(4,k)*valuelistcor(6,k))
280  cor(2,k)=-(valuelistcor(5,k)+detcor(k))/
281  & two/valuelistcor(6,k)
282  cor(1,k)= valuelistcor(4,k)/
283  & valuelistcor(6,k)/cor(2,k)
284  dtcor(1,k)=delt(1)*cor(1,k)
285  dtcor(2,k)=delt(1)*cor(2,k)
286  enddo
287 
288  return
289  end

305  if (stepn. eq. 0) then
306  corconvcoef(1,:)=real(coefcor(1,:)/
307  & dtcor(1,:)/cor(1,:)/alfi*
308  & ((-one+alfi*dtcor(1,:))*
309  & exp(dtcor(1,:)*alfi)+one)
310  & - coefcor(2,:)/
311  & dtcor(2,:)/cor(2,:)/alfi*
312  & ((-one+alfi*dtcor(2,:))*
313  & exp(dtcor(2,:)*alfi)+one))
314  corconvcoef(2,:)= real(CoefCOR(5,:) +
315  & coefcor(1,:)/
316  & dtcor(1,:)/cor(1,:)/alfi*
317  & (exp(dtcor(1,:)*alfi)-
318  & (one+alfi*dtcor(1,:)))-
319  & coefcor(2,:)/dtcor(2,:)/
320  & cor(2,:)/alfi*
321  & (exp(dtcor(2,:)*alfi)-
322  & (one+alfi*dtcor(2,:))))
323  endif
324  if (stepn. ge. 1) then
325  corconvcoef(1,:) =real(coefcor(1,:)/
326  & dtcor(1,:)/cor(1,:)*
327  & (exp(dtcor(1,:)*
328  & (stepn+alfi-one))-
329  & (one-dtcor(1,:))*
330  & exp(dtcor(1,:)*(stepn+alfi)))
331  & -coefcor(2,:)/dtcor(2,:)/cor(2,:)*
332  & (exp(dtcor(2,:)*(stepn+alfi-one))-

334  & exp(dtcor(2,:)*(stepn+alfi))))

336  & dtcor(1,:)/cor(1,:)
337  & /alfi*(alfi*exp(dtcor(1,:)*
338  & (alfi+one))-
339  & (alfi+one)*exp(dtcor(1,:)*
340  & (alfi))+one)-
341  & coefcor(2,:)/dtcor(2,:)/
342  & cor(2,:)/alfi*
343  & (alfi*exp(dtcor(2,:)*
344  & (alfi+one))-
345  & (alfi+one)*exp(dtcor(2,:)
346  & *(alfi))+one))
347  corconvcoef(stepn+2,:) = real(coefcor(5,:)+
348  & coefcor(1,:)/
349  & dtcor(1,:)/cor(1,:)/alfi*
350  & (exp(dtcor(1,:)*alfi)-
351  & one-alfi*dtcor(1,:))-
352  & coefcor(2,:)/dtcor(2,:)
353  & /cor(2,:)/alfi*
354  & (exp(dtcor(2,:)*alfi)
355  & -one-alfi*dtcor(2,:)))
356  endif
357  if (stepn. ge. 2) then
358  do j=2,stepn
359  corconvcoef(j,:) = real(coefcor(1,:)/
360  & dtcor(1,:)/cor(1,:)*
361  & (exp(dtcor(1,:)*
362  & (stepn+alfi-j))-
363  & two*exp(dtcor(1,:)*
364  & (stepn+alfi-j+one))+
365  & exp(dtcor(1,:)*
366  & (stepn+alfi-j+two)))-
367  & coefcor(2,:)/
368  & dtcor(2,:)/cor(2,:)*
369  & (exp(dtcor(2,:)*
370  & (stepn+alfi-j))-
371  & two*exp(dtcor(2,:)*
372  & (stepn+alfi-j+one))+
373  & exp(dtcor(2,:)*
374  & (stepn+alfi-j+two))))
375  enddo
376  endif
377 
378  return
379  end
