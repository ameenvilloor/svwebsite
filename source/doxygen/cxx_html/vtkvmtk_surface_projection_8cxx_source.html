66  int subId;

73  if (!this->ReferenceSurface)

80

82

84  vtkPointData* referencePointData = this->ReferenceSurface->GetPointData();
85 
86  outputPointData->InterpolateAllocate(referencePointData,numberOfPoints);
87 
88  this->ReferenceSurface->BuildCells();
89 
90  locator = vtkCellLocator::New();
91  genericCell = vtkGenericCell::New();
92 
93  locator->SetDataSet(this->ReferenceSurface);
94  locator->BuildLocator();
95 
96  for (i=0; i<numberOfPoints; i++)
97  {
98  input->GetPoint(i,point);
99  locator->FindClosestPoint(point,closestPoint,genericCell,cellId,subId,distance2);
100  if (this->ReferenceSurface->GetCellType(cellId) != VTK_POLY_LINE)
101  {
102  double* weights = new double[genericCell->GetNumberOfPoints()];
103  genericCell->EvaluatePosition(closestPoint,NULL,subId,pcoords,distance2,weights);
104  outputPointData->InterpolatePoint(referencePointData,i,genericCell->GetPointIds(),weights);
105  delete[] weights;
106  }
107  else
108  {
109  double testPoint0[3], testPoint1[3];
110  vtkIdType pointId = genericCell->GetPointId(subId);
111  input->GetPoint(genericCell->GetPointId(subId),testPoint0);
112  input->GetPoint(genericCell->GetPointId(subId+1),testPoint1);
113  if (vtkMath::Distance2BetweenPoints(closestPoint,testPoint1) < vtkMath::Distance2BetweenPoints(closestPoint,testPoint0))
114  {
115  pointId = genericCell->GetPointId(subId+1);
116  }
117  outputPointData->CopyData(referencePointData,pointId,i);
118  }
119  }
120 
121  locator->Delete();
122  genericCell->Delete();
123

125 }
