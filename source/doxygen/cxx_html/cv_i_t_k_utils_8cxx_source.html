182 void vtkGenerateCircle(double radius,double center[3],
183  int numPoints,vtkPolyData* circle)
184 {
185  vtkSmartPointer<vtkPoints> points = vtkSmartPointer<vtkPoints>::New();
186  vtkSmartPointer<vtkCellArray> lines = vtkSmartPointer<vtkCellArray>::New();
187  vtkIdType* lineIndices = new vtkIdType[numPoints+1];
188 
189  vtkSmartPointer<vtkPolyData> tmpc = vtkSmartPointer<vtkPolyData>::New();
190 
191  for( unsigned int i = 0; i < numPoints; i++ )
192  {
193  const double angle = 2.0 * vtkMath::Pi() *
194  static_cast< double >( i ) / static_cast< double >( numPoints );
195 
196  points->InsertPoint( static_cast< vtkIdType >( i ),
197  radius * cos( angle ),radius * sin( angle ), 0. );
198 
199  lineIndices[i] = static_cast< vtkIdType >( i );
200  }
201  lineIndices[numPoints] = 0;
202  lines->InsertNextCell( numPoints+1, lineIndices );
203  delete[] lineIndices;
204 
205  tmpc->SetPoints( points );
206  tmpc->SetLines( lines );
207 
208  vtkSmartPointer<vtkTransform> translation =
209  vtkSmartPointer<vtkTransform>::New();
210  translation->Translate(center[0], center[1], center[3]);
211 
212 
213  vtkSmartPointer<vtkTransformPolyDataFilter> transformFilter =
214  vtkSmartPointer<vtkTransformPolyDataFilter>::New();
215  transformFilter->SetInputData(tmpc);
216  transformFilter->SetTransform(translation);
217  transformFilter->Update();
218 
219  circle->DeepCopy(transformFilter->GetOutput());
220 }
