64 int cvMeshSystem::SetCurrentKernel(cvMeshObject::KernelType kernel_type)
65 {
66  switch (kernel_type) {
67  case cvMeshObject::KERNEL_INVALID:
68  gCurrentKernel = cvMeshObject::KERNEL_INVALID;
69 
70  case cvMeshObject::KERNEL_MESHSIM:
71  if (gMeshSystems[gCurrentKernel] != NULL)

73  gCurrentKernel = cvMeshObject::KERNEL_MESHSIM;

83  return CV_OK;

85  else
86  return CV_ERROR;
87 
88  default:
89  return CV_ERROR;
90  }
91  return CV_OK;

94 int cvMeshSystem::RegisterKernel( cvMeshObject::KernelType kernel_type, cvMeshSystem* pKernel )
95 {
96  if (kernel_type != cvMeshObject::KERNEL_MESHSIM && kernel_type != cvMeshObject::KERNEL_TETGEN)
97  return CV_ERROR;
98 
99  if (kernel_type == cvMeshObject::KERNEL_MESHSIM)
100  gCurrentKernel = cvMeshObject::KERNEL_MESHSIM;
101  else if (kernel_type == cvMeshObject::KERNEL_TETGEN)
102  gCurrentKernel = cvMeshObject::KERNEL_TETGEN;
103 
104  gMeshSystems[gCurrentKernel] = pKernel;
105  return CV_OK;

109 // ----------------------------
110 // DefaultInstantiateMeshObject
111 // ----------------------------
112 // This is the ONLY place that we should actually invoke construction
113 // of any concrete classes derived from cvMeshSystem. In particular,
114 // all the object method handling functions here should just call this
115 // function and use the returned cvMeshSystem*. Note that this function
116 // also takes in the interpreter so that an informative error message
117 // about kernel types can be returned if necessary.
118 
119 // PLEASE note that checking the return value of this function is
120 // CRITICAL for things to work right. If we return NULL here, we
121 // absolutely need to trap that in the callers (i.e. cvMeshSystem
122 // instantiation functions) to make sure those NULL ptr's don't get
123 // registered in the repository. Subsequent cvMeshSystem lookup's
124 // currently DO NOT check for NULL values. The idea is that objects
125 // are checked for validity *before* they get registered.
126 
127 cvMeshObject* cvMeshSystem::DefaultInstantiateMeshObject( Tcl_Interp *interp,
128  char *const meshFileName,
129  char *const solidFileName )
130 {
131  cvMeshSystem* meshSystem = NULL;
132  cvMeshObject* meshObject = NULL;
133  if (gCurrentKernel == cvMeshObject::KERNEL_MESHSIM || gCurrentKernel == cvMeshObject::KERNEL_TETGEN) {
134  if (gMeshSystems[gCurrentKernel] == NULL) {
135  Tcl_SetResult( interp, "current kernel is not available", TCL_STATIC );
136  return NULL;
137  }
138  meshSystem = gMeshSystems[gCurrentKernel];
139 
140  meshObject = (cvMeshObject *) (meshSystem->CreateMeshObject(interp));
141  if (meshObject == NULL) {
142  if (interp != NULL)
143  Tcl_SetResult( interp, "Unable to create mesh object", TCL_STATIC );
144  }
145  else {
146  meshObject->SetMeshFileName( meshFileName );
147  meshObject->SetSolidFileName( solidFileName );
148  }
149  }
150  else {
151  if (interp != NULL) {
152  Tcl_SetResult( interp, "current kernel is not valid", TCL_STATIC );
153  }
154  }
155 
156  return meshObject;
157 }
