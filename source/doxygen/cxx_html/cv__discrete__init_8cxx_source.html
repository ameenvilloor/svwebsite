64 int Meshsimdiscretesolid_Init( Tcl_Interp *interp )
65 {
66 
67  // Initialize discrete_utils
68  if (DiscreteUtils_Init() != CV_OK) {
69  return TCL_OK;
70  }
71 
72  // Get the solid model factory registrar from the main app.
73  cvFactoryRegistrar* solidModelRegistrar =
74  (cvFactoryRegistrar *) Tcl_GetAssocData( interp, "SolidModelRegistrar", NULL);
75 
76  if (solidModelRegistrar != NULL) {
77  // Register this particular factory method with the main app.
78  solidModelRegistrar->SetFactoryMethodPtr( SM_KT_DISCRETE,
79  (FactoryMethodPtr) &CreateMeshSimDiscreteSolidModel );
80 
81  Tcl_CreateCommand( interp, "discrete_available", MeshSimDiscrete_AvailableCmd,
82  (ClientData)NULL, (Tcl_CmdDeleteProc *)NULL );
83  }
84  else {
85  return TCL_ERROR;
86  }
87 
88  return TCL_OK;
89 }
