109 void cvParasolidSolidFromPolyData::GeometryfromVtkPolyData (int inNumShells) {
110 
111  int i;
112 
113  // NOTE: So every geometric entity has a unique number, the following is used:
114 
115  // shells: range from 0 to numShells-1
116  // faces: range from numShells to numShells+numFaces - 1
117  // loops: range from numShells+numFaces to numShells+numFaces+numLoops - 1
118  // edges: range from numShells+numFaces+numLoops to numShells+numFaces+numLoops+numEdges -1
119  // verts: range from numShells+numFaces+numLoops+numEdges to numShells+numFaces+numVerts+numLoops+numEdges - 1
120 
121  // create topologic relationships
122  inputPolyData->BuildCells();
123  inputPolyData->BuildLinks();
124 
125  numShells = inNumShells;
126  numVerts = inputPolyData->GetNumberOfPoints();
127  numFaces = inputPolyData->GetNumberOfPolys();
128  numPoints = numVerts;
129 
130  fprintf(stdout,"Extract edges from vtkPolyData.\n");
131  ExtractEdges();
132 
133  numEdges = inputPolyDataEdges->GetNumberOfLines();
134  numLines = numEdges;
135 
136  // the number of loops is equal to the number of
137  // faces. (note: we are assuming no holes in the
138  // facets!
139  int numLoops = numFaces;
140 
141  fprintf(stdout,"num verts: %i num edges: %i num faces: %i num loops: %i\n",numVerts,numEdges,numFaces,numLoops);
142 
143  // Allocate storage for vertices and lines
144  point_sf = new PK_POINT_sf_t [numVerts];
145  line_sf = new PK_LINE_sf_t [numEdges];
146  vertindex = new int[numVerts];
147  edgeindex = new int[numEdges];
148  faceindex = new int[numFaces];
149 
150  // Create vertices
151  int start_vert_num = numShells+numFaces+numLoops+numEdges;
152 
153  for (i = 0; i < numVerts;i++) {
154  vtkFloatingPointType* curPoint = inputPolyData->GetPoints()->GetPoint(i);
155  vertindex[i] = start_vert_num + i;
156  // find points for current vertex
157  point_sf[i].position.coord[0] = floor(100000.0*curPoint[0]+0.5)/100000.0;
158  point_sf[i].position.coord[1] = floor(100000.0*curPoint[1]+0.5)/100000.0;
159  point_sf[i].position.coord[2] = floor(100000.0*curPoint[2]+0.5)/100000.0;
160  }
161 
162  // edges
163  numLines = numEdges;
164  int start_edge_num=numShells+numFaces+numLoops;
165 
166  for (i = 0; i < numEdges;i++) {
167  edgeindex[i] = start_edge_num + i;
168 
169  // get verts of edge
170  vtkIdList* ptIds = inputPolyDataEdges->GetCell(i)->GetPointIds();
171  int pt1 = ptIds->GetId(0);
172  int pt2 = ptIds->GetId(1);
173 
174  // need to create a normalized direction vector
175  double V[3];
176  V[0] = point_sf[pt2].position.coord[0] - point_sf[pt1].position.coord[0];
177  V[1] = point_sf[pt2].position.coord[1] - point_sf[pt1].position.coord[1];
178  V[2] = point_sf[pt2].position.coord[2] - point_sf[pt1].position.coord[2];
179  double denom = sqrt(V[0]*V[0]+V[1]*V[1]+V[2]*V[2]);
180  V[0] = V[0] / denom; V[1] = V[1] / denom; V[2] = V[2] / denom;
181 
182  for (int j = 0; j < 3; j++) {
183  line_sf[i].basis_set.location.coord[j] = point_sf[pt1].position.coord[j];
184  line_sf[i].basis_set.axis.coord[j] = V[j];
185  }
186 
187  } // edges
188 
189  // faces
190  int start_face_num=numShells;
191  for (i = 0; i < numFaces;i++) {
192  faceindex[i] = start_face_num + i;
193  }
194 
195  // create parasolid list of geometric entities for topology
196  n_topols=numShells+numFaces+numLoops+numEdges+numVerts;
197  classes = new PK_CLASS_t [n_topols];
198 
199  for (i=0;i < numShells;i++) {
200  //fprintf(stdout,"%i: PK_SHELL\n",i);
201  classes[i]=PK_CLASS_shell;
202  }
203  for (i=0;i < numFaces;i++) {
204  //fprintf(stdout,"%i: PK_FACE\n",numShells+i);
205  classes[numShells+i]=PK_CLASS_face;
206  }
207  for (i=0;i < numLoops;i++) {
208  //fprintf(stdout,"%i: PK_LOOP\n",numShells+numFaces+i);
209  classes[numShells+numFaces+i]=PK_CLASS_loop;
210  }
211  for (i=0;i < numEdges;i++) {
212  int curedge = numShells+numFaces+numLoops+i;
213  classes[curedge]=PK_CLASS_edge;
214  //fprintf(stdout,"%i: PK_EDGE\n",curedge);
215  }
216  for (i=0;i < numVerts;i++) {
217  int curvert = numShells+numFaces+numLoops+numEdges+i;
218  classes[curvert]=PK_CLASS_vertex;
219  //fprintf(stdout,"%i: PK_VERT\n",curvert);
220  }
221 
222 }

225 void cvParasolidSolidFromPolyData::GetTopologyFromVtkPolyData() {
226 
227  int i;
228 
229  static char shelltoface[]="Shell to Face";
230  static char facetoloop[]="Face to Loop";
231  static char looptoedge[]="Loop to Edge";
232  static char edgetovertex[]="Edge to Vertex";
233 
234  // the number of loops is equal to the number of
235  // faces. (note: we are assuming no holes in the
236  // facets!
237  int numLoops = numFaces;
238 
239  // need to calculate the dimension of the topology array
240  // shell to face: numFaces relations
241  // face to loop: numLoops relations
242  // loop to edge: 2*numEdges
243  // edge to vertex: 2*numEdges
244 
245  int numRelations = numFaces + numLoops + 4 * numEdges;
246 
247  topstruct* topology = new topstruct [ numRelations ];
248 
249  int numloopssofar = 0;
250  int numTopRelations = 0;
251 
252  int loopFaces, loopEdges, loopLoops, p1, p2;
253  vtkIdList* ptIds;
254 
255  for (loopFaces=0;loopFaces < numFaces;loopFaces++) {
256 
257  //fprintf(stdout,"face: %i n_topols: %i\n",loopFaces,numTopRelations );
258 
259  if (numShells != 0) {
260  // Topologic relationship between shell and face
261  topology[numTopRelations].parent=0;
262  topology[numTopRelations].child=loopFaces+numShells;
263  topology[numTopRelations].comment=shelltoface;
264  topology[numTopRelations].sense=2;
265  numTopRelations++;
266  }
267 
268  int currentNumLoops = 1;
269 
270  for (loopLoops=0;loopLoops < currentNumLoops;loopLoops++) {
271 
272  //fprintf(stdout," loop: %i\n",loopLoops);
273 
274  // Topologic relationship between face and loop
275  topology[numTopRelations].parent=loopFaces+numShells;
276  topology[numTopRelations].child=numloopssofar+loopLoops+numShells+numFaces;
277  topology[numTopRelations].comment=facetoloop;
278  topology[numTopRelations].sense=2;
279  numTopRelations++;
280 
281  int nEdges = inputPolyData->GetCell(loopFaces)->GetNumberOfEdges();
282 
283  for (loopEdges = 0; loopEdges < nEdges; loopEdges++) {
284 
285  // fprintf(stdout," edge: %i\n",loopEdges);
286 
287  vtkCell* edge = inputPolyData->GetCell(loopFaces)->GetEdge(loopEdges);
288  p1 = edge->GetPointIds()->GetId(0);
289  p2 = edge->GetPointIds()->GetId(1);
290 
291  // edgenumber is the id for the edge in list of vtkPolyDataEdges
292  // brgorient corresponds to orientation of edge
293 
294  int edgenumber;
295  int brgorient;
296 
297  for (edgenumber = 0; edgenumber < numEdges;edgenumber++) {
298 
299  // get verts of edge
300  ptIds = inputPolyDataEdges->GetCell(edgenumber)->GetPointIds();
301  if ((ptIds->GetId(0) == p1) && (ptIds->GetId(1) == p2)) {
302  brgorient = 0;
303  break;
304  } else if ((ptIds->GetId(0) == p2) && (ptIds->GetId(1) == p1)) {
305  brgorient = 1;
306  break;
307  }
308 
309  } //edgenumber
310 
311  if (edgenumber == numEdges) {
312  fprintf(stderr," ERROR: Could not find edge given ptIds %i and %i.\n",p1,p2);
313  exit(-1);
314  }
315 
316  topology[numTopRelations].parent=numloopssofar+loopLoops+numShells+numFaces;
317  topology[numTopRelations].child=numShells+numFaces+numLoops+edgenumber;
318  topology[numTopRelations].sense=brgorient;
319  topology[numTopRelations].comment=looptoedge;
320  numTopRelations++;
321 
322  } //loopEdges
323 
324  numloopssofar++;
325 
326  } // loopLoops
327 
328  } //loopFaces
329 
330  // Topologic relationship of edges to vertices
331  for (loopEdges = 0; loopEdges < numEdges;loopEdges++) {
332  ptIds = inputPolyDataEdges->GetCell(loopEdges)->GetPointIds();
333  p1 = ptIds->GetId(0);
334  p2 = ptIds->GetId(1);
335 
336  topology[numTopRelations].parent= numShells+numFaces+numLoops+loopEdges;
337  topology[numTopRelations].child= numShells+numFaces+numLoops+numEdges+p1;
338  topology[numTopRelations].sense= 2;
339  topology[numTopRelations].comment= edgetovertex;
340  numTopRelations++;
341 
342  topology[numTopRelations].parent= numShells+numFaces+numLoops+loopEdges;
343  topology[numTopRelations].child= numShells+numFaces+numLoops+numEdges+p2;
344  topology[numTopRelations].sense= 2;
345  topology[numTopRelations].comment= edgetovertex;
346  numTopRelations++;
347  }
348 
349  fprintf(stdout,"Number of topologic relations: %i\n",numTopRelations);
350  // call bubble sort
351  //fprintf(stdout," Calling bubble sort! \n");
352  //fflush(stdout);
353  //numTopRelations=bubble(numTopRelations,topology);
354  //fprintf(stdout," Done bubble sort. \n");
355  //fflush(stdout);
356 
357  // define the number of relationships for the parasolid topology
358  n_relations = numTopRelations;
359  parents = new int [n_relations];
360  children = new int [n_relations];
361  senses = new PK_TOPOL_sense_t [n_relations];
362 
363  // fprintf(stdout,"n_relations: %i\n",n_relations);
364 
365  for (i=0;i<n_relations;i++) {
366  //fprintf(stdout,"parent: %5i child: %5i sense: %2i comment: %15s\n",
367  //topology[i].parent,topology[i].child,topology[i].sense,topology[i].comment);
368  parents[i]=topology[i].parent;
369  children[i]=topology[i].child;
370  if (topology[i].sense == 0) {
371  senses[i] = PK_TOPOL_sense_positive_c;
372  } else if (topology[i].sense == 1) {
373  senses[i] = PK_TOPOL_sense_negative_c;
374  } else if (topology[i].sense == 2) {
375  senses[i] = PK_TOPOL_sense_none_c;
376  }
377  }
378 
379  delete topology;
380 
381 } //GetTopologyFromVtkPolyData
