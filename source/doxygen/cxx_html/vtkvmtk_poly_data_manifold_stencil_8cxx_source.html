42  {
43  double point[3], point1[3], point2[3];
44  vtkIdType j, numberOfTriangles;
45 
46  if (this->DataSet==NULL)
47  {
48  vtkErrorMacro(<<"No DataSet specified.");

51 
52  this->DataSet->GetPoint(this->DataSetPointId,point);
53  this->Area = 0.0;
54 
55  if (!this->IsBoundary)
