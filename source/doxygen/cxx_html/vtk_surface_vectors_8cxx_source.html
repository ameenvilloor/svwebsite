65 //-----------------------------------------------------------------------------
66 // Generate normals for polygon meshesPrint
67 //----------------------------------------------------------------------------
68 int vtkSurfaceVectors::RequestData(vtkInformation *vtkNotUsed(request),
69  vtkInformationVector **inputVector,
70  vtkInformationVector *outputVector)
71 {
72  // get the info objects
73  vtkInformation *inInfo = inputVector[0]->GetInformationObject(0);
74  vtkInformation *outInfo = outputVector->GetInformationObject(0);
75 
76  // get the input and output
77  vtkDataSet *input = vtkDataSet::SafeDownCast(
78  inInfo->Get(vtkDataObject::DATA_OBJECT()));
79  vtkDataSet *output = vtkDataSet::SafeDownCast(
80  outInfo->Get(vtkDataObject::DATA_OBJECT()));
81 
82  vtkIdType numPoints, pointId, i, cellId;
83  numPoints = input->GetNumberOfPoints();
84  vtkDataArray *inVectors = this->GetInputArrayToProcess(0,inputVector);
85 
86  if(!inVectors)
87  {
88  output->ShallowCopy(input);
89  return 1;
90  }
91 
92  vtkDataArray* newVectors = 0;
93  vtkDoubleArray* newScalars = 0;
94  vtkIdList* cellIds = vtkIdList::New();
95  vtkIdList* ptIds = vtkIdList::New();
96  double p1[3];
97  double p2[3];
98  double p3[3];
99  double normal[3];
100  double tmp[3];
101  double v1[3], v2[3];
102  int count;
103  int cellType;
104 
105  // We could generate both ...
106  if(numPoints)
107  {
108  if (this->ConstraintMode == vtkSurfaceVectors::PerpendicularScale)
109  {
110  newScalars = vtkDoubleArray::New();
111  newScalars->SetNumberOfComponents(1);
112  newScalars->SetNumberOfTuples(numPoints);
113  newScalars->SetName("Perpendicular Scale");
114  }
115  else
116  {
117  newVectors = inVectors->NewInstance();
118  newVectors->SetNumberOfComponents(3);
119  newVectors->SetNumberOfTuples(numPoints);
120  newVectors->SetName(inVectors->GetName());
121  }
122  }
123 
124  for (pointId = 0; pointId < numPoints; ++pointId)
125  {
126  input->GetPointCells(pointId, cellIds);
127  // Compute the point normal.
128  count = 0;
129  normal[0] = normal[1] = normal [2] = 0.0;
130  for (i = 0; i < cellIds->GetNumberOfIds(); ++i)
131  {
132  cellId = cellIds->GetId(i);
133  cellType = input->GetCellType(cellId);
134  if (cellType == VTK_VOXEL || cellType == VTK_POLYGON ||
135  cellType == VTK_TRIANGLE || cellType == VTK_QUAD)
136  {
137  input->GetCellPoints(cellId, ptIds);
138  input->GetPoint(ptIds->GetId(0), p1);
139  input->GetPoint(ptIds->GetId(1), p2);
140  input->GetPoint(ptIds->GetId(2), p3);
141  v1[0] = p2[0] - p1[0];
142  v1[1] = p2[1] - p1[1];
143  v1[2] = p2[2] - p1[2];
144  v2[0] = p3[0] - p1[0];
145  v2[1] = p3[1] - p1[1];
146  v2[2] = p3[2] - p1[2];
147  vtkMath::Cross(v1, v2, tmp);
148  ++count;
149  normal[0] += tmp[0];
150  normal[1] += tmp[1];
151  normal[2] += tmp[2];
152  }
153  if (cellType == VTK_PIXEL)
154  {
155  input->GetCellPoints(cellId, ptIds);
156  input->GetPoint(ptIds->GetId(0), p1);
157  input->GetPoint(ptIds->GetId(1), p2);
158  input->GetPoint(ptIds->GetId(2), p3);
159  v1[0] = p2[0] - p1[0];
160  v1[1] = p2[1] - p1[1];
161  v1[2] = p2[2] - p1[2];
162  v2[0] = p3[0] - p1[0];
163  v2[1] = p3[1] - p1[1];
164  v2[2] = p3[2] - p1[2];
165  vtkMath::Cross(v2, v1, tmp);
166  ++count;
167  normal[0] += tmp[0];
168  normal[1] += tmp[1];
169  normal[2] += tmp[2];
170  }
171  }
172  double inVector[3];
173  inVectors->GetTuple(pointId, inVector);
174  double k = 0.0;
175  if (count > 0)
176  {
177  vtkMath::Normalize(normal);
178  k = vtkMath::Dot(normal, inVector);
179  if (this->ConstraintMode == vtkSurfaceVectors::Parallel)
180  {
181  // Remove non orthogonal component.
182  inVector[0] = inVector[0] - (normal[0]*k);
183  inVector[1] = inVector[1] - (normal[1]*k);
184  inVector[2] = inVector[2] - (normal[2]*k);
185  }
186  else if (this->ConstraintMode == vtkSurfaceVectors::Perpendicular)
187  { // Keep only the orthogonal component.
188  inVector[0] = normal[0]*k;
189  inVector[1] = normal[1]*k;
190  inVector[2] = normal[2]*k;
191  }
192  }
193  if (newScalars)
194  {
195  newScalars->InsertValue(pointId, k);
196  }
197  if (newVectors)
198  {
199  newVectors->InsertTuple(pointId, inVector);
200  }
201  }
202 
203  output->ShallowCopy(input);
204  if (newVectors)
205  {
206  output->GetPointData()->SetVectors(newVectors);
207  newVectors->Delete();
208  }
209  if (newScalars)
210  {
211  output->GetPointData()->SetScalars(newScalars);
212  newScalars->Delete();
213  }
214  cellIds->Delete();
215  ptIds->Delete();
216 
217  // Not implemented for data set.
218  //output->RemoveGhostCells
219  return 1;
220 }
