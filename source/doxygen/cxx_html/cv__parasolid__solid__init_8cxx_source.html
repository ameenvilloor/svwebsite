56 {
57  // Get the solid model factory registrar from the main app.
58  cvFactoryRegistrar* solidModelRegistrar =
59  (cvFactoryRegistrar *) Tcl_GetAssocData( interp, "SolidModelRegistrar", NULL);
60 
61  if (solidModelRegistrar != NULL) {
62  // Register this particular factory method with the main app.
63  solidModelRegistrar->SetFactoryMethodPtr( SM_KT_PARASOLID,
64  (FactoryMethodPtr) &CreateParasolidSolidModel );
65 
66  Tcl_CreateCommand( interp, "parasolid_available", Parasolid_AvailableCmd,
67  (ClientData)NULL, (Tcl_CmdDeleteProc *)NULL );
68  }
69  else {
70  return TCL_ERROR;
71  }
72
