300 // remove leading and trailing spaces (or tabs)
301 void Input::trim_string(string *str)
302 {
303  // check for empty string
304  int length = str->length();
305  if ( length == 0 )
306  return;
307 
308  // erase leading spaces (or tabs)
309  int pos0 = 0;
310  while ( (*str)[pos0] == ' ' || (*str)[pos0] == '\t') {
311  pos0++;
312  }
313  if ( pos0 > 0 ) {
314  str->erase(0,pos0);
315  }
316 
317  length = str->length();
318  if ( length == 0 )
319  return;
320 
321  pos0 = length-1;
322  // erase trailing spaces (or tabs)
323  while ( (*str)[pos0] == ' ' || (*str)[pos0] == '\t' || (*str)[pos0] == '\r') {
324  pos0--;
325  }
326  if ( pos0 < length-1 ) {
327  str->erase(pos0+1, length-pos0);
328  }
329 
330 }
