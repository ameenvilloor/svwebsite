155  * *
156  * sys_geom_SplineInterpolate *
157  * -------------------------- *
158  * Performs spline interpolation. *
159  * *
160  * Assumptions: Open spline. Results of interpolation not clamped *
161  * to input data. *

164 void sys_geom_SplineInterpolate(SplinePoints *input, int type, int numberOfOutputPoints, SplinePoints *output)
165 {
166 
167  vtkSpline *spline;
168  int i, j, k;
169  vtkFloatingPointType t;
170  int dimensions;
171 
172  /*****************
173  * body *
174  *****************/
175 
176  switch (type)
177  {
178  case 1: /* Cardinal basis */
179  spline = (vtkSpline *)vtkCardinalSpline::New();
180  break;
181 
182  case 2: /* Kochanek basis */
183  spline = (vtkSpline *)vtkKochanekSpline::New();
184  break;
185 
186  default:
187  cerr << " ERROR >>>> In sys_geom_SplineInterpolate(...)" << endl;
188  cerr << " Unknown interpolation method requested." << endl;
189  return;
190  }
191 
192 
193  dimensions = input->dim;
194 
195  if ((dimensions > MAX_DIM) || (dimensions < 1))
196  {
197  cerr << " ERROR >>>> In sys_geom_SplineInterpolate(...)" << endl;
198  cerr << " Invalid number of dimensions." << endl;
199  return;
200  }
201 
202  for (i = 0; i < dimensions; i++)
203  {
204  spline->RemoveAllPoints();
205 
206  /* Add points to spline object */
207  for (j = 0; j < (input->numPts); j++)
208  spline->AddPoint(j, input->pts[dimensions * j + i]);
209 
210  /* Set spline parameters */
211  spline->ClampValueOff();
212  spline->ClosedOff();
213  spline->Modified();
214 
215  /* Evaluate spline */
216  for (k = 0; k < numberOfOutputPoints; k++)
217  {
218  t = ((input->numPts - 1.0) / (numberOfOutputPoints - 1)) * k;
219 
220  switch (type)
221  {
222  case 1:
223  output->pts[dimensions * k + i] = ((vtkCardinalSpline *)spline)->Evaluate(t);
224  break;
225 
226  case 2:
227  output->pts[dimensions * k + i] = ((vtkKochanekSpline *)spline)->Evaluate(t);
228  break;
229  }
230  }
231 
232  }
233 
234  output->numPts = numberOfOutputPoints;
235  output->dim = dimensions;
236 
237  /* Clean up */
238  spline->Delete();
239 }

245  * -------------------------- *
246  * Computes the spline tangents and returns the tangents and their *
247  * corresponding points. *

250  * Approach is to compute spline for many, many points and calculate *
251  * the tangent using a finite difference. *

254 int sys_geom_SplineGetTangents(SplinePoints *input, int type, int numberOfOutputPoints, int matchEndPoints, SplinePoints *output)
255 {
256 
257  SplinePoints *tmpSpline;
258  int numberOfTangentPoints;
259  int i, j;
260  int interval;
261  int dimensions;
262  vtkFloatingPointType coordinate;
263  vtkFloatingPointType *tmpTangent = new vtkFloatingPointType [MAX_DIM];
264  vtkFloatingPointType *tmpNormTangent = new vtkFloatingPointType [MAX_DIM];
265  int err;
266 
267  /*****************
268  * body *
269  *****************/
270 
271 
272  /* Want a minimum of 500 points...this number was pulled out of my magical hat. */
273  if (numberOfOutputPoints < 50)
274  numberOfTangentPoints = (ceil(500.0 / numberOfOutputPoints)) * numberOfOutputPoints;
275  else
276  numberOfTangentPoints = 10 * numberOfOutputPoints;
277 
278 
279  tmpSpline = sys_geom_SplinePointsInit(numberOfTangentPoints, input->dim);
280  sys_geom_SplineInterpolate(input, type, numberOfTangentPoints, tmpSpline);
281 
282  if (matchEndPoints != 0)
283  interval = floor(1.0*numberOfTangentPoints / (numberOfOutputPoints - 1));
284  else
285  interval = floor(1.0*numberOfTangentPoints / numberOfOutputPoints);
286 
287  /* Assume input and output have same dimensions */
288  dimensions = output->dim;
289 
290 
291  for (i = 0; i < (numberOfOutputPoints - 1); i++)
292  {
293  for (j = 0; j < MAX_DIM; j++)
294  {
295  tmpTangent[j] = 0.0; /* Reset */
296  tmpNormTangent[j] = 0.0;
297  }
298 
299  for (j = 0; j < dimensions; j++)
300  {
301  coordinate = tmpSpline->pts[dimensions * i * interval + j];
302 
303  /* Copy point coordinates */
304  output->pts[dimensions * i + j] = coordinate;
305 
306  /* Compute forward difference */
307  tmpTangent[j] = tmpSpline->pts[dimensions * (i * interval + 1) + j] - coordinate;
308  }
309 
310 
311  /* Normalize tangent */
312 
313  err = sys_geom_NormalizeVector(tmpTangent, tmpNormTangent, MAX_DIM);
314  if (err == 0)
315  {
316  fprintf(stderr,"Subdivision too small for computing tangents. Decrease number of output points requested.\n");
317  return CV_ERROR;
318  }
319 
320  for (j = 0; j < dimensions; j++)
321  output->tangents[dimensions * i + j] = tmpNormTangent[j];
322  }
323 
324 
325  /*
326  * The last point just gets copied over and the tangent is based on the
327  * backwards difference.
328  */
329  for (j = 0; j < MAX_DIM; j++)
330  {
331  tmpTangent[j] = 0.0; /* Reset */
332  tmpNormTangent[j] = 0.0;
333  }
334 
335 
336  for (j = 0; j < dimensions; j++)
337  {
338 
339  if (matchEndPoints == 0)
340  coordinate = tmpSpline->pts[dimensions * (numberOfTangentPoints - 1) + j];
341  else
342  coordinate = input->pts[dimensions * (input->numPts - 1) + j];
343 
344 
345  output->pts[dimensions * (numberOfOutputPoints - 1) + j] = coordinate;
346 
347  tmpTangent[j] = coordinate - tmpSpline->pts[tmpSpline->dim * (tmpSpline->numPts - 2) + j];
348 
349  }
350 
351  /* Normalize tangent */
352  err = sys_geom_NormalizeVector(tmpTangent, tmpNormTangent, MAX_DIM);
353  if (err == 0)
354  {
355  fprintf(stderr,"Subdivision too small for computing tangents. Decrease number of output points requested.\n");
356  return CV_ERROR;
357  }
358 
359  for (j = 0; j < dimensions; j++)
360  output->tangents[dimensions * (numberOfOutputPoints - 1) + j] = tmpNormTangent[j];
361 
362  /* End code for handling last point */
363 
364  output->numPts = numberOfOutputPoints;
365  output->dim = dimensions;
366 
367 
368  /* Clean up */
369  sys_geom_SplinePointsDelete(tmpSpline);
370  delete[] tmpTangent;
371  delete[] tmpNormTangent;
372 
373  return CV_OK;
374 
375 }

379 /***********************************************************************
380  * *
381  * sys_geom_SplineGetRotVectors *
382  * ---------------------------- *
383  * Computes a rotational/orientation vector. In this case, it is just *
384  * a vector that is perpendicular to the tangent. *
385  * *
386  * We create a vector, with most of its coordinates equal to zero and *
387  * one coordinate equal to 1. This vector will be perpendicular to *
388  * the tangent, if the dot product of the two is zero. Compute the *
389  * value of the remaining coordinate using this fact. *
390  * *
391  * Assumes that the tangents for the SplinePoints has already been *
392  * calculated. *
393  ***********************************************************************/
394 
395 void sys_geom_SplineGetRotVectors(SplinePoints *input)
396 {
397  int i, j;
398  int dimensions;
399  vtkFloatingPointType *tmpRotVector = new vtkFloatingPointType[MAX_DIM];
400  vtkFloatingPointType *tmpNormRotVector = new vtkFloatingPointType[MAX_DIM];
401  vtkFloatingPointType dotProduct;
402 
403  /*****************
404  * body *
405  *****************/
406 
407  dimensions = input->dim;
408 
409  for (i = 0; i < input->numPts; i++)
410  {
411 
412  /* Reset */
413  for (j = 0; j < MAX_DIM; j++)
414  tmpRotVector[j] = 0;
415 
416  int replaceComp;
417 
418  if (fabs(input->tangents[dimensions * i + 2]) > 0.0001) {
419  tmpRotVector[1] = 1;
420  replaceComp = 2;
421  } else if (fabs(input->tangents[dimensions * i + 1]) > 0.0001) {
422  tmpRotVector[0] = 1;
423  replaceComp = 1;
424  } else {
425  tmpRotVector[2] = 1;
426  replaceComp = 0;
427  }
428 
429  dotProduct = 0;
430 
431  for (j = 0; j < (dimensions - 1); j++)
432  dotProduct = dotProduct + (input->tangents[dimensions * i + j] * tmpRotVector[j]);
433 
434  tmpRotVector[replaceComp] = -dotProduct / input->tangents[dimensions * i + replaceComp];
435 
436  /* Now normalize vector */
437  sys_geom_NormalizeVector(tmpRotVector, tmpNormRotVector, MAX_DIM);
438 
439 
440  /* Copy vector to input */
441  for (j = 0; j < dimensions; j++)
442  input->rotVectors[dimensions * i + j] = tmpNormRotVector[j];
443  }
444 
445  /* Clean up */
446  delete[] tmpRotVector;
447  delete[] tmpNormRotVector;
448 }

458 int sys_geom_NormalizeVector(vtkFloatingPointType *input, vtkFloatingPointType *output, int sizeVector)
459 {
460  vtkFloatingPointType magnitude;
461  int j;
462 
463  /*****************
464  * body *
465  *****************/
466 
467  magnitude = 0;
468 
469  for (j = 0; j < sizeVector; j++)
470  magnitude = magnitude + pow(input[j], 2);
471 
472  magnitude = sqrt(magnitude);
473 
474  if (magnitude == 0)
475  {
476  fprintf(stderr, "Vector cannot be normalized. Magnitude of 0.\n");
477  return 0;
478  }
479 
480  for (j = 0; j < sizeVector; j++)
481  output[j] = input[j] / magnitude;
482 
483  return 1;
484 }
