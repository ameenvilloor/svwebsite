285 static int FMMHelper(vtkStructuredPoints *sp, Tcl_Interp *interp, char* levelSetName, double start,
286  double stop, int nlistpts, ARG_List *indpts, int npts, int table_sz, ARG_Entry arg_table[])
287 {

289  double pt[3];
290  int numSeeds = 0;
291  double seeds[itker->SEEDS_ARRAY_MAX][3];
292 
293  for (int i = 0; i < nlistpts; i++) {
294  if ( ARG_ParseTclListStatic( interp, indpts[i], DOUBLE_Type, pt, 3, &npts )
295  != TCL_OK ) {
296  Tcl_SetResult( interp, "error in points list", TCL_VOLATILE );
297  ARG_FreeListArgvs( table_sz, arg_table );
298  delete [] indpts;
299  return TCL_ERROR;
300  }
301  if ( npts != 3 ) {
302  Tcl_SetResult( interp, "error in points list", TCL_VOLATILE );
303  ARG_FreeListArgvs( table_sz, arg_table );
304  delete [] indpts;
305  return TCL_ERROR;
306  }
307  if(i >= itker->SEEDS_ARRAY_MAX ){
308  Tcl_SetResult( interp, "too many seeds", TCL_VOLATILE );
309  ARG_FreeListArgvs( table_sz, arg_table );
310  delete [] indpts;
311  return TCL_ERROR;
312  }
313  seeds[i][0] = pt[0];
314  seeds[i][1] = pt[1];
315  seeds[i][2] = pt[2];
316  numSeeds++;
317  }
318 
319  delete [] indpts;
320 
321  ARG_FreeListArgvs( table_sz, arg_table );
322  vtkStructuredPoints *fmmImg;

335  char* newName = (char *) malloc(strlen(levelSetName) + 11);

708  int* dimensions = imageToPad->GetDimensions();
709  int xDim = dimensions[0];
710  int yDim = dimensions[1];

732  std::cerr<<"Could not interpret type of vtkStructuredPoints data."<<std::endl;
733  return;
734  }

738  int xDimNew = xDim + 2 * padding;
739  int yDimNew = yDim + 2 * padding;

744  struct MapToOneDArray {
745  MapToOneDArray(int dimx, int dimy): dimx(dimx), dimy(dimy){};
746 
747  int operator()(int x, int y, int z){
748  return x+y*dimx+z*dimx*dimy;

755  struct InPaddedRegionT {
756  InPaddedRegionT(int dimx, int dimy, int dimz, int padding)
757  : dimx(dimx), dimy(dimy), dimz(dimz), padding(padding){};
758 
759  bool operator()(int x, int y, int z){
760  return (x < padding || x >= dimx + padding) ||
761  (y < padding || y >= dimy + padding) ||
762  (z < padding || z >= dimz + padding);
763  }
764 
765  int dimx;
766  int dimy;
767  int dimz;
768  int padding;
769  };

772  MapToOneDArray oldMapper(xDim, yDim);
773

776  for( int x = 0; x < xDimNew; x++) {
777  for( int y = 0; y < yDimNew; y++) {

786  }
