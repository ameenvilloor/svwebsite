371  SUBROUTINE normcon (iens, y, xKebe, xGoC, rl)
372 
373 ! This is how the xKebe is formed
374 ! 1 2 3
375 ! 4 5 6
376 ! 7 8 9
377 
378  USE pvsqbi
379 
380  include "global.h"
381  include "common_blocks/propar.h"
382  include "common_blocks/shpdat.h"
383  include "common_blocks/conpar.h"
384  include "common_blocks/elmpar.h"
385  include "common_blocks/nomodule.h"
386 
387  INTEGER iens(npro,nshape), ie, ia, ib, iac, i, j, l, isrf
388 
389  REAL*8 norm(nsd,nsd), ndy, y(nshg, ndofl),
390  2 xkebe(npro,9,nshl,nshl),
391  3 k(nsd,nsd), nk(nsd,nsd), nknt(nsd,nsd),
392  4 xgoc(npro,4,nshl,nshl), g(nsd), ng(nsd),
393  5 rl(npro,nshl,nflow), r(nsd), nr(nsd)
394 
395  INTENT(IN) iens, y
396  INTENT(INOUT) xkebe, xgoc, rl
397 
398 c return
399  DO isrf = 1, numnormalsrfs
400  DO ie = 1, npro
401  DO ia = 1, nshl
402  iac = abs(iens(ie,ia))
403  IF (ndsurf(iac) .EQ. nsrflistnormal(isrf)) THEN
404  CALL getnorm(iac, norm)
405 
406  ndy = 0d0
407  DO i = 1, nsd
408  ndy = ndy + norm(1,i)*y(ia,i)
409  END DO
410  IF (.true.)then!ndy .LT. 0D0) THEN
411 c********************************************************************
412 c First modifing RHS matrix
413  DO i = 1, nsd
414  r(i) = rl(ie,ia,i)
415  END DO
416 c Here setting nR = n*R
417  DO i=1, nsd
418  nr(i) = 0d0
419  DO j=1,nsd
420  nr(i) = nr(i) + norm(i,j)*r(j)
421  END DO
422  END DO
423 
424 c Now setting second and third row equal to zero, since the normal to the face is
425 c in the first direction
426  DO i=2, nsd
427  nr(i) = 0d0
428  END DO
429 
430 c Rotating back the coordinate R = n^t*nR
431  DO i=1, nsd
432  r(i) = 0d0
433  DO j=1,nsd
434  r(i) = r(i) + norm(j,i)*nr(j)
435  END DO
436  END DO
437 
438 c Assigning the new residual vector
439  DO i = 1, nsd
440  rl(ie,ia,i) = r(i)
441  END DO
442 
443 c********************************************************************
444 c Now modifing LHS matrices
445 
446  DO ib = 1, nshl
447  DO i=1, nsd
448  g(i) = xgoc(ie,i,ia,ib)
449  DO j=1,nsd
450  k(i,j) = xkebe(ie,j+nsd*(i-1),ia,ib)
451  END DO
452  END DO
453 c
454 c Here setting nK = n*K and nG = n*G
455 c
456  DO i=1, nsd
457  ng(i) = 0d0
458  DO j=1,nsd
459  ng(i) = ng(i) + norm(i,j)*g(j)
460  nk(i,j) = 0d0
461  DO l=1, nsd
462  nk(i,j) = nk(i,j) + norm(i,l)*k(l,j)
463  END DO
464  END DO
465  END DO
466 c
467 c Here setting nKnt = nK*n^t
468 c
469  DO i=1, nsd
470  DO j=1,nsd
471  nknt(i,j) = 0d0
472  DO l=1, nsd
473  nknt(i,j) = nknt(i,j) +
474  2 nk(i,l)*norm(j,l)
475  END DO
476  END DO
477  END DO
478 c
479 c Now setting second and third row equal to zero, since the normal to the face is
480 c in the first direction and we dont want to constrain the velocity in normal direction,
481 c first row should be conserved. Note that diagonal element is not chaged to prevent
482 c singularity of stiffness matrix
483 c
484  DO i=2, nsd
485  IF (ia .NE. ib) THEN
486  ng(i) = 0d0
487  DO j=1,nsd
488  nknt(i,j) = 0d0
489  END DO
490  ELSE
491  ng(i) = 0d0
492  DO j=1,nsd
493  IF (i.NE.j) THEN
494  nknt(i,j) = 0d0
495  END IF
496  END DO
497  END IF
498  END DO
499 
500 c Rotating back the coordinate nK = nKnt*n and G = nt*nG
501 
502  DO i=1, nsd
503  g(i) = 0d0
504  DO j=1,nsd
505  g(i) = g(i) + norm(j,i)*ng(j)
506  nk(i,j) = 0d0
507  DO l=1, nsd
508  nk(i,j) = nk(i,j) + nknt(i,l)*norm(l,j)
509  END DO
510  END DO
511  END DO
512 c
513 c Rotating back the coordinate K = n^t*nK
514 c
515  DO i=1, nsd
516  DO j=1,nsd
517  k(i,j) = 0d0
518  DO l=1, nsd
519  k(i,j) = k(i,j) + norm(l,i)*nk(l,j)
520  END DO
521  END DO
522  END DO
523 c
524 c Assigning the new stiffness matrix
525 c
526  DO i=1, nsd
527  xgoc(ie, i, ia, ib) = g(i)
528  DO j=1,nsd
529  xkebe(ie, j + nsd*(i-1), ia, ib) = k(i,j)
530  END DO
531  END DO
532  END DO ! loop for another local node
533  END IF ! if clause for inward velocity
534  END IF ! if clause for nodes inside the face
535  END DO ! loop for the node
536  END DO ! loop over the elements
537  END DO ! loop over the faces
538 
539  RETURN
540 
541  END SUBROUTINE normcon

545  SUBROUTINE getnorm(iAc, norm)
546 
547  USE pvsqbi
548 
549  include "global.h"

551  INTEGER iac, i
552  REAL*8 normal(nsd), tang1(nsd), tang2(nsd), temp, norm(nsd, nsd)
553 
554  INTENT(IN) iac
555  INTENT(OUT) norm
556 
557  temp = 0d0
558  DO i=1,nsd
559  normal(i) = nabi(iac,i)
560  temp = temp + normal(i)**2d0
561  END DO
562  temp = sqrt(temp)
563  normal = normal/temp
564

566  tang1(1) = 0d0
567  tang1(2) = normal(3)

569 
570  temp = 0d0

581  tang2(2) = normal(3)*tang1(1) - normal(1)*tang1(3)
582  tang2(3) = normal(1)*tang1(2) - normal(2)*tang1(1)
583 
584  norm(1,:) = normal
585  norm(2,:) = tang1
586  norm(3,:) = tang2
