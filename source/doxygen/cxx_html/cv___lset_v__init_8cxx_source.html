332 int LsetVKGI_NewCmd( ClientData clientData, Tcl_Interp *interp,
333  int argc, CONST84 char *argv[] )
334 {
335  CONST84 char* vName;
336  cvLevelSetVelocityKGI *v;
337  Tcl_HashEntry *entryPtr;
338  int newEntry = 0;
339 
340  // Check syntax:
341  if (argc != 2) {
342  Tcl_AppendResult( interp, "usage: ", argv[0], " <objName>",
343  (char *)NULL );
344  return TCL_ERROR;
345  }
346 
347  // Make sure this is a new object name:
348  vName = argv[1];
349  if ( !NewName( vName, interp ) ) {
350  Tcl_AppendResult( interp, "object \"", vName,
351  "\" already exists", (char *)NULL );
352  return TCL_ERROR;
353  }
354 
355  // Allocate new cvLevelSetVelocity object (of class cvLevelSetVelocityKGI):
356  v = new cvLevelSetVelocityKGI;
357  if ( v == NULL ) {
358  Tcl_AppendResult( interp, "error allocating object \"", vName,
359  "\"", (char *)NULL );
360  return TCL_ERROR;
361  }
362 
363  strcpy( v->tclName_, vName );
364  entryPtr = Tcl_CreateHashEntry( &gLsetVTable, vName, &newEntry );
365  if ( !newEntry ) {
366  Tcl_SetResult( interp, "error updating LsetV hash table",
367  TCL_STATIC );
368  delete v;
369  return TCL_ERROR;
370  }
371  Tcl_SetHashValue( entryPtr, (ClientData)v );
372  char vNameStr[2048];
373  vNameStr[0]='\0';
374  sprintf(vNameStr,"%s",vName);
375  Tcl_SetResult( interp, vNameStr, TCL_VOLATILE );
376  Tcl_CreateCommand( interp, Tcl_GetStringResult(interp), LsetVKGI_ObjectCmd,
377  (ClientData)v, DeleteLsetV );
378  return TCL_OK;
379 }
