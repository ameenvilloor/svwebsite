123 
124  initializationArray = NULL;

137

140

142  this->TScalars->FillComponent(0,VTK_VMTK_LARGE_DOUBLE);
143 
144  this->NumberOfAcceptedPoints = 0;
145 
146  this->ConsideredMinHeap->SetMinHeapScalars(this->TScalars);
147  this->ConsideredMinHeap->Initialize();
148 
149  if (this->SeedsBoundaryConditions)

231  }
232

234  {
235  for (i=0; i<numberOfBoundaryPointIds; i++)

268 void vtkvmtkNonManifoldFastMarching::SolveQuadratic(double a, double b, double c, char &nSol, double &x0, double &x1)
269 {
270  double delta, q;
271 
272  delta = b*b - 4*a*c;
273 
274  if (delta < -VTK_VMTK_DOUBLE_TOL)
275  {
276  nSol = -1;
277  x0 = VTK_VMTK_LARGE_DOUBLE;
278  x1 = VTK_VMTK_LARGE_DOUBLE;
279  return;
280  }
281 
282  if (fabs(a) > VTK_VMTK_DOUBLE_TOL)
283  {
284  nSol = 2;
285  if (delta < VTK_VMTK_DOUBLE_TOL)
286  {
287  x0 = -b / 2.0*a;
288  x1 = -b / 2.0*a;
289  return;
290  }
291 
292  if (b < - VTK_VMTK_DOUBLE_TOL)
293  {
294  q = - 0.5 * (b - sqrt(delta));
295  }
296  else if (b > VTK_VMTK_DOUBLE_TOL)
297  {
298  q = - 0.5 * (b + sqrt(delta));
299  }
300  else
301  {
302  x0 = - sqrt(-c / a);
303  x1 = sqrt(-c / a);
304  return;
305  }
306 
307  x0 = q / a;
308  x1 = c / q;
309  }
310  else if (b > VTK_VMTK_DOUBLE_TOL)
311  {
312  nSol = 1;
313  x0 = -c / b;
314  x1 = VTK_VMTK_LARGE_DOUBLE;
315  }
316  else
317  {
318  nSol = 0;
319  x0 = VTK_VMTK_LARGE_DOUBLE;
320  x1 = VTK_VMTK_LARGE_DOUBLE;
321  }
322

325 double vtkvmtkNonManifoldFastMarching::ComputeUpdateFromCellNeighbor(vtkPolyData* input, vtkIdType neighborId, vtkIdType* trianglePts)
326 {
327  double fScalar, neighborT;
328  vtkIdType i;
329  bool canUpdateFromTriangle, canUpdateFromLine;
330  vtkIdType pointIdForLineUpdate, pointId;
331  vtkIdType edgesPointId[2];
332  double edgesLength[2], edgesTScalar[2];
333  double edgesVector[2][3];
334  double edgesNormal[2][3];
335  double cosTheta;
336  vtkIdType aEdgeId, bEdgeId;
337  char nSol;
338  double bEq, aEq, cEq, uEq, FEq, tEq, tCompEq, t0Eq, t1Eq, t0CompEq, tCompEqLower, tCompEqHigher;
339  double edgeLength;
340  vtkDataArray* costFunctionArray;
341 
342  pointIdForLineUpdate = -1;
343  tCompEq = 0.0;
344  if (!this->UnitSpeed)
345  {
346  costFunctionArray = input->GetPointData()->GetArray(this->CostFunctionArrayName);
347  }
348 
349  if (this->UnitSpeed)
350  {
351  fScalar = 1.0;
352  }
353  else
354  {
355  fScalar = costFunctionArray->GetTuple1(neighborId);
356  }
357 
358  neighborT = this->TScalars->GetValue(neighborId);
359 
360  canUpdateFromTriangle = true;
361  canUpdateFromLine = false;
362  for (i=1; i<3; i++)
363  {
364  pointId = trianglePts[i];
365  if (pointId!=neighborId)
366  {
367  if ((this->StatusScalars->GetValue(pointId) != VTK_VMTK_ACCEPTED_STATUS) ||
368  ((this->StatusScalars->GetValue(pointId) == VTK_VMTK_FAR_STATUS)&&(this->UpdateFromConsidered)))
369  {
370  canUpdateFromTriangle = false;
371  }
372  else
373  {
374  canUpdateFromLine = true;
375  pointIdForLineUpdate = pointId;
376  }
377  }
378  }
379 
380  if (!canUpdateFromTriangle)
381  {
382  if ((canUpdateFromLine)&&(this->AllowLineUpdate))
383  {
384  double neighborPoint[3], lineUpdatePoint[3];
385  input->GetPoint(neighborId,neighborPoint);
386  input->GetPoint(pointIdForLineUpdate,lineUpdatePoint);
387  edgeLength = sqrt(vtkMath::Distance2BetweenPoints(neighborPoint,lineUpdatePoint));
388  neighborT = this->Min(this->TScalars->GetValue(pointIdForLineUpdate) + edgeLength * fScalar,neighborT);
389  return neighborT;
390  }
391  else
392  return VTK_VMTK_LARGE_DOUBLE;
393  }
394 
395  if (trianglePts[0]==neighborId)
396  {
397  edgesPointId[0] = trianglePts[1];
398  edgesPointId[1] = trianglePts[2];
399  }
400  else if (trianglePts[1]==neighborId)
401  {
402  edgesPointId[0] = trianglePts[2];
403  edgesPointId[1] = trianglePts[0];
404  }
405  else
406  {
407  edgesPointId[0] = trianglePts[0];
408  edgesPointId[1] = trianglePts[1];
409  }
410 
411  double neighborPoint[3];
412  input->GetPoint(neighborId,neighborPoint);
413 
414  double edgePoint0[3], edgePoint1[3];
415  input->GetPoint(edgesPointId[0],edgePoint0);
416  input->GetPoint(edgesPointId[1],edgePoint1);
417 
418  edgesVector[0][0] = neighborPoint[0] - edgePoint0[0];
419  edgesVector[0][1] = neighborPoint[1] - edgePoint0[1];
420  edgesVector[0][2] = neighborPoint[2] - edgePoint0[2];
421 
422  edgesVector[1][0] = neighborPoint[0] - edgePoint1[0];
423  edgesVector[1][1] = neighborPoint[1] - edgePoint1[1];
424  edgesVector[1][2] = neighborPoint[2] - edgePoint1[2];
425 
426  edgesLength[0] = vtkMath::Norm(edgesVector[0]);
427  edgesLength[1] = vtkMath::Norm(edgesVector[1]);
428 
429  if (edgesLength[0] > VTK_VMTK_DOUBLE_TOL)
430  {
431  edgesNormal[0][0] = edgesVector[0][0] / edgesLength[0];
432  edgesNormal[0][1] = edgesVector[0][1] / edgesLength[0];
433  edgesNormal[0][2] = edgesVector[0][2] / edgesLength[0];
434  }
435  else
436  {
437  edgesLength[0] = 0.0;
438  edgesNormal[0][0] = 0.0;
439  edgesNormal[0][1] = 0.0;
440  edgesNormal[0][2] = 0.0;
441  }
442 
443  if (edgesLength[1] > VTK_VMTK_DOUBLE_TOL)
444  {
445  edgesNormal[1][0] = edgesVector[1][0] / edgesLength[1];
446  edgesNormal[1][1] = edgesVector[1][1] / edgesLength[1];
447  edgesNormal[1][2] = edgesVector[1][2] / edgesLength[1];
448  }
449  else
450  {
451  edgesLength[1] = 0.0;
452  edgesNormal[1][0] = 0.0;
453  edgesNormal[1][1] = 0.0;
454  edgesNormal[1][2] = 0.0;
455  }
456 
457  cosTheta = vtkMath::Dot(edgesNormal[0],edgesNormal[1]);
458 
459  if (fabs(cosTheta) < VTK_VMTK_DOUBLE_TOL)
460  cosTheta = 0.0f;
461 
462  edgesTScalar[0] = this->TScalars->GetValue(edgesPointId[0]);
463  edgesTScalar[1] = this->TScalars->GetValue(edgesPointId[1]);
464 
465  if (edgesTScalar[0]-edgesTScalar[1] > VTK_VMTK_DOUBLE_TOL)
466  {
467  aEdgeId = 0;
468  bEdgeId = 1;
469  }
470  else
471  {
472  aEdgeId = 1;
473  bEdgeId = 0;
474  }
475 
476 
477  FEq = fScalar + this->Regularization;
478 
479  uEq = edgesTScalar[aEdgeId]-edgesTScalar[bEdgeId];
480  aEq = edgesLength[aEdgeId]*edgesLength[aEdgeId] + edgesLength[bEdgeId]*edgesLength[bEdgeId] - 2*edgesLength[aEdgeId]*edgesLength[bEdgeId]*cosTheta;
481  bEq = 2 * edgesLength[bEdgeId] * uEq * (edgesLength[aEdgeId] * cosTheta - edgesLength[bEdgeId]);
482  cEq = edgesLength[bEdgeId]*edgesLength[bEdgeId] * (uEq*uEq - FEq*FEq*edgesLength[aEdgeId]*edgesLength[aEdgeId]*(1-cosTheta*cosTheta));
483 
484  this->SolveQuadratic(aEq,bEq,cEq,nSol,t0Eq,t1Eq);
485 
486  if (nSol==2)
487  {
488  tEq = this->Min(t0Eq,t1Eq);
489  }
490  else
491  {
492  tEq = t0Eq;
493  }
494 
495  if (fabs(tEq) > VTK_VMTK_DOUBLE_TOL)
496  {
497  tCompEq = edgesLength[bEdgeId] * (tEq - uEq) / tEq;
498  }
499  else
500  {
501  t0CompEq = VTK_VMTK_LARGE_DOUBLE;
502  }
503 
504  tCompEqLower = edgesLength[aEdgeId]*cosTheta;
505 
506  if (fabs(cosTheta) > VTK_VMTK_DOUBLE_TOL)
507  {
508  tCompEqHigher = edgesLength[aEdgeId]/cosTheta;
509  }
510  else
511  {
512  tCompEqHigher = VTK_VMTK_LARGE_DOUBLE;
513  }
514 
515  if ((uEq - tEq < -VTK_VMTK_DOUBLE_TOL) && (tCompEq - tCompEqLower > VTK_VMTK_DOUBLE_TOL) && (tCompEq - tCompEqHigher < -VTK_VMTK_DOUBLE_TOL))
516  {
517  neighborT = this->Min(tEq + edgesTScalar[bEdgeId],neighborT);
518  }
519  else
520  {
521  neighborT = this->Min(edgesLength[aEdgeId]*FEq + edgesTScalar[aEdgeId] , edgesLength[bEdgeId]*FEq + edgesTScalar[bEdgeId]);
522  }
523 
524  return neighborT;
525 }

528 {
529  vtkIdType i, j, k;
530  vtkIdType npts, *pts;
531  vtkIdType trianglePts[3];
532  double tMin, tScalar;
533  vtkIdList* neighborCellNeighborIds;
534 
535  if ((neighborId<0)||(neighborId>=this->TScalars->GetNumberOfTuples()))
536  {
537  vtkErrorMacro("Requested id exceeds TScalars dimension.");
538  return;
539  }
540 
541  neighborCellNeighborIds = vtkIdList::New();
542 
543  input->GetPointCells(neighborId,neighborCellNeighborIds);
544 
545  tMin = this->TScalars->GetValue(neighborId);
546  trianglePts[0] = neighborId;
547  for (i=0; i<neighborCellNeighborIds->GetNumberOfIds(); i++)
548  {
549  // virtual triangulation
550  input->GetCellPoints(neighborCellNeighborIds->GetId(i),npts,pts);
551  for (j=0; j<npts; j++)
552  {
553  if (pts[j]!=neighborId)
554  {
555  trianglePts[1] = pts[j];
556  for (k=j+1; k<npts; k++)
557  {
558  if (pts[k]!=neighborId)
559  {
560  trianglePts[2] = pts[k];
561  tScalar = this->ComputeUpdateFromCellNeighbor(input,neighborId,trianglePts);
562  tMin = this->Min(tScalar,tMin);
563  }
564  }
565  }
566  }
567  }
568 
569  this->TScalars->SetValue(neighborId,tMin);
570 
571  neighborCellNeighborIds->Delete();
572 }

574 void vtkvmtkNonManifoldFastMarching::UpdateNeighborhood(vtkPolyData* input, vtkIdType pointId)
575 {

582  for (i=0; i<neighborIds->GetNumberOfIds(); i++)
583  {

597  }

603 void vtkvmtkNonManifoldFastMarching::Propagate(vtkPolyData* input)
604 {
605  double currentTravelTime;
606  vtkIdType trialId;
607 
608  while (this->ConsideredMinHeap->GetSize()>0)
609  {
610  trialId = this->ConsideredMinHeap->RemoveMin();
611  this->StatusScalars->SetValue(trialId,VTK_VMTK_ACCEPTED_STATUS);
612  this->NumberOfAcceptedPoints++;
613 
614  this->UpdateNeighborhood(input,trialId);
615 
616  currentTravelTime = this->TScalars->GetValue(trialId);
617 
618  if ((this->StopNumberOfPoints)||(this->StopTravelTime))
619  {
620  if ((this->NumberOfAcceptedPoints >= this->StopNumberOfPoints)||(currentTravelTime - this->StopTravelTime > VTK_VMTK_DOUBLE_TOL))
621  {
622  break;
623  }
624  }
625  }
