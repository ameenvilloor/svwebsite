217 int tetIntPnt(int nint, DARR4 **bcord, double **wt)
218 {
219  int retval = 1;
220 
221  if( nint == 1 ) {*bcord = rstw1 ; *wt = twt1; }
222  else if( nint == 4 ){*bcord = rstw4 ; *wt = twt4; }
223  else if( nint == 5 ){*bcord = rstw5 ; *wt = twt5; }
224  else if( nint == 16 ){*bcord = rstw16 ; *wt = twt16; }
225  else if( nint == 17 ){*bcord = rstw17 ; *wt = twt17; }
226  else if( nint == 29 ){*bcord = rstw29 ; *wt = twt29; }
227  else
228  {
229  fprintf(stderr,"\n%d integration points unsupported in symtet.c; give {1,4,5,16,17,29}\n",nint);
230  retval = 0;
231  }
232  return retval ;
233 }
