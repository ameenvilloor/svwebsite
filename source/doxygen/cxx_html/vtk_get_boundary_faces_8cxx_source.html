136  vtkSmartPointer<vtkCellArray> boundaryLines = vtkSmartPointer<vtkCellArray>::New();

138  vtkSmartPointer<vtkIntArray> ptMarkers = vtkSmartPointer<vtkIntArray>::New();
139 
140  //Get input points, polys and set the up in the vtkPolyData mesh
141  inpts = input->GetPoints();
142  inPolys = input->GetPolys();
143  this->mesh->SetPoints(inpts);
144  this->mesh->SetPolys(inPolys);
145 
146  //Get the number of Polys for scalar allocation
147  numPolys = input->GetNumberOfPolys();
148  numPts = input->GetNumberOfPoints();
149  this->allIds->Allocate(numPolys);

151  //Create vector of Scalars to Mark Points to pass through filter
152  ptMarkers->SetNumberOfValues(numPts);
153  for (pt=0;pt<numPts;pt++)

156  }
157

160  input->GetPointData()->AddArray(ptMarkers);
161  input->GetPointData()->SetActiveScalars("PointMarkers");
162

164  if (numPolys < 1)
165  {
166  vtkDebugMacro("No input!");
167  return 1;
168  }
169

171  this->newScalars->SetNumberOfTuples(numPolys);
172

191  this->boundaryMesh->SetPoints(boundaryPts);
192  this->boundaryMesh->SetLines(boundaryLines);
193  this->boundaryMesh->GetPointData()->PassData(this->boundaries->GetOutput()->GetPointData());
194 
195  vtkDebugMacro("Starting Boundary Face Separation");
196 
197  //Build Links in the mesh to be able to perform complex polydata processes;
198  this->mesh->BuildLinks();
199 
200  //Set Region value of each cell to be zero initially
201  for(cellId = 0; cellId < numPolys ; cellId ++)
202  {
203  this->newScalars->InsertValue(cellId, reg);
204  }
205

235 void vtkGetBoundaryFaces::FindBoundaryRegion(const vtkIdType cellId, int reg)
236 {
237  //Variables used in function
238  int i;
239  vtkIdType j,k,l;
240  static vtkIdType Id1=0;
241  static vtkIdType Id2=0;
242  vtkIdType *pts = 0;
243  vtkIdType npts = 0;
244  vtkIdType numNei, nei, p1, p2, nIds, neis;
245 
246  //Id List to store neighbor cells, added to member data allIds and delted at end of function
247  vtkIdList *neiIds;
248  neiIds = vtkIdList::New();
249  neiIds->Allocate(VTK_CELL_SIZE);
250  vtkIdType numBoundPts,numBoundLines;
251 
252  //Id List to store the one neighbor cell for each set of nodes and a cell, deleted at end of function
253  vtkIdList *neighbors;
254  neighbors = vtkIdList::New();
255  neighbors->Allocate(VTK_CELL_SIZE);
256 
257  //Variable for accessing neiIds list
258  vtkIdType sz = 0;
259 
260  //Variables for the boundary cells adjacent to the boundary point
261  vtkSmartPointer<vtkIdList> bLinesOne = vtkSmartPointer<vtkIdList>::New();
262  vtkSmartPointer<vtkIdList> bLinesTwo = vtkSmartPointer<vtkIdList>::New();
263  vtkSmartPointer<vtkIntArray> meshPtMarkers = vtkSmartPointer<vtkIntArray>::New();
264 
265  meshPtMarkers = vtkIntArray::SafeDownCast(this->boundaryMesh->GetPointData()->GetScalars("PointMarkers"));
266 
267  if (meshPtMarkers->GetNumberOfTuples() < 1)
268  {
269  std::cout<<"Data wasn't transferred correctly"<<std::endl;
270  }
271 
272  //Get output set of points and lines from the Feature Edges
273  numBoundPts = this->boundaryMesh->GetPoints()->GetNumberOfPoints();
274  numBoundLines = this->boundaryMesh->GetLines()->GetNumberOfCells();
275 
276  //Mark point with region number and get all points in the cell
277  this->newScalars->InsertValue(cellId,reg);
278  this->mesh->GetCellPoints(cellId,npts,pts);
279 
280  //Set up point arrays for checking of boundary points
281  double* boundPtComps = new double[npts];
282  double* meshPt1Comps = new double[npts];
283  double* meshPt2Comps = new double[npts];
284 
285  //Get neighboring cell for each pair of points in current cell
286  for (i=0; i < npts; i++)
287  {
288  p1 = pts[i];
289  p2 = pts[(i+1)%(npts)];
290 
291 
292  //Initial check to make sure the cell is in fact a face cell
293  this->mesh->GetCellEdgeNeighbors(cellId,p1,p2, neighbors);
294  numNei = neighbors->GetNumberOfIds();
295 
296  //Check to make sure it is an oustide surface cell, i.e. one neighbor
297  if (numNei==1)
298  {
299  int count = 0;
300  for (j=0; j<numBoundPts;j++)
301  {
302 
303  //Check to see if cell is on the boundary, if it is get adjacent lines
304  if (meshPtMarkers->GetValue(j) == p1)
305  {
306  this->boundaryMesh->GetPointCells(j,bLinesOne);
307  count++;
308  }
309  if (meshPtMarkers->GetValue(j) == p2)
310  {
311  this->boundaryMesh->GetPointCells(j,bLinesTwo);
312  count++;
313  }
314 
315  }
316 
317  nei=neighbors->GetId(0);
318  //if cell is not on the boundary, add new cell to check list
319  if (count < 2)
320  {
321  if (this->newScalars->GetValue(nei)==0)
322  {
323  neiIds->InsertId(sz,nei);
324  sz++;
325  }
326  }
327  //if cell is on boundary, check to make sure it isn't false positive; don't add to check list
328  else
329  {
330  bLinesOne->IntersectWith(bLinesTwo);
331  if (bLinesOne->GetNumberOfIds() == 0)
332  {
333  neiIds->InsertId(sz,nei);
334  sz++;
335  }
336  }
337  }
338  }
339 
340  nIds = neiIds->GetNumberOfIds();
341  neighbors->Delete();
342  //If there are neighboring cells in the check list, run function recursively
343  if (nIds>0)
344  {
345  //Add all Ids in current list to global list of Ids
346  for (k=0; k< nIds;k++)
347  {
348  neis = neiIds->GetId(k);
349  this->allIds->InsertId(Id1,neis);
350  Id1++;
351  }
352  neiIds->Delete();
353  //Run function recursively for each Id just placed in the global list
354  for (l=0; l< nIds;l++)
355  {
356  neis = this->allIds->GetId(Id2);
357  Id2++;
358  if (this->newScalars->GetValue(neis)==0)
359  {
360  this->FindBoundaryRegion(neis,reg);
361  }
362  }
363  }
364 
365  else
366  {
367  neiIds->Delete();
368  }
369 
370  delete [] boundPtComps;
371  delete [] meshPt1Comps;
372  delete [] meshPt2Comps;
373 
374 }
