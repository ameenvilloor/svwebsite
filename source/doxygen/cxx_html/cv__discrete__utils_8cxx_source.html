149 int discreteCalcJacDet(double xx[2][4], double r, double s, double *determinant) {
150 
151  /**************************************************************
152  * The code below was adapted from Finite Element Procedures *
153  * by K. Bathe, 1996, pages 482-3. *
154  **************************************************************/
155 
156  *determinant = 0.0;
157 
158  int i,j,k;
159  double rp,sp,rm,sm;
160  double h[4],p[2][4],xj[2][2];
161  double dum,det;
162 
163  rp = 1.0 + r;
164  sp = 1.0 + s;
165  rm = 1.0 - r;
166  sm = 1.0 - s;
167 
168  // interpolation functions
169 
170  h[0] = 0.25 * rp * sp;
171  h[1] = 0.25 * rm * sp;
172  h[2] = 0.25 * rm * sm;
173  h[3] = 0.25 * rp * sm;
174 
175  //
176  // natural coordinate derviatives of the interpolation functions
177  //
178 
179  // with respect to r
180 
181  p[0][0] = 0.25 * sp;
182  p[0][1] = -p[0][0];
183  p[0][2] = -0.25 * sm;
184  p[0][3] = -p[0][2];
185 
186  // with respect to s
187 
188  p[1][0] = 0.25 * rp;
189  p[1][1] = 0.25 * rm;
190  p[1][2] = -p[1][1];
191  p[1][3] = -p[1][0];
192 
193  // evaluate the jacobian at point (r,s)
194 
195  for (i = 0; i <= 1; i++) {
196  for (j = 0; j <= 1; j++) {
197  dum = 0;
198  for (k = 0; k <= 3; k++) {
199  dum = dum + p[i][k]*xx[j][k];
200  }
201  xj[i][j]=dum;
202  }
203  }
204 
205  // compute the determinant of the jacobian at point (r,s)
206  det = xj[0][0]*xj[1][1] - xj[1][0]*xj[0][1];
207 
208  *determinant = det;
209 
210  if (det < 0.0) {
211  //fprintf(stderr,"ERROR: Jacobian determinant negative! (%lf)\n"
212  // ,det);
213  return CV_ERROR;
214  }
215 
216  return CV_OK;
217 
218 }
