123

125  vtkSmartPointer<vtkvmtkPolyDataSurfaceRemeshing> remesher =
126  vtkSmartPointer<vtkvmtkPolyDataSurfaceRemeshing>::New();
127 
128  //Check the inputs to the function and apply default values if necessary
129  if (trianglesplitfactor == NULL)
130  {
131  trianglesplitfactor = 1.0;
132  }
133  if (collapseanglethreshold == NULL)
134  {
135  collapseanglethreshold = 0.5;
136  }
137  if (meshcapsonly && excludedIds == NULL)
138  {
139  //Need list of ids to mesh only caps

143 
144  if (useSizingFunction && meshSizingFunction == NULL)
145  {

148  }
149 
150  //Apply the Entity Ids array name if not preserving edges and meshing all
151  if (!preserveedges && !meshcapsonly)
152  {
153  vtkSmartPointer<vtkIntArray> cellIds = vtkSmartPointer<vtkIntArray>::New();
154  cellIds->SetName(cellEntityIdsArrayName.c_str());
155  cellIds->SetNumberOfTuples(surfacepd->GetNumberOfCells());
156  cellIds->FillComponent(0,0.0);
157  surfacepd->GetCellData()->AddArray(cellIds);
158  }
159  //If applying a meshsizing function, transfer the values to an array of
160  //areas based on the edge sizes specified
161  if (useSizingFunction)
162  {
163  meshSizingFunction->SetName("SizingFunction");
164  for (i=0;i<meshSizingFunction->GetNumberOfTuples();i++)
165  {
166  edgesize = meshSizingFunction->GetValue(i);
167  if (edgesize == 0)
168  {
169  meshSizingFunction->SetValue(i,(0.25)*pow(3.0,0.5)*pow(maxEdgeSize,2.0));
170  }
171  else
172  {
173  meshSizingFunction->SetValue(i,(0.25)*pow(3.0,0.5)*pow(edgesize,2.0));
174  }
175  }
176  surfacepd->GetPointData()->AddArray(meshSizingFunction);
177  }
178 
179  cleaner->SetInputData(surfacepd);
180  cleaner->Update();
181 
182  tris1->SetInputData(cleaner->GetOutput());
183  tris1->Update();
184 
185  //Set up input based on full or cap remesh
186  if (meshcapsonly)
187  {
188  remesher->SetInputData(surfacepd);
189  }
190  else
191  {
192  remesher->SetInputData(tris1->GetOutput());
193  }
194 
195  //Set the remesher options based on the input
196  remesher->SetPreserveBoundaryEdges(preserveedges);
197  remesher->SetCellEntityIdsArrayName(cellEntityIdsArrayName.c_str());

202  }
203  else
204  {
205  remesher->SetElementSizeModeToTargetArea();

207  }
208  remesher->SetMaxArea(1e16);
209  remesher->SetMinArea(0.0);
210  remesher->SetTriangleSplitFactor(trianglesplitfactor);
211  remesher->SetCollapseAngleThreshold(collapseanglethreshold);
212  if (meshcapsonly)
213  {
214  remesher->SetExcludedEntityIds(excludedIds);
215  }
216  //Remesh the surface
217  remesher->Update();
218 
219  if (meshcapsonly)
220  {
221  surfaceMesh->DeepCopy(remesher->GetOutput());
222  }
223  //If meshing the whole thing, project it back to the original surface,
224  //compute the normals, and clean the mesh
225  else
226  {
227  projector->SetInputData(remesher->GetOutput());
228  projector->SetReferenceSurface(surfacepd);
229  projector->Update();
230 
231  normaler->SetInputData(projector->GetOutput());
232  normaler->SetConsistency(1);
233  normaler->SetAutoOrientNormals(1);
234  normaler->SetFlipNormals(0);
235  normaler->SetComputeCellNormals(0);
236  normaler->SplittingOff();
237  normaler->Update();
238 
239  cleaner2->SetInputData(normaler->GetOutput());
240  cleaner2->Update();
241 
242  surfaceMesh->DeepCopy(cleaner2->GetOutput());
243  surfaceMesh->GetPointData()->GetNormals()->SetName("Normals");
244  }
245 
246  return CV_OK;
247 }

539 int VMTKUtils_InsertIds(vtkUnstructuredGrid *fullmesh, vtkPolyData *fullpolydata)
540 {
541  int k;
542  int modelId = 1;
543  int globalId = 1;
544  int numNewCells,numStartCells;
545  double pt1[3];
546  double pt2[3];
547  double pt3[3];
548  double pt4[3];
549  double a[3];
550  double b[3];
551  double c[3];
552  double norm[3];
553  double mydot;
554  double tmp;
555 
556  vtkIdType i,j;
557  vtkIdType vtkId;
558  vtkIdType count=0;
559  vtkIdType npts = 0;
560  vtkIdType *pts = 0;
561  vtkIdType numPts,numCells;
562  vtkSmartPointer<vtkIntArray> modelRegionIds =
563  vtkSmartPointer<vtkIntArray>::New();
564  vtkSmartPointer<vtkIntArray> globalNodeIds =
565  vtkSmartPointer<vtkIntArray>::New();
566  vtkSmartPointer<vtkIntArray> globalElementIds =
567  vtkSmartPointer<vtkIntArray>::New();
568  vtkSmartPointer<vtkIntArray> isTriangle =
569  vtkSmartPointer<vtkIntArray>::New();
570  vtkSmartPointer<vtkUnstructuredGrid> fullcopy =
571  vtkSmartPointer<vtkUnstructuredGrid>::New();
572  vtkSmartPointer<vtkDataSetSurfaceFilter> getSurface =
573  vtkSmartPointer<vtkDataSetSurfaceFilter>::New();
574  vtkSmartPointer<vtkThreshold> thresholder =
575  vtkSmartPointer<vtkThreshold>::New();
576 
577  fullcopy->DeepCopy(fullmesh);
578  numPts = fullcopy->GetNumberOfPoints();
579  numCells = fullcopy->GetNumberOfCells();
580 
581 
582  //Remove all triangle Cells! These are triangles on surface of mesh
583  //left from the append filter
584  isTriangle->SetNumberOfComponents(1);
585  isTriangle->Allocate(numCells,1000);
586  isTriangle->SetNumberOfTuples(numCells);
587  isTriangle->SetName("isTriangle");
588  for (i=0;i<numCells;i++)
589  {
590  if (fullcopy->GetCellType(i) == VTK_TRIANGLE)
591  {
592  isTriangle->InsertTuple1(i,1);
593  }
594  else
595  {
596  isTriangle->InsertTuple1(i,0);
597  }
598  }
599  fullcopy->GetCellData()->AddArray(isTriangle);
600  fullcopy->GetCellData()->SetActiveScalars("isTriangle");
601 
602  thresholder->SetInputData(fullcopy);
603  thresholder->SetInputArrayToProcess(0,0,0,1,"isTriangle");
604  thresholder->ThresholdBetween(0,0);
605  thresholder->Update();
606 
607  fullcopy->DeepCopy(thresholder->GetOutput());
608  fullcopy->GetCellData()->RemoveArray("isTriangle");
609  numPts = fullcopy->GetNumberOfPoints();
610  numCells = fullcopy->GetNumberOfCells();
611 
612  //Add global node Ids to points
613  for (i=0;i<numPts;i++)
614  {
615  globalNodeIds->InsertValue(i,globalId);
616  globalId++;
617  }
618 
619  //Add global element Ids to cells
620  globalId = 1;
621  for (i=0;i<numCells;i++)
622  {
623  globalElementIds->InsertValue(i,globalId);
624  modelRegionIds->InsertValue(i,modelId);
625  globalId++;
626  }
627 
628  //Make Global Ids active scalars
629  modelRegionIds->SetName("ModelRegionID");
630  fullcopy->GetCellData()->AddArray(modelRegionIds);
631  fullcopy->GetCellData()->SetActiveScalars("ModelRegionID");
632 
633  globalNodeIds->SetName("GlobalNodeID");
634  fullcopy->GetPointData()->AddArray(globalNodeIds);
635  fullcopy->GetPointData()->SetActiveScalars("GlobalNodeID");
636 
637  globalElementIds->SetName("GlobalElementID");
638  fullcopy->GetCellData()->AddArray(globalElementIds);
639  fullcopy->GetCellData()->SetActiveScalars("GlobalElementID");
640 
641  //This is the full mesh with scalars
642  fullmesh->DeepCopy(fullcopy);
643 
644  //Flip the cells within the volume if nodes are numbered wrong
645  for (i=0;i<fullmesh->GetNumberOfCells();i++)
646  {
647  fullmesh->GetCellPoints(i,npts,pts);
648  fullmesh->GetPoint(pts[0],pt1);
649  fullmesh->GetPoint(pts[1],pt2);
650  fullmesh->GetPoint(pts[2],pt3);
651  fullmesh->GetPoint(pts[3],pt4);
652 
653  for (k=0;k<3;k++)
654  {
655  a[k] = pt2[k]-pt1[k];
656  b[k] = pt3[k]-pt1[k];
657  c[k] = pt4[k]-pt1[k];
658  }
659 
660  vtkMath::Cross(a,b,norm);
661  mydot = vtkMath::Dot(norm,c);
662 
663  if (mydot < 0)
664  {
665  tmp = pts[1];
666  pts[1] = pts[2];
667  pts[2] = tmp;
668  fullmesh->ReplaceCell(i,npts,pts);
669  }
670  }
671 
672  getSurface->SetInputData(fullcopy);
673  getSurface->Update();
674 
675  fullpolydata->DeepCopy(getSurface->GetOutput());
676 
677  return CV_OK;
678 }
