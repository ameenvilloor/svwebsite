147  int *forceFlag, double toDot[] )

151  double K;
152  double n[3];
153  double tmpF0, tmpF1;
154  double mag;
155  double dynamicRatio;
156 
157  if ( !Valid() ) return CV_ERROR;

159  if ( ls_->GetGrid()->InterpN( pos, n ) != CV_OK ) return CV_ERROR;
160 
161  if ( GetImageClosed( image_ ) && InBorder( image_, pos, 1 ) ) {
162  toDot[0] = 0.0;
163  toDot[1] = 0.0;
164  toDot[2] = 0.0;
165  *f0 = 0.0;
166  *f1 = 0.0;
167  if ( v != NULL ) {
168  v[0] = 0.0;
169  v[1] = 0.0;
170  v[2] = 0.0;
171  }
172  return CV_OK;
173  }
174 
175  // Look up value of potential:
176  if ( ! GetIntensity( image_, pos, &p ) ) {
177  return CV_ERROR;
178  }
179 
180  // Look up potential gradient:
181  if ( ! GetGradIx( image_, pos, &(gradP[0]) ) ) {
182  return CV_ERROR;
183  }
184  if ( ! GetGradIy( image_, pos, &(gradP[1]) ) ) {
185  return CV_ERROR;
186  }
187  if ( image_->dim == 3 ) {
188  if ( ! GetGradIz( image_, pos, &(gradP[2]) ) ) {
189  return CV_ERROR;
190  }
191  } else {
192  gradP[2] = 0.0;
193  }
194 
195 
196  tmpF0 = 0.0;
197  tmpF1 = 0.0;
198  if ( K > Kupp_ ) {
199  tmpF1 = - eK_ * fabs(K-Kupp_);
200  } else if ( K < Klow_ ) {
201  tmpF1 = eK_ * fabs(Klow_-K);
202  } else {
203  tmpF0 = eP_ * ( gradP[0] * n[0] + gradP[1] * n[1] + gradP[2] * n[2] );
204  }
205 
206 
207  /* 2/21/00:
208  * ---
209  tmpF0 = 0.0;
210  tmpF1 = 0.0;
211  if ( p >= Pc_ ) {
212  tmpF0 = eP_ * ( gradP[0] * n[0] + gradP[1] * n[1] + gradP[2] * n[2] );
213  } else if ( K > Kupp_ ) {
214  tmpF1 = - eK_ * fabs(K);
215  } else if ( K < Klow_ ) {
216  tmpF1 = eK_ * fabs(K);
217  } else {
218  tmpF0 = balloonF_;
219  }
220  */
221 
222 
223  /* This is an unintelligible experiment from ~2/9/00:
224  * ---
225  if ( autoBalloon_ ) {
226  if ( fabs(tmpF0) < 1.0 ) {
227  tmpF0 = balloonF_;
228  } else {
229  dynamicRatio = fabs( tmpF0 / balloonF_ );
230  tmpF0 /= dynamicRatio;
231  }
232  } else {
233  tmpF0 += balloonF_;
234  }
235  */
236 
237  (*forceFlag) = 0;
238  toDot[0] = 0.0;
239  toDot[1] = 0.0;
240  toDot[2] = 0.0;
241 
242  *f0 = tmpF0;
243  *f1 = tmpF1;
244  mag = (*f0) + (*f1);
245 
246  if ( v != NULL ) {
247  v[0] = n[0] * mag;
248  v[1] = n[1] * mag;
249  v[2] = n[2] * mag;
250  }
251 
252  return CV_OK;
253 }
