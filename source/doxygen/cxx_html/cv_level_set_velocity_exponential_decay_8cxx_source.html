75 // SetEIneg
76 // --------
77 // NOTE that eIneg_ is a new parameter (as of 4/11/00) which is being
78 // added to enable cvLevelSetVelocityExponentialDecay to apply a different decay rate based on
79 // the sign of the intensity gradient. Previously, we always applied
80 // the same decay regardless of the sign of the gradient. Now, in
81 // order to better handle CT data and the range of tissues which may
82 // be resolved, we may want to treat positive gradients differently
83 // from negative ones. Note also, however, that eIneg_ is an OPTIONAL
84 // parameter, and that if it is not specified, behavior defaults to
85 // the case where gradient direction is not considered in computing
86 // velocity.
87

89 {
90  eIneg_ = d;

92  return CV_OK;

240 int cvLevelSetVelocityExponentialDecay::Evaluate( double pos[], double *f0, double *f1, double v[],
241  int *forceFlag, double toDot[] )
242 {
243  double gradI[3];
244  double K;
245  double km, kg;
246  double ks[2];
247  double n[3];
248  double decay;
249  double tmpF0, tmpF1;
250  double dir;
251  double mag;
252  double grad_dir;
253 
254  if ( !Valid() ) return CV_ERROR;
255 
256  if ( ls_->GetGrid()->InterpN( pos, n ) != CV_OK ) return CV_ERROR;
257 
258  if ( ls_->GetGrid()->GetDim() == 3 ) {
259  switch (ktype_) {
260  case VED_Default_K:
261  if ( ls_->GetGrid()->InterpK( pos, &K ) != CV_OK ) return CV_ERROR;
262  case VED_Mean_K:
263  if ( ls_->GetGrid()->InterpKm( pos, &K ) != CV_OK ) return CV_ERROR;
264  break;
265  case VED_Gaussian_K:
266  if ( ls_->GetGrid()->InterpKg( pos, &K ) != CV_OK ) return CV_ERROR;
267  break;
268  case VED_PrincipleK1_K:
269  if ( ls_->GetGrid()->InterpKg( pos, &kg ) != CV_OK ) return CV_ERROR;
270  if ( ls_->GetGrid()->InterpKm( pos, &km ) != CV_OK ) return CV_ERROR;
271  Compute3dks( kg, km, tol_, ks );
272  K = ks[0];
273  break;
274  case VED_PrincipleK2_K:
275  if ( ls_->GetGrid()->InterpKg( pos, &kg ) != CV_OK ) return CV_ERROR;
276  if ( ls_->GetGrid()->InterpKm( pos, &km ) != CV_OK ) return CV_ERROR;
277  Compute3dks( kg, km, tol_, ks );
278  K = ks[1];
279  break;
280  default:
281  assert(0);
282  }
283  } else {
284  if ( ls_->GetGrid()->InterpK( pos, &K ) != CV_OK ) return CV_ERROR;
285  }
286 
287  // This closed-image business is anachronistic now that
288  // cvLevelSetStructuredGrid::GetFront supports a "closed-front" option. But
289  // I'll keep this here for backward compatibility.
290  if ( GetImageClosed( image_ ) && InBorder( image_, pos, 1 ) ) {
291  toDot[0] = 0.0;
292  toDot[1] = 0.0;
293  toDot[2] = 0.0;
294  *f0 = 0.0;
295  *f1 = 0.0;
296  if ( v != NULL ) {
297  v[0] = 0.0;
298  v[1] = 0.0;
299  v[2] = 0.0;
300  }
301  return CV_OK;
302  }
303 
304  // Look up intensity gradient:
305  if ( ! GetGradIx( image_, pos, &(gradI[0]) ) ) {
306  return CV_ERROR;
307  }
308  if ( ! GetGradIy( image_, pos, &(gradI[1]) ) ) {
309  return CV_ERROR;
310  }
311  if ( image_->dim == 3 ) {
312  if ( ! GetGradIz( image_, pos, &(gradI[2]) ) ) {
313  return CV_ERROR;
314  }
315  } else {
316  gradI[2] = 0.0;
317  }
318 
319  // Recall:
320  // - tmpF0 is for K-independent terms
321  // - tmpF1 is for K-dependent terms
322 
323  // Set the inflation direction:
324  if ( expand_ ) {
325  dir = 1.0;
326  } else {
327  dir = -1.0;
328  }
329 
330  // ------------------------------------------------------------------
331  // If eIneg_ was specified, then apply a different decay rate
332  // depending on whether this point on the front is on an uphill
333  // intensity gradient or downhill one.
334  // ------------------------------------------------------------------
335 
336  if ( eInegValid_ ) {
337  grad_dir = Dot( n[0], n[1], n[2], gradI[0], gradI[1], gradI[2] );
338  if ( grad_dir >= 0 ) {
339  // decay = exp( - eI_ * Magnitude( gradI[0], gradI[1], gradI[2] ) );
340  decay = exp( - eI_ * fabs(grad_dir) );
341  } else {
342  // decay = exp( - eIneg_ * Magnitude( gradI[0], gradI[1], gradI[2] ) );
343  decay = exp( - eIneg_ * fabs(grad_dir) );
344  }
345  }
346 
347  // ------------------------------------------------------------------
348  // Otherwise, if eIneg_ was not specified, then just use eI_ as a
349  // uniform decay rate regardless of the direction of the gradient.
350  // These semantics will provide backward compatibility with scripts
351  // that don't know about eIneg_.
352  // ------------------------------------------------------------------
353 
354  else {
355  decay = exp( - eI_ * Magnitude( gradI[0], gradI[1], gradI[2] ) );
356  }
357 
358  // ------------------------------------------------------------------
359  // Now, if the monotonic_ flag is on, eliminate curvature
360  // dependence. Otherwise, apply the Kt_ constraint.
361  // ------------------------------------------------------------------
362 
363  if ( ! monotonic_ ) {
364  tmpF0 = dir * Kt_ * decay;
365  tmpF1 = - K * decay;
366  } else {
367  tmpF0 = dir * decay;
368  tmpF1 = 0.0;
369  }
370 
371  // ------------------------------------------------------------------
372  // If clamping has been specified, then stop the front locally in
373  // places where velocity has fallen below stopV_.
374  // ------------------------------------------------------------------
375 
376  if ( clamp_ ) {
377  if ( fabs( tmpF0 + tmpF1 ) < stopV_ ) {
378  tmpF0 = 0.0;
379  tmpF1 = 0.0;
380  }
381  }
382 
383  (*forceFlag) = 0;
384  toDot[0] = 0.0;
385  toDot[1] = 0.0;
386  toDot[2] = 0.0;
387 
388  *f0 = tmpF0;
389  *f1 = tmpF1;
390  mag = (*f0) + (*f1);
391 
392  if ( v != NULL ) {
393  v[0] = n[0] * mag;
394  v[1] = n[1] * mag;
395  v[2] = n[2] * mag;
396  }
397 
398  return CV_OK;
399 }
