54 // FindMachineEpsilon
55 // ------------------
56 // By epsilon, we're referring to the smallest quantity e which causes
57 // 1.0 + e to still be > 1.0. (Note that tests against 0.0 is
58 // potentially misleading since there may be special hardware for 0.0
59 // arithmetic.) More precisely, perhaps, the returned value e will be
60 // the smallest value s.t. values e' an order of magnitude or more
61 // smaller than e will not give 1.0 + e' > 1.0. Also note that
62 // epsilon is completely different from the smallest double values
63 // which the machine can represent... instead, epsilon is the smallest
64 // *interval* which the machine can distinguish.
65 
66 double FindMachineEpsilon()
67 {
68  double num = 1.0;
69  double test = 1.0;
70 
71  while ( num + test > num ) {
72  test /= 10.0;
73  }
74  return (test * 10.0);
75 }

78 // -----------
79 // Compute3dks
80 // -----------
81 // This function computes the principle curvatures, k1 and k2, given
82 // quantities for Gaussian curvature Kg and mean curvature Km. Note
83 // that while we compute Kg and Km from the level set function, they
84 // are defined as:
85 //
86 // Kg = k1 * k2 Km = (k1 + k2) / 2
87 //
88 // As such, we can solve for the two unknowns k1 and k2. Note that
89 // while Km given this definition is ambiguous w.r.t. sign (i.e. a
90 // direction for the principle curvatures k1 and k2 determines their
91 // sign), Km as derived from the level set equation embeds information
92 // about inside vs. outside. As a result, we can use additional
93 // information in the sign of Km in selecting the appropriate root of
94 // the quadratic.
95 //
96 // Note that the parameter ks must be able to hold 2 returned values:
97 // ks[0] <--> k1
98 // ks[1] <--> k2
99 
100 int Compute3dks( double Kg, double Km, double tol, double ks[] )
101 {
102  double kia, kja;
103  double kib, kjb;
104  double k1a, k2a;
105  double k1b, k2b;
106  double tmp;
107 
108  tmp = (4 * Km * Km) - (4 * Kg);
109  if ( tmp < 0.0 ) {
110  return CV_ERROR;
111  }
112  tmp = sqrt( tmp );
113 
114  kia = (2 * Km + tmp) / 2;
115  kja = Kg / kia;
116  k1a = minimum( kia, kja );
117  k2a = maximum( kia, kja );
118 
119  kib = (2 * Km - tmp) / 2;
120  kjb = Kg / kib;
121  k1b = minimum( kib, kjb );
122  k2b = maximum( kib, kjb );
123 
124  if ( ( (Km > 0.0) && (k2a > 0.0) ) ||
125  ( (Km < 0.0) && (k1a < 0.0) ) ||
126  ( (Sign( Km, tol ) == 0.0) && (k1a <= 0.0 ) ) ) {
127  ks[0] = k1a;
128  ks[1] = k2a;
129  return CV_OK;
130  }
131 
132  if ( ( (Km > 0.0) && (k2b > 0.0) ) ||
133  ( (Km < 0.0) && (k1b < 0.0) ) ||
134  ( (Sign( Km, tol ) == 0.0) && (k1b <= 0.0 ) ) ) {
135  ks[0] = k1b;
136  ks[1] = k2b;
137  return CV_OK;
138  }
139 
140  return CV_ERROR;
141 }

144 // ==============
145 // CountLines
146 // ==============
147 // For files which have a newline as the last character, the value
148 // returned is equal to the number of newline characters found in the
149 // file. However, if the file does not end with a newline, CountLines
150 // returns the number of newlines plus one, so that any distinct line
151 // is counted.
152 
153 int CountLines( char *filename )
154 {
155  FILE *fp;
156  int c, prevC;
157  int count = 0;
158 
159  prevC = EOF;
160  fp = fopen( filename, "r" );
161  if (fp == NULL) return -1;
162  while ( (c = fgetc(fp)) != EOF ) {
163  if (c == '\n') {
164  count++;
165  }
166  prevC = c;
167  }
168  if (prevC != '\n') {
169  count++;
170  }
171  fclose(fp);
172  return count;
173 }
