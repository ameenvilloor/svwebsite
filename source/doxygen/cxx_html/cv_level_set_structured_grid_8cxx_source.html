47 // Some elementary notes on constructor ordering:
48 //
49 // - constructors for member objects in a container class are called
50 // BOTTOM-UP... i.e. member objects are constructed before the
51 // containing object (see Lippman p. 290)
52 // - constructors for a class in a derivation hierarchy are called
53 // TOP-DOWN... i.e. parent classes are constructed before derived
54 // classes (see Lippman p. 433)
55 //
56 // So, the cvLevelSetStructuredGrid constructor will be invoked BEFORE the
57 // constructor of any derived class (e.g. cvLevelSetDenseGrid, cvLevelSetSparseGrid).
58 //
59 // The idea is that cvLevelSetStructuredGrid can set up the abstract, logical
60 // notion of the grid. Derived classes will deal with the specifics
61 // of their physical implementation (e.g. dense vs. sparse storage).

63 cvLevelSetStructuredGrid::cvLevelSetStructuredGrid( double h[], int dims[], double o[] )

65  if ( (dims[0] < 1) || (dims[1] < 1) || (dims[2] < 1) ) {
66  return;
67  }

72 
73  if ( (h[0] < 0.0) || (h[1] < 0.0) || (h[2] < 0.0) ) {
74  return;
75  }
76  hv_[0] = h[0];
77  hv_[1] = h[1];
78  hv_[2] = h[2];

102  assert(0);
103  }

105  grid_ = NULL; // May be overridden in a moment if derived class

107  init_ = 0;
108  curr_ = NULL;

190  double v[3];
191  int forceMinVFlag;
192  double toDot[3] = {0.0, 0.0, 0.0}; // initialization is critical!
193 
194  // Stuff to build up the set of velocity vectors:
195  vtkPolyData *pd = vtkPolyData::New();
196  vtkPoints *pts = vtkPoints::New();
197  vtkFloatingPointArrayType *vec = vtkFloatingPointArrayType::New();
198  vec->SetNumberOfComponents(3);
199  pts->Allocate(100,100);
200  vec->Allocate(100,100);
201  vtkFloatingPointType zlsf[3];
202  vtkFloatingPointType vf[3];
203 
204  if ( ! vfn->Valid() ) {
205  return CV_ERROR;
206  }
207 
208  tol = relTol_ * minh_;
209  k = 0;
210 
211  FindK();
212  FindN();
213  ClearActive();
214  ClearForceMinV();
215 
216  // Foreach grid node:
217  InitIter();
218  while ( currNode = GetNext() ) {
219 
220  // If phi(i) ~= 0:
221  if ( fabs( currNode->phi_ ) < tol_ ) {
222  zls[0] = currNode->pos_[0];
223  zls[1] = currNode->pos_[1];
224  zls[2] = currNode->pos_[2];
225  status = vfn->Evaluate( zls, &f0, &f1, v, &forceMinVFlag, toDot );
226  if ( status != CV_OK ) {
227  break; // i.e. out of global node loop
228  }
229  mag = fabs( f0 + f1 );
230 
231  AssignNode( currNode, f0, f1, forceMinVFlag, toDot );
232 
233  zlsf[0] = zls[0];
234  zlsf[1] = zls[1];
235  zlsf[2] = zls[2];
236  //pts->InsertPoint( k, zlsf );
237  pts->InsertNextPoint(zlsf);
238  vf[0] = v[0];
239  vf[1] = v[1];
240  vf[2] = v[2];
241  //vec->InsertTuple( k, vf );
242  vec->InsertNextTuple(vf);
243  k++;
244 
245  continue;
246  }
247 
248  // Look for active edges radiating from i:
249  GetAdjacentIxs( currNode, adjIxs );
250  for (j = 0; j < 6; j++) {
251  ix = adjIxs[j];
252  if ( ix >= 0 ) {
253  adjNode = &(grid_[ix]);
254 
255  // If this pair of nodes have phi of opposite sign, then:
256  // - set the direction code c
257  // - interpolate the zls position
258  // - evaluate velocity at the zls
259  // - set active bits for curr and adj if not already set
260  // - assign v components as above
261  if ( ( currNode->phi_ * adjNode->phi_ ) < 0.0 ) {
262  if ( j < 2 ) {
263  c = 'x';
264  } else if ( j < 4 ) {
265  c = 'y';
266  } else if ( j < 6 ) {
267  c = 'z';
268  }
269  InterpZLS( currNode, adjNode, c, zls );
270  status = vfn->Evaluate( zls, &f0, &f1, v, &forceMinVFlag, toDot );
271  if ( status != CV_OK ) {
272  break; // i.e. out of local radiating edge loop
273  }
274  mag = fabs( f0 + f1 );
275 
276  AssignNode( currNode, f0, f1, forceMinVFlag, toDot );
277  AssignNode( adjNode, f0, f1, forceMinVFlag, toDot );
278 
279  zlsf[0] = zls[0];
280  zlsf[1] = zls[1];
281  zlsf[2] = zls[2];
282  //pts->InsertPoint( k, zlsf );
283  pts->InsertNextPoint( zlsf );
284  vf[0] = v[0];
285  vf[1] = v[1];
286  vf[2] = v[2];
287  //vec->InsertTuple( k, vf );
288  vec->InsertNextTuple(vf);
289  k++;
290 
291  continue;
292  }
293  }
294  }
295  if ( status != CV_OK ) {
296  break; // i.e. out of global node loop
297  }
298  }
299 
300  if ( status != CV_OK ) {
301 
302  pts->Delete();
303  vec->Delete();
304  pd->Delete();
305 
306  return CV_ERROR;
307  }
308 
309  pd->SetPoints( pts );
310  pd->GetPointData()->SetVectors( vec );
311 
312  if ( velocityVectors_ != NULL ) delete velocityVectors_;
313  velocityVectors_ = new cvPolyData( pd );
314 
315  pts->Delete();
316  vec->Delete();
317  pd->Delete();
318 
319  deltaPhiValid_ = 0;
320 
321  return CV_OK;
322 }

331 double cvLevelSetStructuredGrid::ComputeDeltaPhi( double factor )
332 {
333  cvLevelSetNode *currNode;
334  double f0Contrib, f1Contrib;
335  double maxVal, minVal;
336  double tmp, f1MagGradPhi;
337  double phi_t;
338  double v;
339  double dt;
340 
341  dt = ComputeDt( factor );
342 
343  // ComputeDt returns a negative value if maxF < tol_, signifying a
344  // stop condition which is independent of the user-specified stopV
345  // parameter.
346  if ( dt < 0.0 ) {
347  return -1.0;
348  }
349 
350  if ( deltaPhiValid_ ) {
351  return dt;
352  }
353 
354  FindD0();
355  FindDelPlus();
356  FindDelMinus();
357  FindN();
358 
359  InitIter();
360  while ( currNode = GetNext() ) {
361 
362  // This should be relevant for cvLevelSetSparseGrid only, but in general is
363  // not a harmful thing to include at the abstract cvLevelSetStructuredGrid
364  // level:
365  if ( ! ( currNode->state_ & CV_NODE_COVERED ) &&
366  ! ( currNode->state_ & CV_NODE_ACTIVE ) ) {
367  continue;
368  }
369 
370  // Choose the right entropy-satisfying approximation for use with
371  // the constant velocity term F0 in computing the contribution of
372  // this term to phi_t:
373  maxVal = maximum( currNode->F0_, 0.0 );
374  minVal = minimum( currNode->F0_, 0.0 );
375  f0Contrib = maxVal * currNode->delPlus_ + minVal * currNode->delMinus_;
376 
377  // Put this in on 2/16/00 as part of an attempt to deal with
378  // singularities in the distance function:
379  /*
380  if ( ( fabs(currNode->delPlus_) < tol_ ) &&
381  ( fabs(currNode->delMinus_) < tol_ ) ) {
382  f0Contrib = IntSign( currNode->F0_, tol_ ) * relTol_;
383  }
384  */
385 
386  // Use a central difference approximation to the magnitude of the
387  // gradient with the curvature-dependent portion of velocity
388  // (since curvature dependence corresponds to diffusive flow of
389  // information):
390  tmp = currNode->d0_[0] * currNode->d0_[0];
391  tmp += currNode->d0_[1] * currNode->d0_[1];
392  tmp += currNode->d0_[2] * currNode->d0_[2];
393  f1MagGradPhi = sqrt( tmp );
394  f1Contrib = currNode->F1_ * f1MagGradPhi;
395 
396  phi_t = - ( f0Contrib + f1Contrib );
397 
398  currNode->velocity_ = f0Contrib + f1Contrib;
399  currNode->deltaPhi_ = dt * phi_t;
400 
401  // This is a simple but important mechanism by which we can allow
402  // singularities to move away from the phi==0 surface. The
403  // problem is that at singularities, where the distance function
404  // forms a cusp, curvature values blow up. In the FindK methods
405  // we (arbitrarily) choose oneOverTol_ as the value which
406  // represents "blown up" curvatures. At the same time, values of
407  // |grad(phi)| at these singularities will tend towards zero, and
408  // we have the situation where computed velocity --> 0, and the
409  // cusp will not evolve. Here, by forcing a bit of evolution at
410  // cusps, we allow these cusps to move away from the phi==0
411  // surface, where their blown-up curvatures are actually seen
412  // during velocity evaluation (cusps away from phi==0 do not
413  // contribute large curvatures to computed velocities).
414 
415  if ( ( fabs(f1MagGradPhi) < relTol_ ) &&
416  ( currNode->state_ & CV_NODE_ACTIVE ) ) {
417  currNode->deltaPhi_ = - IntSign( currNode->phi_, tol_ ) * relTol_;
418  currNode->velocity_ = - currNode->deltaPhi_ / dt;
419  }
420 
421  // Check for CFL violation:
422  v = fabs( currNode->F0_ + currNode->F1_ );
423  if ( 2*v*dt > minh_ ) {
424  if ( fabs( 2*v*dt - minh_ ) > tol_ ) {
425  printf( " CFL violation (nodeId=%d): 2*v*dt = %f\n",
426  currNode->index_, 2*v*dt );
427  printf(" minH [%f]\t maxF [%f]\t v [%f]\t dt [%f]\n",
428  minh_, GetMaxF(), v, dt );
429  }
430  }
431  }
432 
433  deltaPhiValid_ = 1;
434 
435  return dt;
436 }

1096  currNode->K_ = 1.0;
1097  } else {

1137  num += ( d0x[0] + d0y[1] ) * d0[2] * d0[2];
