541 // MakeApproxCurveLoop
542 // -------------------
543 // An approximating curve is one that is NOT constrained to pass
544 // through the given points.
545 
546 int cvParasolidSolidModel::MakeApproxCurveLoop( cvPolyData *pd, double tol, int closed )
547 {
548  double *ord_pts;
549  int num_pts;
550 
551  // BTW, the reason we always do this check for geom_ != NULL is
552  // because we want to be invoking construction methods on NULL
553  // models... that is, models which do not already have a solid
554  // associated with them. If you want to use an already-allocated
555  // cvSolidModel to invoke a new constructive method, call the Clear
556  // method first.
557  if ( geom_ != NULL ) {
558  return CV_ERROR;
559  }
560 
561  // We're assuming / requiring that the given cvPolyData describes a
562  // closed loop. Get those points in their connected order.
563  if ( sys_geom_GetOrderedPts( pd, &ord_pts, &num_pts ) != CV_OK ) {
564  return CV_ERROR;
565  }
566 
567  if ( num_pts < 3 && closed != 0) {
568  delete [] ord_pts;
569  return CV_ERROR;
570  }
571 
572  // Note that the PsdUtils_ functions can be more kernel-specific in
573  // their naming. Here we make reference to fitted B-curves, which
574  // is in line with Parasolid phraseology. The (hopefully) more
575  // generic approximating curve is used as the cvSolidModel method
576  // name.
577  if ( PsdUtils_MakeFittedBCurve( num_pts, ord_pts, tol, closed, &geom_ ) != CV_OK ) {
578  delete [] ord_pts;
579  return CV_ERROR;
580  }
581 
582  // See notes at MakeInterpCurveLoop.
583  geom_->Front()->src_pd = sys_geom_DeepCopy( pd );
584 
585  return CV_OK;
586 }

589 // -------------------
590 // MakeInterpCurveLoop
591 // -------------------
592 // An interpolating curve is one that IS constrained to pass through
593 // the given points.
594 
595 int cvParasolidSolidModel::MakeInterpCurveLoop( cvPolyData *pd, int closed )
596 {
597  double *ord_pts;
598  int num_pts;
599 
600  if ( geom_ != NULL ) {
601  return CV_ERROR;
602  }
603 
604  // We're assuming / requiring that the given cvPolyData describes a
605  // closed loop. Get those points in their connected order.
606  if ( sys_geom_GetOrderedPts( pd, &ord_pts, &num_pts ) != CV_OK ) {
607  return CV_ERROR;
608  }
609 
610  if ( num_pts < 3 ) {
611  delete [] ord_pts;
612  return CV_ERROR;
613  }
614 
615  if ( PsdUtils_MakeSplinewiseBCurve( num_pts, ord_pts, closed, &geom_ ) != CV_OK ) {
616  delete [] ord_pts;
617  return CV_ERROR;
618  }
619 
620  // Make a copy of the source cvPolyData, which will be useful later
621  // for creating caps. Note that a possible alternative would be to
622  // re-extract the cvPolyData later only for those profiles which
623  // actually need to be used for caps (i.e. the end profiles). This
624  // has the double advantages of using less space (we don't carry
625  // around this deep copy for each profile) and taking less time (we
626  // don't waste time making these copies, most of which won't be
627  // used). However, the compelling reason to do things this way is
628  // that we do a better job of preserving the cvSolidModel
629  // abstraction. The special requirement that Parasolid imposes for
630  // cap construction should not have to be exposed to the abstract
631  // interface, which it would have to be if we wanted to treat end
632  // profiles in a special way. Also, by avoiding a re-generation of
633  // the cvPolyData, we eliminate the possibility of inconsistencies
634  // that could be introduced as a result (e.g. number of cvPolyData
635  // points used to create cap).
636 
637  geom_->Front()->src_pd = sys_geom_DeepCopy( pd );
638 
639  return CV_OK;
640 }

647 int cvParasolidSolidModel::MakeLoftedSurf( cvSolidModel **curves, int numCurves , char *name )
648 {
649  PsdListT **bcurves;
650  cvParasolidSolidModel *paraPtr;
651  PsdListT *capA, *capB;
652  int i;
653 
654  if ( geom_ != NULL ) {
655  return CV_ERROR;
656  }
657 
658  if ( numCurves < 2 ) {
659  return CV_ERROR;
660  }
661 
662  // Build an array of PsdListT*'s to pass down to the PsdUtils_
663  // interface. For the purposes of this method, we're going to
664  // assume that each individual cvSolidModel describes a SINGLE curve
665  // to be part of the curve set for surface lofting. Furthermore,
666  // that single curve must have a B-form (i.e. PK_BCURVE_t), but we
667  // can defer checking on this condition to the PsdUtils_ layer which
668  // deals with PK types directly.
669 
670  bcurves = new PsdListT* [numCurves];
671  for ( i = 0; i < numCurves; i++ ) {
672  if ( curves[i]->GetKernelT() != SM_KT_PARASOLID ) {
673  delete [] bcurves;
674  return CV_ERROR;
675  }
676  paraPtr = (cvParasolidSolidModel *) curves[i];
677  if ( paraPtr->geom_->NumInList() > 1 ) {
678  delete [] bcurves;
679  return CV_ERROR;
680  }
681  if ( i == 0 ) {
682  if ( PsdUtils_MakeCap( paraPtr->geom_->Front()->src_pd, &capA )
683  != CV_OK ) {
684  delete [] bcurves;
685  return CV_ERROR;
686  }
687  bcurves[i] = capA;
688  } else if ( i == (numCurves - 1) ) {
689  if ( PsdUtils_MakeCap( paraPtr->geom_->Front()->src_pd, &capB )
690  != CV_OK ) {
691  delete [] bcurves;
692  return CV_ERROR;
693  }
694  bcurves[i] = capB;
695  } else {
696  bcurves[i] = paraPtr->geom_;
697  }
698  }
699 
700  /*
701  if ( PsdUtils_MakeLoftedBSurf( numCurves, bcurves, &geom_ ) != CV_OK ) {
702  return CV_ERROR;
703  }
704  */
705  if ( PsdUtils_MakeLoftedSurf( numCurves, bcurves, &geom_ , name ) != CV_OK ) {
706  return CV_ERROR;
707  }
708 
709  delete [] bcurves;
710  PsdUtils_DeleteListOnly( capA );
711  PsdUtils_DeleteListOnly( capB );
712 
713  return CV_OK;

752 
753 int cvParasolidSolidModel::Intersect( cvSolidModel *a, cvSolidModel *b,
754  SolidModel_SimplifyT st )
755 {
756  cvParasolidSolidModel *paraAPtr;
757  cvParasolidSolidModel *paraBPtr;
758  int code;
759  PsdListT *copyA, *copyB;
760  PsdSmpT psdSmp;
761 
762  if ( a->GetKernelT() != SM_KT_PARASOLID ) {
763  return CV_ERROR;
764  }
765  if ( b->GetKernelT() != SM_KT_PARASOLID ) {
766  return CV_ERROR;
767  }
768 
769  if ( geom_ != NULL ) {
770  return CV_ERROR;
771  }
772 
773  // Make copies of the operands so that this is not a destructive
774  // operation:
775  paraAPtr = (cvParasolidSolidModel *)( a );
776  paraBPtr = (cvParasolidSolidModel *)( b );
777  if ( PsdUtils_DeepCopy( paraAPtr->geom_, &copyA ) != CV_OK ) {
778  return CV_ERROR;
779  }
780  if ( PsdUtils_DeepCopy( paraBPtr->geom_, &copyB ) != CV_OK ) {
781  return CV_ERROR;
782  }
783 
784  // Translate the given generic simplification code:
785  psdSmp = SmpCode_GenericToPsd( st );
786  if ( psdSmp == PsdSmpT_Invalid ) {
787  printf("ERR: Invalid simplification code received by cvParasolidSolidModel.\n");
788  return CV_ERROR;
789  }
790 
791  // Invoke boolean:
792  if ( PsdUtils_Intersect( copyA, copyB, psdSmp, &geom_ ) != CV_OK ) {
793  return CV_ERROR;
794  }
795 
796  if ( surf_ != NULL ) {
797  delete surf_;
798  surf_ = NULL;
799  }
800  return CV_OK;
801 }
