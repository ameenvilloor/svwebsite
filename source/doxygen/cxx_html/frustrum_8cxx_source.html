387 Description:
388 
389  machine specific (NT) function to check whether the device implied
390  by the root portion of a path E:\users\me\xx or \\mc123\share\yy
391  or the "currently selected device" implies a short name filesystem.

396  filesystem will only support short 8.3 type names. This assumption
397  is true for Win32S applications (IX86 only) and pre NT version 3.5
398  and where the NT registry option "Win32FileSystem"
399  HKEY_LOCAL_MACHINE\System\CurrentControlSet\Control\FileSystem
400  has been set explicitly to 1.
401 
402  The 'short_filesystem' feature is only relevant if your application
403  is ported to IX86 and is required to run under Windows 3.1 / Win32S
404  otherwise your application can just use 7 character file extensions

406 :::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::*/

408 #ifdef _WIN32
409 static int nt_filesystem_type(char* pathname)
410  /* this can be a fully specified file or directory name */
411  /* (only the first part of the pathname is examined) */
412  /* or it can be blank or null implying "current device" */
413  {
414  int default_return;
415  int res;
416  char *root_of_filesystem;
417  char root_of_file_system_name[256] = "";
418  char filesystemtype[256] = "";
419  DWORD maxnamelength;
420  DWORD filesystemtypelen;
421 #ifdef _M_IX86
422  /* running on an I386 machine where FAT filesystems are more common */
423  default_return = PS_SHORT_NAME;
424 #else
425  /* running on AlphaNT/MipsNT where NTFS filesystems are more common */
426  default_return = PS_LONG_NAME;
427 #endif
428  res = default_return;
429  root_of_filesystem = 0L;
430  maxnamelength = (DWORD) -1;
431  filesystemtypelen = (DWORD) 256;
432  if (pathname != 0L && strlen( pathname ) > 0)
433  {
434  /* examine first few characters in pathname */
435  strcpy( root_of_file_system_name, pathname );
436  if ( root_of_file_system_name[0] == dir_separator_c &&
437  root_of_file_system_name[1] == dir_separator_c )
438  {
439  /* look for a UNC name of the form \<node><share>\... */
440  /* constructed filesystem name must must include the trailing */
441  /* backslash after sharename see KnowledgeBase PSS ID Q119219 */
442  int i;
443  int len;
444  int share_found = 0;
445  len = (int) (strlen( root_of_file_system_name ));
446  for ( i = 2 ; i < len && root_of_filesystem == 0L; i++ )
447  {
448  if ( root_of_file_system_name[i] == dir_separator_c )
449  {
450  if ( share_found )
451  /* second single backslash marks end of share name */
452  {
453  root_of_file_system_name[i + 1] = '\0';
454  root_of_filesystem = root_of_file_system_name;
455  }
456  else
457  share_found = 1;
458  }
459  }
460  }
461  else
462  /* look for a name of form Z:\... or Z:... */
463  {
464  if (root_of_file_system_name[1] == ':')
465  {
466  /* assume root_of_file_system_name[0] contains drive letter */
467  /* (trailing slash makes this a 'root' directory specifier) */
468  root_of_file_system_name[2] = dir_separator_c;
469  root_of_file_system_name[3] = '\0';
470  root_of_filesystem = root_of_file_system_name;
471  }
472  }
473  }
474 
475  if (GetVolumeInformation(root_of_filesystem,/* receive: system name */
476  0L, /* return : volume name */
477  (DWORD) 0, /* */
478  0L, /* return : no serial num */
479  &maxnamelength, /* return : name length */
480  0L, /* return : no system flags */
481  filesystemtype, /* return : file system */
482  filesystemtypelen) == 1 )
483  /* GetVolumeInformation function is implemented in WIN32 and Win32S */
484  /* so this function is always available. If the pathname is null or */
485  /* it doesn't include any device information, GetVolumeInformation */
486  /* returns whether or not the current device is a FAT file system. */
487  /* (GetVolumeInformation returns zero if the device does not exist) */
488  {
489  if (strcmp( filesystemtype, "FAT" ) == 0)
490  res = PS_SHORT_NAME;
491  else if (maxnamelength <= 0)
492  /* not sure (return default value) */
493  res = default_return;
494  else
495  {
496  /* use maxnamelength to decide (12 chars in name "abcdefgh.ijk") */
497  if ( maxnamelength <= 12 )
498  res = PS_SHORT_NAME;
499  else
500  res = PS_LONG_NAME;
501  }
502  }
503  return res;
504  }
