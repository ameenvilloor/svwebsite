578 void siftDownEdges(int **edges, int root, int bottom, int array_size)
579 {
580  int done, maxChild;
581  int temp0,temp1;
582 
583  done = 0;
584  while ((root*2 <= bottom) && (!done))
585  {
586  if (root*2 == bottom)
587  maxChild = root * 2;
588  else if ((edges[0][root * 2] > edges[0][root * 2 + 1]) || \
589  ((edges[0][root * 2] == edges[0][root *2 + 1]) && (edges[1][root*2] > edges[1][root*2+1])) )
590  maxChild = root * 2;
591  else
592  maxChild = root * 2 + 1;
593 
594  // hack since this seems to constantly end
595  if (root < 0 || root >= array_size) break;
596  if (maxChild < 0 || maxChild >= array_size) {
597  fprintf(stdout,"warning: illegal maxChild (%i)\n",maxChild);
598  break;
599  }
600 
601  if ( (edges[0][root] < edges[0][maxChild]) || \
602  ((edges[0][root] == edges[0][maxChild]) && (edges[1][root] < edges[1][maxChild]))
603  )
604 
605  {
606  temp0 = edges[0][root];
607  temp1 = edges[1][root];
608  edges[0][root] = edges[0][maxChild];
609  edges[1][root] = edges[1][maxChild];
610  edges[0][maxChild] = temp0;
611  edges[1][maxChild] = temp1;
612  root = maxChild;
613  }
614  else
615  done = 1;
616  }
