162 int cvLevelSetVelocitySmooth::Evaluate( double pos[], double *f0, double *f1, double v[],
163  int *forceFlag, double toDot[] )
164 {
165  double n[3];
166  double k;
167  double km, kg;
168  double ks[2];
169 
170  if ( ! Valid() ) return CV_ERROR;
171  if ( ls_->GetGrid()->InterpN( pos, n ) != CV_OK ) return CV_ERROR;
172  if ( ls_->GetGrid()->GetDim() == 3 ) {
173  switch (ktype_) {
174  case VS_Mean_K:
175  if ( ls_->GetGrid()->InterpKm( pos, &k ) != CV_OK ) return CV_ERROR;
176  break;
177  case VS_Gaussian_K:
178  if ( ls_->GetGrid()->InterpKg( pos, &k ) != CV_OK ) return CV_ERROR;
179  break;
180  case VS_PrincipleK1_K:
181  if ( ls_->GetGrid()->InterpKg( pos, &kg ) != CV_OK ) return CV_ERROR;
182  if ( ls_->GetGrid()->InterpKm( pos, &km ) != CV_OK ) return CV_ERROR;
183  Compute3dks( kg, km, tol_, ks );
184  k = ks[0];
185  break;
186  case VS_PrincipleK2_K:
187  if ( ls_->GetGrid()->InterpKg( pos, &kg ) != CV_OK ) return CV_ERROR;
188  if ( ls_->GetGrid()->InterpKm( pos, &km ) != CV_OK ) return CV_ERROR;
189  Compute3dks( kg, km, tol_, ks );
190  k = ks[1];
191  break;
192  default:
193  assert(0);
194  }
195  } else {
196  ls_->GetGrid()->InterpK( pos, &k );
197  }
198 
199  if ( k < lower_kt_ ) {
200  // *f0 = 1.0;
201  *f0 = 0.0;
202  *f1 = (lower_kt_ - k);
203  } else if ( k > upper_kt_ ) {
204  // *f0 = -1.0;
205  *f0 = 0.0;
206  *f1 = - (k - upper_kt_);
207  } else {
208  *f0 = 0.0;
209  *f1 = 0.0;
210  }
211 
212  *forceFlag = 0;
213  toDot[0] = 0.0;
214  toDot[1] = 0.0;
215  toDot[2] = 0.0;
216 
217  if ( v != NULL ) {
218  v[0] = n[0] * (*f0);
219  v[1] = n[1] * (*f0);
220  v[2] = n[2] * (*f0);
221  }
222 
223  return CV_OK;
224 }
