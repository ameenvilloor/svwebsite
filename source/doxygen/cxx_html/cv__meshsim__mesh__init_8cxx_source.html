110 int Meshsimmesh_Init( Tcl_Interp *interp )
111 {
112  try
113  {
114  // Initialize the MeshSim libraries
115  printf(" %-12s %s\n", "MeshSim:", MS_version());
116 
117  MS_init();
118  } catch (...) {
119  fprintf(stdout," ERROR Initializng MeshSim.\n");
120  return TCL_OK;
121  }
122 #if defined(MESHSIM_LICENSE_IN_WIN32_REGISTRY)
123  try
124  {
125  MeshSim_Win32ReadRegistrySimRegister("MESHSIM_KEY_ATTRIBUTES");
126  MeshSim_Win32ReadRegistrySimRegister("MESHSIM_KEY_PARASOLID");
127  MeshSim_Win32ReadRegistrySimRegister("MESHSIM_KEY_DISCRETE");
128  MeshSim_Win32ReadRegistrySimRegister("MESHSIM_KEY_SURFACE");
129  MeshSim_Win32ReadRegistrySimRegister("MESHSIM_KEY_VOLUME");
130  MeshSim_Win32ReadRegistrySimRegister("MESHSIM_KEY_EXPORT");
131  MeshSim_Win32ReadRegistrySimRegister("MESHSIM_KEY_ADV");
132  MeshSim_Win32ReadRegistrySimRegister("MESHSIM_KEY_ADAPT");
133 } catch (...) {
134  fprintf(stdout," ERROR MeshSim Registry.\n");
135  return TCL_OK;
136 }
137 #elif defined(MESHSIM_USE_LICENSE_FILE)
138 try
139 {
140 
141  Sim_readLicenseFile(NULL);
142 
143 } catch (...) {
144  fprintf(stdout," Note: Missing or expired MeshSim License. This causes a General error.\n");
145  fprintf(stdout," This means MeshSim functionality will be unavailable.\n");
146  return TCL_OK;
147 }
148 #elif defined(MESHSIM_EMBED_LICENSE_KEYS)
149 #include "../../../Licenses/MeshSim/meshsim_license.h"
150 try
151 {
152  Sim_registerKey(MESHSIM_KEY_ATTRIBUTES);
153  Sim_registerKey(MESHSIM_KEY_PARASOLID);
154  Sim_registerKey(MESHSIM_KEY_DISCRETE);
155  Sim_registerKey(MESHSIM_KEY_SURFACE);
156  Sim_registerKey(MESHSIM_KEY_VOLUME);
157  Sim_registerKey(MESHSIM_KEY_EXPORT);
158  Sim_registerKey(MESHSIM_KEY_ADV);
159  Sim_registerKey(MESHSIM_KEY_ADAPT);
160 } catch (...) {
161  fprintf(stdout," ERROR MeshSim Embeded Keys.\n");
162  return TCL_OK;
163 }
164 #else
165 fprintf(stdout,"ERROR: need to register license keys somehow!\n");
166 return CV_ERROR;
167 #endif
168 try{
169  SimModel_start();
170  #ifdef USE_PARASOLID
171  if (SimParasolid_start( 0 ) != 0) {
172  fprintf(stdout,"ERROR starting MeshSim Parasolid Interface!\n");
173  return CV_ERROR;
174  }
175  #endif
176 
177  SimAdvMeshing_start();
178 }
179 catch (...) {
180  fprintf(stdout," ERROR starting MeshSim. Will not be available!\n");
181  return TCL_OK;
182 }
183  // Get the main mesh
184  MeshKernelRegistryMethodPtr pMeshKernelRegistryMethod =
185  (MeshKernelRegistryMethodPtr) Tcl_GetAssocData( interp, "MeshSystemRegistrar", NULL);
186 
187  if (pMeshKernelRegistryMethod != NULL) {
188  cvMeshSystem* meshSimSystem = new cvMeshSimMeshSystem();
189  if ((*pMeshKernelRegistryMethod)( cvMeshObject::KERNEL_MESHSIM, meshSimSystem ) == CV_OK) {
190  //printf(" MeshSim module registered\n");
191  Tcl_CreateCommand( interp, "meshsim_mesh_available", MeshSimMesh_AvailableCmd,
192  (ClientData)NULL, (Tcl_CmdDeleteProc *)NULL );
193  }
194  }

196  return TCL_ERROR;
197  }
198 
199  return TCL_OK;
