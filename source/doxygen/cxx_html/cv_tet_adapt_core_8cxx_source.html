85  vtkSmartPointer<vtkDoubleArray> averageHessians =
86  vtkSmartPointer<vtkDoubleArray>::New();
87  vtkSmartPointer<vtkDoubleArray> nodalHessians =
88  vtkSmartPointer<vtkDoubleArray>::New();
89  vtkSmartPointer<vtkIdList> patchElements =
90  vtkSmartPointer<vtkIdList>::New();
91 
92  numVerts = mesh->GetNumberOfPoints();
93  bool *pointOnSurface = new bool[numVerts];
94 
95  averageHessians->SetNumberOfComponents(6);
96  averageHessians->Allocate(numVerts,10000);
97  averageHessians->SetNumberOfTuples(numVerts);
98  averageHessians->SetName("averagehessians");
99 
100  nodalHessians = vtkDoubleArray::SafeDownCast(mesh->GetPointData()->GetArray("hessians"));
101 
102  //Have no purpose for point mapping here
103  getSurfaceBooleans(mesh,pointOnSurface,NULL);
104 
105  for (pointId=0;pointId<numVerts;pointId++)
106  {
107  numSurroundingVerts = 0;
108  for (i=0;i<6;i++)
109  {
110  averageHessian[i] = 0.0;
111  nodalHessian[i] = nodalHessians->GetComponent(pointId,i);
112  averageHessian[i] = averageHessian[i] + nodalHessian[i];
113  }
114  mesh->GetPointCells(pointId,patchElements);
115 
116  vtkSmartPointer<vtkIdList> neighborPoints =
117  vtkSmartPointer<vtkIdList>::New();
118  for (cellId=0;cellId<patchElements->GetNumberOfIds();cellId++)
119  {
120  mesh->GetCellPoints(patchElements->GetId(cellId),npts,pts);
121  for (j=0;j<npts;j++)
122  {
123  if (pointOnSurface[pts[j]] == false)
124  {
125  neighborPoints->InsertUniqueId(pts[j]);
126  }
127  }
128  }
129  numSurroundingVerts = neighborPoints->GetNumberOfIds();
130 
131  if (numSurroundingVerts != 0)
132  {
133  for (i=0;i<numSurroundingVerts;i++)
134  {
135  checkPt = neighborPoints->GetId(i);
136  for (j=0;j<6;j++)
137  {
138  newHessian = nodalHessians->GetComponent(checkPt,j);
139  averageHessian[j] = averageHessian[j] + newHessian;
140  }
141  }
142  for (i=0;i<6;i++)
143  {
144  averageHessian[i] = averageHessian[i]/numSurroundingVerts;
145  }
146  }
147  averageHessians->SetTuple(pointId,averageHessian);
148  }
149 
150  mesh->GetPointData()->AddArray(averageHessians);
151  mesh->GetPointData()->SetActiveScalars("averagehessians");
152 
153  delete [] pointOnSurface;
154 }

573 // -----------------------------

585 void cvTetAdaptCore::setSizeFieldUsingHessians(vtkUnstructuredGrid *mesh,
586  tetgenio *inmesh,
587  double factor,
588  double hmax,
589  double hmin)
590 {
591  int i,j,k;
592  int nshg;
593  int ierr;
594  int three = 3;
595  int bdryNumNodes = 0;
596  double xyz[3];
597  double T[3][3];
598  double tol=1.e-12;
599  double maxEigenval=0;
600  double eloc; // local error at a vertex
601  double etot=0.; // total error for all vertices
602  double emean; // emean = etot / nv
603  double elocmax=0.; // max local error
604  double elocmin=1.e20; // min local error
605  double z[3][3];
606  vtkIdType pointId,averageHessian;
607 
608  vtkSmartPointer<vtkDoubleArray> averageHessians =
609  vtkSmartPointer<vtkDoubleArray>::New();
610 
611  nshg = mesh->GetNumberOfPoints();
612  averageHessians = vtkDoubleArray::SafeDownCast(mesh->GetPointData()->GetArray("averagehessians"));
613 
614  // struct Hessian contains decomposed values
615  // mesh sizes and directional information
616  Hessian *hess = new Hessian[nshg];
617 
618  inmesh->numberofpointmtrs = 3;
619  inmesh->pointmtrlist = new REAL[nshg*inmesh->numberofpointmtrs];
620 
621  i=0;
622  for (pointId=0;pointId<nshg;pointId++)
623  {
624  mesh->GetPoint(pointId,xyz);
625 
626  getHessian(averageHessians,pointId,T);
627 
628  double eigenVals[3];
629  double e[3];
630  double Tfoo[3][3];
631 
632  // copy T into temporary buffer
633  for (j=0;j<3;j++)
634  {
635  for (k=0;k<3;k++)
636  {
637  Tfoo[j][k] = T[j][k];
638  }
639  }
640 
641  mytred(&three,&three,Tfoo,eigenVals,e,z);
642 
643  tql2(&three,&three,eigenVals,e,z,&ierr);
644 
645  for (j=0;j<3;j++)
646  {
647  hess[i].h[j] = eigenVals[j];
648  for (k=0;k<3;k++)
649  {
650  hess[i].dir[j][k]=z[j][k];
651  }
652  }
653 
654  hess[i].h[0] = ABS(hess[i].h[0]);
655  hess[i].h[1] = ABS(hess[i].h[1]);
656  hess[i].h[2] = ABS(hess[i].h[2]);
657 
658  if( MAX(hess[i].h[0],MAX(hess[i].h[1],hess[i].h[2])) < tol ) {
659  printf("Warning: zero maximum eigenvalue for node %d !!!\n",i);
660  printf(" %f %f %f\n", hess[i].h[0],
661  hess[i].h[1],hess[i].h[2]);
662  continue;
663  }
664 
665  // estimate relative interpolation error
666  // needed for scaling metric field (mesh size field)
667  // to get an idea refer Appendix A in Li's thesis
668  eloc=maxLocalError(mesh,pointId,T);
669  etot += eloc;
670  if( eloc>elocmax ) elocmax=eloc;
671  if( eloc<elocmin ) elocmin=eloc;
672 
673  i++;
674  }
675 
676  printf("Info: Reading hessian... done...\n");
677 
678  emean = etot / nshg;
679  printf("\n Info on relative interpolation error: ");
680  printf(" total: %f\n",etot);
681  printf(" mean: %f\n",emean);
682  printf(" max local: %f\n",elocmax);
683  printf(" min local: %f\n",elocmin);
684 
685  eloc = emean*factor;
686 
687  printf("\n towards uniform local error distribution of %f\n", eloc);
688  printf(" with max edge length=%f; min edge length=%f\n\n",hmax,hmin);
689 
690  adaptSimLog<<"Strategy chosen is isotropic adaptation, i.e., size-field driven"<<endl;
691  adaptSimLog<<"Info on relative interpolation error :"<<endl;
692  adaptSimLog<<"total : "<<etot<<endl;
693  adaptSimLog<<"mean : "<<emean<<endl;
694  adaptSimLog<<"factor : "<<factor<<endl;
695  adaptSimLog<<"min. local : "<<elocmin<<endl;
696  adaptSimLog<<"max. local : "<<elocmax<<endl;
697  adaptSimLog<<"towards uniform local error distribution of "<<eloc<<endl;
698  adaptSimLog<<"with min. edge length : "<<hmin<<endl;
699  adaptSimLog<<"with max. edge length : "<<hmax<<endl;
700 
701  int foundHmin = 0;
702  int foundHmax = 0;
703  int hminCount = 0;
704  int hmaxCount = 0;
705  int bothHminHmaxCount = 0;
706  int insideSphereCount = 0;
707 
708  double tol2, tol3;
709 
710  i=0;
711  for (pointId=0;pointId<nshg;pointId++)
712  {
713  tol2 = 0.01*hmax;
714  tol3 = 0.01*hmin;
715  foundHmin = 0;
716  foundHmax = 0;
717  for( j=0; j<3; j++ ) {
718  if( hess[i].h[j] < tol )
719  hess[i].h[j] = hmax;
720  else {
721  hess[i].h[j] = sqrt(eloc/hess[i].h[j]);
722  if( hess[i].h[j] > hmax )
723  hess[i].h[j] = hmax;
724  if( hess[i].h[j] < hmin )
725  hess[i].h[j] = hmin;
726  }
727  }
728
