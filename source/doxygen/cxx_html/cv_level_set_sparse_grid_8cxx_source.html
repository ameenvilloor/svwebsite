303 // -------
304 // InitPhi
305 // -------
306 // This is the publicly-callable cvPolyData-based phi initialization.
307 // This means that it handles the case where init_ is NOT true. In
308 // this case, sign initialization could be done by either:
309 // - cvSolidModel point classify (SLOW)
310 // - Graphics Gems algorithms (faster)
311 
312 // Although cvSolidModel's CAN be much more general than the simple
313 // polygons / polyhedra handled by the Graphics Gems algorithms, there
314 // is currently no easy way to SPECIFY structure of non-simple objects
315 // through the cvPolyData interface which we're dealing with in this
316 // method. Generally, by "simple" I mean objects which consist of a
317 // single, closed, manifold region. Note that cvPolyData can be readily
318 // used to transmit such objects, including ones with holes. Since
319 // the point-in-polyhedron algorithm from Graphics Gems does NOT
320 // appear to function for simple objects with holes, cvSolidModel can
321 // provide a more general solution for this method than other
322 // algorithms. However, since general objects can already be used for
323 // initialization via the cvSolidModel-based InitPhi, I will assume that
324 // all cvPolyData's passed to this method are assumed to be simple
325 // (i.e. single closed manifold region) and to have NO HOLES.
326 
327 // One important distinction, however, remains: cvSolidModel methods
328 // generally will be able to distinguish among three cases: in, out
329 // and on. The Graphics Gems algorithms do not generally make
330 // consistent distinction of the on case, classifying such points as
331 // either in or out instead. The reason this is OK is that the +/-
332 // sign which is (erroneously) applied to "on" points will be caught
333 // in the ConstructHT(_Sweep) methods: when distance is computed and
334 // found to be under a certain precision-related magnitude, then sign
335 // is ignored, and the "on" state and zero distance are applied.
336

339  if ( ! init_ ) {
340  if ( InitSign_Internal( front ) != CV_OK ) { // on success, init_ --> 1

342  }
343  }
344  return InitPhi_Internal( front );
345 }

489 // -------
490 // InitPhi
491 // -------
492 // There's a somewhat pathological case which can arise. If there is
493 // an isolated pixel which is equal to the given threshold, then it
494 // will be labelled here as having STATE_PHI_ON, since in general we
495 // expect values at the threshold to be on the initial zero level
496 // set. However, with an isolated pixel, vtk's contouring does not
497 // generate any contour geometry. As a result, the computed distance
498 // at this position will be something larger than the STATE_PHI_ON
499 // topological tag would lead us to expect.
500 
501 int cvLevelSetSparseGrid::InitPhi( cvStrPts *img, double thr )
502 {
503  int i, j, k, n;
504  double pos[3];
505  cvPolyData *front;
506  vtkStructuredPoints *vtksp;
507  vtkDataArray *vtkdata;
508  int numdata;
509  float *fdata;
510  Image_T *local_img;
511  int imgDims[3];
512  vtkFloatingPointType fpixDims[3];
513  double pixDims[3];
514  vtkFloatingPointType vtkorig[3];
515  double orig[3];
516  int status;
517  double intensity;
518  double delta;
519 
520  // Set topological state {on, out, in} based on data compared to
521  // given threshold:
522 
523  vtksp = (vtkStructuredPoints*)(img->GetVtkPtr());
524  vtkdata = vtksp->GetPointData()->GetScalars();
525  if ( vtkdata == NULL ) {
526  printf( "ERR: cvStrPts used for lset init needs to have scalar data\n" );
527  return CV_ERROR;
528  }
529 
530  // Copy vtk scalars into a local array so we can make an Image_T
531  // which we can then use for bi-/tri-linearly interpolated queries:
532  numdata = vtkdata->GetNumberOfTuples();
533  fdata = new float [numdata];
534  for ( i = 0; i < numdata; i++ ) {
535  fdata[i] = (float)(vtkdata->GetTuple1(i));
536  }
537  vtksp->GetDimensions( imgDims );
538  vtksp->GetSpacing( fpixDims );
539  vtksp->GetOrigin( vtkorig );
540  pixDims[0] = fpixDims[0];
541  pixDims[1] = fpixDims[1];
542  pixDims[2] = fpixDims[2];
543  orig[0] = vtkorig[0] - (pixDims[0] / 2.0);
544  orig[1] = vtkorig[1] - (pixDims[0] / 2.0);
545  if ( imgDims[2] > 1 ) {
546  orig[2] = vtkorig[2] - (pixDims[0] / 2.0);
547  } else {
548  orig[2] = vtkorig[2];
549  }
550 
551  // CreateImage allocates its own memory, so we don't need fdata
552  // anymore once this call returns:
553  local_img = CreateImage( (void*)fdata, numdata, "-float", imgDims, pixDims );
554  delete [] fdata;
555  if ( local_img == NULL ) {
556  printf( "ERR: couldn't create scalar map from given cvStrPts\n" );
557  return CV_ERROR;
558  }
559  SetLowerLeft( local_img, orig );
560 
561  for (k = 0; k < K_; k++) {
562  for (j = 0; j < J_; j++) {
563  for (i = 0; i < I_; i++) {
564 
565  n = IJKToDenseIx( i, j, k );
566 
567  pos[0] = i * hv_[0] + origin_[0];
568  pos[1] = j * hv_[1] + origin_[1];
569  pos[2] = k * hv_[2] + origin_[2];
570 
571  status = GetIntensity( local_img, pos, &intensity );
572  if ( ! status ) {
573  printf( "ERR: image query error\n" );
574  Image_Delete( local_img );
575  return CV_ERROR;
576  }
577 
578  delta = thr - intensity;
579  if ( fabs( delta ) < tol_ ) {
580  (*topoOverlay_)[n] |= STATE_PHI_ON;
581  } else if ( delta > 0.0 ) {
582  (*topoOverlay_)[n] |= STATE_PHI_OUTSIDE;
583  } else {
584  (*topoOverlay_)[n] |= STATE_PHI_INSIDE;
585  }
586  }
587  }
588  }
589  Image_Delete( local_img );
590  init_ = 1;
591 
592  // Extract contour geometry based on given threshold:
593 
594  vtkContourFilter *contour = vtkContourFilter::New();
595  contour->SetInputDataObject( img->GetVtkPtr() );
596  contour->SetValue( 0, (vtkFloatingPointType)thr );
597  contour->Update();
598  front = new cvPolyData( contour->GetOutput() );
599  contour->Delete();
600 
601  // Now call InitPhi( cvPolyData* ):
602 
603  bandMethod_ = BAND_SWEEP;
604  return InitPhi( front );
605 }

631 int cvLevelSetSparseGrid::BuildSeedInterface( cvSolidModel *seed )
632 {

634  double boxCtr[3];
635  cvPolyData *seedPd;
636  double bboxSeed[6];
637  double bboxGrid[6];
638  int status;
639 
640  cvSolidModel *logicalSolid;
641  cvSolidModel *seedBound;
642  cvSolidModel *seedIntfSolid;
643 
644  // First, check to see if this is necessary at all. If the seed
645  // model is entirely contained within the logical grid domain, then
646  // we have no need to construct a seed interface because the seed
647  // itself gives us seedInterface_.
648 
649  boxDims[0] = ( I_ - 1 ) * hv_[0]; // think carefully about the fencepost
650  boxDims[1] = ( J_ - 1 ) * hv_[1]; // phenomenon going on here... I_-1
651  boxDims[2] = ( K_ - 1 ) * hv_[2]; // IS correct... this is NOT an image
652 
653  boxCtr[0] = origin_[0] + boxDims[0] / 2;
654  boxCtr[1] = origin_[1] + boxDims[1] / 2;
655  boxCtr[2] = origin_[2] + boxDims[2] / 2;
656 
657  seedPd = seed->GetPolyData(0,0);
658  if ( sys_geom_BBox( seedPd, bboxSeed ) != CV_OK ) {
659  delete seedPd;
660  return CV_ERROR;
661  }
662  bboxGrid[0] = origin_[0];
663  bboxGrid[1] = origin_[0] + boxDims[0];
664  bboxGrid[2] = origin_[1];
665  bboxGrid[3] = origin_[1] + boxDims[1];
666  bboxGrid[4] = origin_[2];
667  bboxGrid[5] = origin_[2] + boxDims[2];
668 
669  if ( ( bboxSeed[0] > bboxGrid[0] ) && ( bboxSeed[1] < bboxGrid[1] ) &&
670  ( bboxSeed[2] > bboxGrid[2] ) && ( bboxSeed[3] < bboxGrid[3] ) &&
671  ( bboxSeed[4] > bboxGrid[4] ) && ( bboxSeed[5] < bboxGrid[5] ) ) {
672  seedInterface_ = seedPd;
673  return CV_OK;
674  }
675 
676  logicalSolid = cvSolidModel::DefaultInstantiateSolidModel();
677  seedBound = cvSolidModel::DefaultInstantiateSolidModel();
678  seedIntfSolid = cvSolidModel::DefaultInstantiateSolidModel();
679 
680  if ( ( logicalSolid == NULL ) || ( seedBound == NULL ) ||
681  ( seedIntfSolid == NULL ) ) {
682  return CV_ERROR;
683  }
684 
685  ClearSeedInterface();
686 
687  if ( dim_ == 2 ) {
688  logicalSolid->MakeBox2d( boxDims, boxCtr );
689  seedBound->MakePoly2d( seedPd );
690 
691  } else if ( dim_ == 3 ) {
692  logicalSolid->MakeBox3d( boxDims, boxCtr );
693  seedBound->SetPoly3dFacetMethod( SM_Facet_Union );
694  seedBound->MakePoly3dSurface( seedPd );
695 
696  } else {
697  return CV_ERROR;
698  }
699 
700  status = seedIntfSolid->Intersect( logicalSolid, seedBound );
701 
702  delete logicalSolid;
703  delete seedPd;
704  delete seedBound;
705 
706  if ( status != CV_OK ) {
707  delete seedIntfSolid;
708  return CV_ERROR;
709  }
710 
711  seedInterface_ = seedIntfSolid->GetPolyData(0,0);
712 
713  delete seedIntfSolid;
714 
715  if ( seedInterface_ == NULL ) {
716  return CV_ERROR;
717  }
718 
719  return CV_OK;
720 }
