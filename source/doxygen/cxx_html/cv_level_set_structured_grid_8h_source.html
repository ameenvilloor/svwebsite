295 int cvLevelSetStructuredGrid::LogicalIxToIJK( int log_ix, int *i, int *j, int *k )
296 {
297  if ( ( log_ix < 0 ) || ( log_ix >= I_*J_*K_ ) ) {
298  return CV_ERROR;
299  }
300  *k = log_ix / (I_*J_);
301  *j = ( log_ix % (I_*J_) ) / I_;
302  *i = log_ix % I_;
303  return CV_OK;
304 }
