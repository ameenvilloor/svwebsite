147 int cmd_proc (char *cmd, int *ok) {
148 
149  int i, j;

154 
155  char name[MAXSTRINGLENGTH];
156 
157  /**************
158  *** body ***
159  **************/
160 
161  if (cmd[0] == '#' || cmd[0] == '\n') {
162  return CV_OK;
163  }
164 
165  sscanf(cmd, "%s", name);
166 
167  debugprint(stddbg,"command being processed is %s. \n",name);
168 
169  for (i = 0; cmd_table[i].name != NULL; i++) {
170  if (!strcmp(name, cmd_table[i].name)) {
171  pt2Function = cmd_table[i].pt2Function;
172  return (*pt2Function)(cmd);
173  //return CV_OK;
174  }
175  }
176 
177  for (i = 0; i < strlen(cmd); i++) {
178  if ((cmd[i] != ' ') && (cmd[i] != '\n')) {
179  fprintf(stderr, "\n **** error: unknown command: %s", cmd);
180  return CV_ERROR;
181  }
182  }
183 
184  return CV_OK;
185 
186 }

218  *end = 0;

220  if (n >= len)

238  else if (c == '\n') {
239  token[i] = '\0';
240  *p_n = n;
241  *end = 1;
242  return CV_OK;

244 
245  token[i++] = c;
246  token[i] = '\0';
247  }
248  }
249  else if (c == '\n') {
250  *end = 1;
251 
252  if (i != 0) {
253  token[i] = '\0';

255  return CV_OK;
