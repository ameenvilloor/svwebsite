66 {
67  int i;
68  ggemsGeoPoint v0, v1, p;
69 
70  ggemsGeoZeroVec ( *n );
71  ggemsGeo_Vet ( verts[0], verts[1], v0 );
72  for ( i = 2; i < n_verts; i++ )
73  {
74  ggemsGeo_Vet ( verts[0], verts[i], v1 );
75  ggemsGeoCrossProd ( &v0, &v1, &p );
76  n->x += p.x; n->y += p.y; n->z += p.z;
77  v0 = v1;
78  }
79

81  if (n_size > 0.0)
82  {
83  ggemsGeoMultVec ( 1/n_size, *n, *n );
84  return 1;
85  }
86  else
87  return 0;

96 Rdouble ggemsgeo_solid_angle (
97  int n_vert, /* number of vertices */
98  ggemsGeoPoint *verts, /* vertex coordinates list */
99  ggemsGeoPoint *p ) /* point to be tested */
100 {
101  int i;
102  Rdouble area = 0.0, ang, s, l1, l2;
103  ggemsGeoPoint p1, p2, r1, a, b, n1, n2;
104  ggemsGeoPoint plane;
105 
106  if ( n_vert < 3 ) return 0.0;
107 
108  ggemsGeoPolyNormal ( n_vert, verts, &plane );
109 
110  /*
111  WARNING: at this point, a practical implementation should check
112  whether p is too close to the polygon plane. If it is, then
113  there are two possibilities:
114  a) if the projection of p onto the plane is outside the
115  polygon, then area zero should be returned;
116  b) otherwise, p is on the polyhedron boundary.
117  */
118 
119  p2 = verts[n_vert-1]; /* last vertex */
120  p1 = verts[0]; /* first vertex */
121  ggemsGeo_Vet ( p1, p2, a ); /* a = p2 - p1 */
122 
123  for ( i = 0; i < n_vert; i++ )
124  {
125  ggemsGeo_Vet(*p, p1, r1);
126  p2 = verts[(i+1)%n_vert];
127  ggemsGeo_Vet ( p1, p2, b );
128  ggemsGeoCrossProd ( &a, &r1, &n1 );
129  ggemsGeoCrossProd ( &r1, &b, &n2 );
130 
131  l1 = ggemsGeoVecLen ( &n1 );
132  l2 = ggemsGeoVecLen ( &n2 );
133  s = ggemsGeoDotProd ( &n1, &n2 ) / ( l1 * l2 );
134  ang = acos ( maximum(-1.0,minimum(1.0,s)) );
135  s = ggemsGeoTripleProd( &b, &a, &plane );
136  area += s > 0.0 ? PI - ang : PI + ang;
137 
138  ggemsGeoMultVec ( -1.0, b, a );
139  p1 = p2;
140  }
141 
142  area -= PI*(n_vert-2);
143 
144  return ( ggemsGeoDotProd ( &plane, &r1 ) > 0.0 ) ? -area : area;
145 }

158 int ggems_CrossingsMultiplyTest( double pgon[], int numverts, double point[] )
159 {
160 register int j, yflag0, yflag1, inside_flag ;
161 register double ty, tx, vtx0[2], vtx1[2], *mark ;
162 
163  tx = point[X] ;
164  ty = point[Y] ;
165 
166  /* Start vtx0 at end of point list: */
167  vtx0[0] = pgon[(numverts-1)*2] ;
168  vtx0[1] = pgon[(numverts-1)*2+1] ;
169  /* get test bit for above/below X axis */
170  yflag0 = ( vtx0[Y] >= ty ) ;
171 
172  /* Wrap vtx1 to beginning: */
173  mark = pgon;
174  vtx1[0] = mark[0] ;
175  vtx1[1] = mark[1] ;
176 
177  inside_flag = 0 ;
178  for ( j = numverts+1 ; --j ; ) {
179 
180  yflag1 = ( vtx1[Y] >= ty ) ;
181  /* Check if endpoints straddle (are on opposite sides) of X axis
182  * (i.e. the Y's differ); if so, +X ray could intersect this edge.
183  * The old test also checked whether the endpoints are both to the
184  * right or to the left of the test point. However, given the faster
185  * intersection point computation used below, this test was found to
186  * be a break-even proposition for most polygons and a loser for
187  * triangles (where 50% or more of the edges which survive this test
188  * will cross quadrants and so have to have the X intersection computed
189  * anyway). I credit Joseph Samosky with inspiring me to try dropping
190  * the "both left or both right" part of my code.
191  */
192  if ( yflag0 != yflag1 ) {
193  /* Check intersection of pgon segment with +X ray.
194  * Note if >= point's X; if so, the ray hits it.
195  * The division operation is avoided for the ">=" test by checking
196  * the sign of the first vertex wrto the test point; idea inspired
197  * by Joseph Samosky's and Mark Haigh-Hutchinson's different
198  * polygon inclusion tests.
199  */
200  if ( ((vtx1[Y]-ty) * (vtx0[X]-vtx1[X]) >=
201  (vtx1[X]-tx) * (vtx0[Y]-vtx1[Y])) == yflag1 ) {
202  inside_flag = !inside_flag ;
203  }
204  }
205 
206  /* Move to the next pair of vertices, retaining info as possible. */
207  yflag0 = yflag1 ;
208  vtx0[0] = vtx1[0] ;
209  vtx0[1] = vtx1[1] ;
210  mark += 2;
211  vtx1[0] = mark[0];
212  vtx1[1] = mark[1];
213  }
214 
215  return( inside_flag ) ;
216 }
