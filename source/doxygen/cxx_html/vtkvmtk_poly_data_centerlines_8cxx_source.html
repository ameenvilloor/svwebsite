367 void vtkvmtkPolyDataCenterlines::FindVoronoiSeeds(vtkUnstructuredGrid *delaunay, vtkIdList *boundaryBaricenterIds, vtkDataArray *normals, vtkIdList *seedIds)
368 {
369  vtkIdType i, j;
370  vtkIdList *pointCells;
371  vtkIdType baricenterId;
372  double baricenter[3], normal[3];
373  double maxRadius, secondMaxRadius;
374  vtkTetra* tetra;
375  double p0[3], p1[3], p2[3], p3[3];
376  double circumcenter[3], circumradius, tetraRadius;
377  double referenceVector[3];
378  double pole[3], poleVector[3], secondPole[3], secondPoleVector[3];
379  pole[0] = pole[1] = pole[2] = 0.0;
380  vtkIdType maxRadiusCellId, secondMaxRadiusCellId;
381 
382  pointCells = vtkIdList::New();
383 
384  for (i=0; i<boundaryBaricenterIds->GetNumberOfIds(); i++)
385  {
386  baricenterId = boundaryBaricenterIds->GetId(i);
387  delaunay->GetPoint(baricenterId,baricenter);
388  normals->GetTuple(baricenterId,normal);
389  pointCells->Initialize();
390  delaunay->GetPointCells(baricenterId,pointCells);
391  maxRadius = 0.0;
392  maxRadiusCellId = -1;
393  secondMaxRadiusCellId = -1;
394 
395  for (j=0; j<pointCells->GetNumberOfIds(); j++)
396  {
397  tetra = vtkTetra::SafeDownCast(delaunay->GetCell(pointCells->GetId(j)));
398  tetra->GetPoints()->GetPoint(0,p0);
399  tetra->GetPoints()->GetPoint(1,p1);
400  tetra->GetPoints()->GetPoint(2,p2);
401  tetra->GetPoints()->GetPoint(3,p3);
402 
403  circumradius = vtkTetra::Circumsphere(p0,p1,p2,p3,circumcenter);
404  tetraRadius = sqrt(circumradius);
405 
406  if (tetraRadius - maxRadius > VTK_VMTK_DOUBLE_TOL)
407  {
408  maxRadius = tetraRadius;
409  maxRadiusCellId = pointCells->GetId(j);
410  pole[0] = circumcenter[0];
411  pole[1] = circumcenter[1];
412  pole[2] = circumcenter[2];
413  }
414  }
415 
416  poleVector[0] = pole[0] - baricenter[0];
417  poleVector[1] = pole[1] - baricenter[1];
418  poleVector[2] = pole[2] - baricenter[2];
419 
420  secondMaxRadius = 0.0;
421 
422  for (j=0; j<pointCells->GetNumberOfIds(); j++)
423  {
424  tetra = vtkTetra::SafeDownCast(delaunay->GetCell(pointCells->GetId(j)));
425  tetra->GetPoints()->GetPoint(0,p0);
426  tetra->GetPoints()->GetPoint(1,p1);
427  tetra->GetPoints()->GetPoint(2,p2);
428  tetra->GetPoints()->GetPoint(3,p3);
429 
430  circumradius = vtkTetra::Circumsphere(p0,p1,p2,p3,circumcenter);
431  tetraRadius = sqrt(circumradius);
432 
433  referenceVector[0] = circumcenter[0] - baricenter[0];
434  referenceVector[1] = circumcenter[1] - baricenter[1];
435  referenceVector[2] = circumcenter[2] - baricenter[2];
436 
437  if ((tetraRadius - secondMaxRadius > VTK_VMTK_DOUBLE_TOL) && (vtkMath::Dot(poleVector,referenceVector) < VTK_VMTK_DOUBLE_TOL))
438  {
439  secondMaxRadius = tetraRadius;
440  secondMaxRadiusCellId = pointCells->GetId(j);
441  secondPole[0] = circumcenter[0];
442  secondPole[1] = circumcenter[1];
443  secondPole[2] = circumcenter[2];
444  }
445  }
446 
447  secondPoleVector[0] = secondPole[0] - baricenter[0];
448  secondPoleVector[1] = secondPole[1] - baricenter[1];
449  secondPoleVector[2] = secondPole[2] - baricenter[2];
450 
451  if (vtkMath::Dot(poleVector,normal) < VTK_VMTK_DOUBLE_TOL)
452  {
453  seedIds->InsertNextId(maxRadiusCellId);
454  }
455  else
456  {
457  seedIds->InsertNextId(secondMaxRadiusCellId);
458  }
459  }
460 
461  pointCells->Delete();
462 }

484  endPointId1 = completeCenterlinesPoints->InsertNextPoint(endPointPairs->GetPoint(2*k));

486 
487  completeCell->Initialize();
488  completeCell->SetNumberOfIds(cell->GetNumberOfPoints()+2);
489 
490  completeCell->SetId(0,endPointId1);
491 
492  for (int i=0; i<cell->GetNumberOfPoints(); i++)
493  {
494  completeCell->SetId(i+1,cell->GetPointId(i));
495  }
496  completeCell->SetId(cell->GetNumberOfPoints()+1,endPointId2);
497 
498  completeCenterlinesCellArray->InsertNextCell(completeCell);
499 
500  completeCenterlinesRadiusArray->InsertNextValue(centerlinesRadiusArray->GetValue(cell->GetPointId(0)));
501  completeCenterlinesRadiusArray->InsertNextValue(centerlinesRadiusArray->GetValue(cell->GetPointId(cell->GetNumberOfPoints()-1)));
502  }
503 
504  completeCenterlines->SetPoints(completeCenterlinesPoints);
505  completeCenterlines->SetLines(completeCenterlinesCellArray);
506  completeCenterlines->GetPointData()->AddArray(completeCenterlinesRadiusArray);
507  completeCenterlines->BuildLinks();
508 
509  output->ShallowCopy(completeCenterlines);
510 
511  completeCell->Delete();
512  completeCenterlines->Delete();
513  completeCenterlinesPoints->Delete();
514  completeCenterlinesCellArray->Delete();
515  completeCenterlinesRadiusArray->Delete();
516 }

518 void vtkvmtkPolyDataCenterlines::ResampleCenterlines()
519 {
520  vtkPolyData* output = this->GetOutput();
521  vtkPolyData* resampledCenterlines = vtkPolyData::New();
522  vtkPoints* resampledCenterlinesPoints = vtkPoints::New();
523  vtkCellArray* resampledCenterlinesCellArray = vtkCellArray::New();
524  vtkDoubleArray* resampledCenterlinesRadiusArray = vtkDoubleArray::New();
525  resampledCenterlinesRadiusArray->SetName(this->RadiusArrayName);
526  vtkIdList* resampledCell = vtkIdList::New();
527 
528  vtkDoubleArray* centerlinesRadiusArray = vtkDoubleArray::SafeDownCast(output->GetPointData()->GetArray(this->RadiusArrayName));
529 
530  for (int k=0; k<output->GetNumberOfCells(); k++)
531  {
532  vtkCell* cell = output->GetCell(k);
533 
534  resampledCell->Initialize();
535 
536  vtkIdType id = resampledCenterlinesPoints->InsertNextPoint(cell->GetPoints()->GetPoint(0));
537  resampledCell->InsertNextId(id);
538  resampledCenterlinesRadiusArray->InsertNextValue(centerlinesRadiusArray->GetValue(cell->GetPointId(0)));
539 
540  double point0[3], point1[3], point[3];
541  double abscissa, lineAbscissa, lineLength, stepAbscissa;
542 
543  abscissa = 0.0;
544  lineAbscissa = 0.0;
545  lineLength = 0.0;
546  stepAbscissa = 0.0;
547 
548  for (int i=0; i<cell->GetNumberOfPoints()-1; i++)
549  {
550  cell->GetPoints()->GetPoint(i,point0);
551  cell->GetPoints()->GetPoint(i+1,point1);
552 
553  double scalar0 = centerlinesRadiusArray->GetValue(cell->GetPointId(i));
554  double scalar1 = centerlinesRadiusArray->GetValue(cell->GetPointId(i+1));
555 
556  double length = sqrt(vtkMath::Distance2BetweenPoints(point0,point1));
557 
558  if (length < this->ResamplingStepLength - stepAbscissa)
559  {
560  stepAbscissa = stepAbscissa + length;
561  continue;
562  }
563 
564  double pcoord = 0.0;
565  double pcoordStep = this->ResamplingStepLength / length;
566  while (pcoord < 1.0)
567  {
568  point[0] = point0[0] + (point1[0] - point0[0]) * pcoord;
569  point[1] = point0[1] + (point1[1] - point0[1]) * pcoord;
570  point[2] = point0[2] + (point1[2] - point0[2]) * pcoord;
571 
572  double scalar = scalar0 + (scalar1 - scalar0) * pcoord;
573 
574  vtkIdType id = resampledCenterlinesPoints->InsertNextPoint(point);
575  resampledCell->InsertNextId(id);
576  resampledCenterlinesRadiusArray->InsertNextValue(scalar);
577 
578  if (pcoord + pcoordStep > 1.0)
579  {
580  break;
581  }
