119 
120  for (int i=0; i<boundaries->GetNumberOfCells(); i++)
121  {
122  if (this->BoundaryIds)

125  {

127  }

136  }
137  newPolys->InsertNextCell(boundaryPointIds);
138  if (markCells)
139  {
140  cellEntityIdsArray->InsertNextValue(i+1+this->CellEntityIdOffset);
141  }
142  boundaryPointIds->Delete();
143  }
144

146  output->SetPolys(newPolys);
147

149 
150  if (markCells)
151  {
152  output->GetCellData()->AddArray(cellEntityIdsArray);
153  cellEntityIdsArray->Delete();
154  }
155 
156  newPoints->Delete();
157  newPolys->Delete();
158  boundaryExtractor->Delete();
159

161 }
