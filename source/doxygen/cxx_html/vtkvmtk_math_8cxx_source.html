167 {

175 
176  if (distance<GetTolerance(distance))
177  {

180  origin[2] = center1[2];
181  normal[0] = normal[1] = normal[2] = 0.0;

192  else

195  {

197  }
198  else
199  {
200  displacement = 0.5 * (displacementRatio + distance);
201  }
202  }
203
