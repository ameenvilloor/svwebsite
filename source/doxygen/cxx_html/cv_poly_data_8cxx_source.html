105 double cvPolyData::FindDistance2( double x, double y, double z )
106 {
107  vtkFloatingPointType pt[3];
108  vtkFloatingPointType closestPt[3];
109  vtkIdType closestPtId;
110  int subId;
111  vtkFloatingPointType dist2;
112  double result;
113 
114  if ( InitDistance() != CV_OK ) {
115  return -1.0;
116  }
117 
118  if ( (static_cast<vtkDataSet*>(data_))->GetNumberOfPoints() == 0 ) {
119  return -1.0;
120  }
121 
122  pt[0] = x;
123  pt[1] = y;
124  pt[2] = z;
125 
126  switch (distMethod_) {
127  case PD_DIST_VTK:
128  locator_->FindClosestPoint( pt, closestPt, genericCell_, closestPtId,
129  subId, dist2 );
130  break;
131 
132  case PD_DIST_INVALID:
133  dist2 = -1.0;
134  break;
135  }
136 
137  result = dist2;
138  return result;
139 }

171 // Find distance-squared to the poly data from the given point within
172 // radius.
173 
174 double cvPolyData::FindDistance2( double x, double y, double z, double radius )
175 {
176  int found = 0;
177  vtkFloatingPointType pt[3];
178  vtkFloatingPointType closestPt[3];
179  vtkIdType closestPtId;
180  int subId;
181  vtkFloatingPointType dist2;
182  double result;
183 
184  if ( InitDistance() != CV_OK ) {
185  return -1.0;
186  }
187 
188  pt[0] = x;
189  pt[1] = y;
190  pt[2] = z;
191 
192  switch (distMethod_) {
193  case PD_DIST_VTK:
194  found = locator_->FindClosestPointWithinRadius( pt, radius, closestPt,
195  genericCell_, closestPtId,
196  subId, dist2 );
197  break;
198 
199  case PD_DIST_INVALID:
200  return -1.0;
201  }
202 
203  if ( !found ) {
204  result = radius * radius;
205  } else {
206  result = dist2;
207  }
208 
209  return result;
210 }
