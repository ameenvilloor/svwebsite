109 void dp_cont_norm_comp( int n, double pts[], double norm[] )
110 {
111 
112  int i, j;
113 
114  vtkFloatingPointType nx, ny, nz;
115 
116  vtkFloatingPointType mag;
117 
118  /**************
119  *** body ***
120  **************/
121 
122  nx = ny = nz = 0.0;
123 
124  for (i = 0; i < n; i++) {
125  if (i == (n - 1)) {
126  j = 0;
127  }
128  else {
129  j = i + 1;
130  }
131 
132  nx += pts[3*j+1]*pts[3*i+2] - pts[3*j+2]*pts[3*i+1];
133  ny += pts[3*j+2]*pts[3*i] - pts[3*j]*pts[3*i+2];
134  nz += pts[3*j]*pts[3*i+1] - pts[3*j+1]*pts[3*i];
135  }
136 
137  mag = sqrt(nx*nx + ny*ny + nz*nz);
138 
139  if (mag != 0.0) {
140  norm[0] = -nx / mag;
141  norm[1] = -ny / mag;
142  norm[2] = -nz / mag;
143  }
144  }

236 void dp_mth_mat3x3_rot_extract( double mat[3][3], double rot[3] )
237 {
238  double xr, yr, zr;
239 
240  double arg;
241 
242  int i, j;
243 
244  double r11, r12, r13;
245 
246  double r21, r22, r23;
247 
248  double r31, r32, r33;
249 
250  double f;
251 
252  double pi;
253 
254  double sy, cy;
255 
256  double tmp;
257 
258  double check_mat[3][3];
259 
260  int y_def, x_def;
261 
262  static double tol = 1.0e-7;
263 
264  /**************
265  *** body ***
266  **************/
267 
268  r11 = mat[0][0];
269  r12 = mat[0][1];
270  r13 = mat[0][2];
271 
272  r21 = mat[1][0];
273  r22 = mat[1][1];
274  r23 = mat[1][2];
275 
276  r31 = mat[2][0];
277  r32 = mat[2][1];
278  r33 = mat[2][2];
279 
280  pi = 4.0 * atan(1.0);
281  xr = yr = zr = 0.0;
282  yr = asin (-r13);
283  cy = cos (yr);
284 
285  if (fabs(cy) < tol) {
286  zr = 0.0;
287  xr = acos (r22);
288  tmp = sin(xr) * sin(yr);
289 
290  if (dp_mth_sign(tmp) != dp_mth_sign(r33)) {
291  xr = -xr;
292  }
293  }
294  else {
295  arg = (r23 / cy);
296  if (arg > 1.0) arg = 1.0;
297  xr = asin (arg);
298  tmp = cos (xr) * cy;
299 
300  if (dp_mth_sign(tmp) != dp_mth_sign(r33)) {
301  xr += pi;
302  tmp = sin(xr) * cy;
303 
304  if (dp_mth_sign(tmp) != dp_mth_sign(r23)) {
305  xr = -xr;
306  }
307  }
308 
309  arg = (r12 / cy);
310  if (arg > 1.0) arg = 1.0;
311  zr = asin (arg);
312  tmp = cos(zr) * cy;
313 
314  if (dp_mth_sign(tmp) != dp_mth_sign(r11)) {
315  zr += pi;
316  tmp = sin(zr) * cy;
317 
318  if (dp_mth_sign(tmp) != dp_mth_sign(r12)) {
319  zr = -zr;
320  }
321  }
322  }
323 
324  xr = xr * 180.0 / pi;
325  yr = yr * 180.0 / pi;
326  zr = zr * 180.0 / pi;
327 
328  rot[0] = xr;
329  rot[1] = yr;
330  rot[2] = zr;
331 }

365 void dp_paras_xform (PK_BODY_t body, double *center, double *rot,
366  double *trans, double *scale, PK_BODY_t *p_body)
367 {
368  PK_VECTOR_t pos;
369 
370  PK_VECTOR1_t axis;
371 
372  PK_TRANSF_t xrot, yrot, zrot, transl, xform, uscale;
373 
374  int i;
375 
376  double f;
377 
378  double tol;
379 
380  PK_BODY_t new_body;
381 
382  PK_ERROR_code_t status;
383 
384  PK_LOGICAL_t *exact;
385 
386  double paras_trans[3];
387 
388  double paras_rot[3];
389 
390  double paras_center[3];
391 
392  double scale_x;
393 
394  double GMOD_MATH_DEG2RAD;
395 
396  int n;
397 
398  PK_GEOM_t *xform_geom;
399 
400  /**************
401  *** body ***
402  **************/
403 
404  /*
405  fprintf (stderr, "\n ---------- gmod_paras_xform ---------- \n");
406  */
407 
408  GMOD_MATH_DEG2RAD = 4.0 * atan (1.0) / 180.0;
409 
410 
411  /* create transformation. */
412 
413  for (i = 0; i < 3; i++) {
414  paras_rot[i] = (double)(GMOD_MATH_DEG2RAD * rot[i]);
415  paras_trans[i] = trans[i];
416  paras_center[i] = center[i];
417  }
418 
419  scale_x = scale[0];
420  tol = 0.0;
421 
422  PARAS_VEC_SET (pos, -paras_center[0], -paras_center[1], -paras_center[2]);
423  PK_TRANSF_create_translation (pos, &transl);
424  status = PK_BODY_transform (body, transl, tol, &n, &xform_geom, &exact);
425 
426  PARAS_VEC_SET (pos, 0.0, 0.0, 0.0);
427  PARAS_VEC_SET (axis, 1.0, 0.0, 0.0);
428  PK_TRANSF_create_rotation (pos, axis, paras_rot[0], &xrot);
429  status = PK_BODY_transform (body, xrot, tol, &n, &xform_geom, &exact);
430 
431  PARAS_VEC_SET (axis, 0.0, 1.0, 0.0);
432  PK_TRANSF_create_rotation (pos, axis, paras_rot[1], &yrot);
433  status = PK_BODY_transform (body, yrot, tol, &n, &xform_geom, &exact);
434 
435  PARAS_VEC_SET (axis, 0.0, 0.0, 1.0);
436  PK_TRANSF_create_rotation (pos, axis, paras_rot[2], &zrot);
437  status = PK_BODY_transform (body, zrot, tol, &n, &xform_geom, &exact);
438 
439  PK_TRANSF_create_equal_scale (scale_x, pos, &uscale);
440  status = PK_BODY_transform (body, uscale, tol, &n, &xform_geom, &exact);
441 
442  PARAS_VEC_SET (pos, paras_center[0], paras_center[1], paras_center[2]);
443  PK_TRANSF_create_translation (pos, &transl);
444  status = PK_BODY_transform (body, transl, tol, &n, &xform_geom, &exact);
445 
446  PARAS_VEC_SET (pos, paras_trans[0], paras_trans[1], paras_trans[2]);
447  PK_TRANSF_create_translation (pos, &transl);
448  status = PK_BODY_transform (body, transl, tol, &n, &xform_geom, &exact);
449 
450  *p_body = body;
451 }

1076 // ----------------
1077 // Special subtract notes:
1078 // - PsdUtils_DispatchBoolean calls PK_BODY_boolean_2
1079 // - PK_BODY_boolean_2 performs "target - tools"
1080 // - we want PsdUtils_Subtract to provide "a - b"
1081 // - we need to treat multi-body operands a bit more carefully in
1082 // subtract than for the other booleans (intersect, union)
1083 //
1084 // Here's our strategy. Foreach body in geomA, we will subtract all
1085 // of the bodies in geomB. This will give us a list of result
1086 // entities, which we will concatenate together (excluding null
1087 // entities) to give result.
1088 //
1089 // Also note that the semantics of the boolean PsdUtils_ functions is
1090 // that the boolean will DESTROY the input operands.
1091 // PsdUtils_DispatchBoolean will destroy its inputs. We need to be
1092 // aware of this as we may be calling PsdUtils_DispatchBoolean
1093 // repeatedly for each component of geomA to have all the components
1094 // of geomB subtracted. As a result, we need a deep copy of geomB for
1095 // each such call.
1096 
1097 int PsdUtils_Subtract( PsdListT *geomA, PsdListT *geomB,
1098  PsdSmpT smp, PsdListT **result )
1099 {

1101  PsdIterT *iter_j;
1102  PsdItem *entry;
1103  PsdListT *copyB;
1104  PsdListT *componentIn;
1105  PsdListT *componentOut;
1106  int code;
1107 
1108  *result = new PsdListT;
1109  while ( ! iter_i->IsDone() ) {
1110 
1111  // Make a copy of geomB whose entities can be surrendered in the
1112  // call to PsdUtils_DispatchBoolean:
1113  if ( PsdUtils_DeepCopy( geomB, &copyB ) != CV_OK ) {
1114  delete iter_i;
1115  PsdUtils_Delete( *result );
1116  PsdUtils_Delete( geomA );
1117  PsdUtils_Delete( geomB );
1118  (*result) = NULL;
1119  return CV_ERROR;
1120  }
1121 
1122  // Make a PsdListT which has one entity from geomA for use as the
1123  // "target" geom in PsdUtils_DispatchBoolean:
1124  entry = PsdUtils_NewItem();
1125  entry->item = iter_i->Item()->item;
1126  componentIn = new PsdListT;
1127  componentIn->Append( entry );
1128 
1129  // PK_BODY_boolean_2 semantics:
1130  // - tool bodies deleted
1131  // - target body modified / replaced by results
1132  code = PsdUtils_DispatchBoolean( componentIn, copyB, smp,
1133  PK_boolean_subtract_c, &componentOut );
1134 
1135  // Since it's not clear to me whether the target body actually now
1136  // refers to the boolean result or not, I will not invoke deep
1137  // deletion here on the target. The tools, once again, were
1138  // deleted in the boolean call.
1139  PsdUtils_DeleteListOnly( componentIn );
1140 
1141  if ( code != CV_OK ) {
1142  delete iter_i;
1143  PsdUtils_Delete( *result );
1144  PsdUtils_Delete( geomA );
1145  PsdUtils_Delete( geomB );
1146  PsdUtils_Delete( copyB );
1147  (*result) = NULL;
1148  return CV_ERROR;
1149  }
1150 
1151  // Current boolean call succeeded, so append each body created by
1152  // the boolean onto result:
1153  iter_j = new PsdIterT( componentOut );
1154  while ( ! iter_j->IsDone() ) {
1155  if ( iter_j->Item()->item != PK_ENTITY_null ) {
1156  entry = PsdUtils_NewItem();
1157  entry->item = iter_j->Item()->item;
1158  (*result)->Append( entry );
1159  iter_j->Next();
1160  }
1161  }
1162 
1163  // Here, we've only transferred handles from componentOut to
1164  // result, so let's do list-only deletion:
1165  PsdUtils_DeleteListOnly( componentOut );
1166  delete iter_j;
1167  iter_i->Next();
1168  }

1172  // the boolean target geoms, we may not want to delete those
1173  // entities since the target bodies may actually refer to the
1174  // results. (Fun, eh?) The geomB bodies, on the other hand, were
1175  // also passed as handles to the boolean, but these are destroyed by
1176  // the boolean.
1177 
1178  PsdUtils_DeleteListOnly( geomA );
1179  PsdUtils_DeleteListOnly( geomB );
1180 
1181  delete iter_i;
