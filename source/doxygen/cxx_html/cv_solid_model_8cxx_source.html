51 // cvSolidModel::DefaultInstantiateSolidModel
52 // ----------------------------
53 // This is the ONLY place that we should actually invoke construction
54 // of any concrete classes derived from cvSolidModel. In particular,
55 // all the object method handling functions here should just call this
56 // function and use the returned cvSolidModel*. Note that this function
57 // also takes in the interpreter so that an informative error message
58 // about kernel types can be returned if necessary.
59 
60 // PLEASE note that checking the return value of this function is
61 // CRITICAL for things to work right. If we return NULL here, we
62 // absolutely need to trap that in the callers (i.e. cvSolidModel
63 // instantiation functions) to make sure those NULL ptr's don't get
64 // registered in the repository. Subsequent cvSolidModel lookup's
65 // currently DO NOT check for NULL values. The idea is that objects
66 // are checked for validity *before* they get registered.
67 
68 cvSolidModel* cvSolidModel::DefaultInstantiateSolidModel( Tcl_Interp *interp )
69 {
70 #ifdef EXCLUDE_SOLID_MODEL
71  printf( "ERR: attempted to create cvSolidModel w/ no available kernel\n" );
72  return NULL;
73 
74 #else
75 
76  // Get the solid model factory registrar associated with this Tcl interpreter.
77  cvFactoryRegistrar* solidModelRegistrar;
78  if (interp == NULL) {
79  fprintf(stdout,"WARNING: Null interpreter passed to SolidModel. Overriding with default.\n");
80  fflush(stdout);
81  }
82  Tcl_Interp* myinterp = NULL;
83  myinterp = gVtkTclInterp;
84  assert(myinterp);
85 
86  solidModelRegistrar = (cvFactoryRegistrar *) Tcl_GetAssocData( myinterp, "SolidModelRegistrar", NULL);
87 
88  cvSolidModel* solid = NULL;
89  if (cvSolidModel::gCurrentKernel == SM_KT_PARASOLID ||
90  cvSolidModel::gCurrentKernel == SM_KT_DISCRETE ||
91  cvSolidModel::gCurrentKernel == SM_KT_POLYDATA) {
92 
93  solid = (cvSolidModel *) (solidModelRegistrar->UseFactoryMethod( cvSolidModel::gCurrentKernel ));
94  if (solid == NULL) {
95  fprintf( stdout, "Unable to create solid model kernal (%i)\n",cvSolidModel::gCurrentKernel);
96  //Tcl_SetResult( interp, "Unable to create solid model", TCL_STATIC );
97  }
98  } else {
99  fprintf( stdout, "current kernel is not valid (%i)\n",cvSolidModel::gCurrentKernel);
100  //Tcl_SetResult( interp, "current kernel is not valid", TCL_STATIC );
101  }
102  return solid;
103 
104 #endif
105 }
