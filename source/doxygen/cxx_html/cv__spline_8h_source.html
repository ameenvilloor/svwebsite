94 /***********************************************************************
95  * *
96  * sys_geom_SplinePath *
97  * ------------------- *
98  * Performs spline interpolation and computes tangents and rotation *
99  * vectors. *
100  * *
101  * INPUTS *
102  * ------ *
103  * SplinePoints *input: set of input points for interpolation, limited*
104  * to MAX_DIM dimensions (vtk limitation). *
105  * input->points are in the format x1, y1, x2, *
106  * y2, x3, y3, .... *
107  * int type: specifies basis used in spline interpolation *
108  * type = 1 --> cardinal basis *
109  * type = 2 --> Kochanek basis *
110  * see vtk for more details *
111  * int numOutputPts: determines number of points to interpolate *
112  * *
113  * OUTPUTS *
114  * ------- *
115  * SplinePoints *output: interpolated points, in the format x1, y1, *
116  * x2, y2, x3, y3, .... *
117  * tangents are normalized vectors in the format*
118  * x1, y1, x2, y2, x3, y3, .... *
119  * rotVectors are normalized vectors in the *
120  * format x1, y1, x2, y2, x3, y3,... *
121  * The number of coordinates for each point and *
122  * each vector is dependent on the dimensions *
123  * of the input. *
124  * *
125  * EXAMPLE *
126  * ------- *
127  * Given 5 points in variable x, you want to use the cardinal *
128  * basis for spline interpolation to get out 25 points in variable y. *
129  * *
130  * sys_geom_SplinePath (x, 1, 25, y) *
131  * *
132  ***********************************************************************/
133 
134 int sys_geom_SplinePath (SplinePoints *input, int type, int numOutputPts, int matchEndPoints, SplinePoints *output);
