261 int cvRepositoryData::SetLabel( char *key, char *value )
262 {
263  Tcl_HashEntry *entryPtr;
264  Tcl_DString *dsPtr;
265  int newFlag;
266 
267  entryPtr = Tcl_CreateHashEntry( &labels_, key, &newFlag );
268  if ( ! newFlag ) {
269  return 0;
270  }
271 
272  dsPtr = new Tcl_DString;
273  Tcl_DStringInit( dsPtr );
274  Tcl_SetHashValue( entryPtr, dsPtr );
275  Tcl_DStringAppend( dsPtr, value, -1 );
276  return 1;
277 }
