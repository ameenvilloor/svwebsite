61 char *cvMeshObject::GetKernelName( cvMeshObject::KernelType kernel )
62 {
63  char *result;
64 
65  result = new char[100];
66  result[0]='\0';
67  switch (kernel) {
68  case KERNEL_MESHSIM:
69  strcpy( result, "MeshSim" );
70  break;
71  case KERNEL_GMSH:
72  strcpy( result, "GMsh" );
73  break;
74  case KERNEL_TETGEN:
75  strcpy( result, "TetGen" );
76  break;
77  default:
78  strcpy( result, "Invalid kernel name; must be one of "
79  "{ MeshSim, GMsh,TetGen }" );
80  return NULL;
81  break;
82  }
83 
84  return result;
85 }

87 cvMeshObject::KernelType cvMeshObject::GetKernelType( const char* kernel_name )
88 {
89  if (strcmp(kernel_name, "MeshSim") == 0)
90  return cvMeshObject::KERNEL_MESHSIM;

92  return cvMeshObject::KERNEL_GMSH;

94  return cvMeshObject::KERNEL_TETGEN;
95 
96  return cvMeshObject::KERNEL_INVALID;

100 int cvMeshObject::openOutputFile(char* filename) {
101  fp_ = NULL;
102  // open the output file
103  #ifdef USE_ZLIB
104  char filenamegz[MAXPATHLEN];
105  filenamegz[0]='\0';
106  sprintf (filenamegz, "%s.gz", filename);
107  fp_ = gzopen (filenamegz, "wb");
108  if (fp_ == NULL) {
109  fprintf(stderr,"Error: Could not open output file %s.\n",filenamegz);
110  return CV_ERROR;
111  }
112  #else
113  fp_ = gzopen (filename, "wb");
114  if (fp_ == NULL) {
115  fprintf(stderr,"Error: Could not open output file %s.\n",filename);
116  return CV_ERROR;
117  }
118  #endif
119  return CV_OK;
120 }

445 
446  return closeOutputFile();

457 int cvMeshObject::WriteMeshVrml (char *filename) {
458 
459  // open output file
460  if (openOutputFile(filename) != CV_OK) return CV_ERROR;
461 
462  // Calculate bounding box for mesh and scale to unit cube
463  double maxpoint = 0;
464  initNodeTraversal();
465  while (getNextNode() == 1) {
466  if (fabs(nodeX_) > maxpoint) {
467  maxpoint = fabs(nodeX_);
468  } else if (fabs(nodeY_) > maxpoint) {
469  maxpoint = fabs(nodeY_);
470  } else if (fabs(nodeZ_) > maxpoint) {
471  maxpoint = fabs(nodeZ_);
472  }
473  }
474 
475  // output the header
476  gzprintf(fp_,"#VRML V2.0 utf8\n");
477 
478  double invertMax = 1.0/maxpoint;
479 
480  // loop over all of the material regions, making each a separate object
481  initRegionTraversal();
482  int i = 0;
483  while (getNextRegion() == 1) {
484  initElementTraversal();
485  double colorme = 1.0*i/numModelRegions_;
486  gzprintf(fp_,"DEF region_%i Transform {\n",curMdlRegID_);
487  gzprintf(fp_," center 0 0 0\n");
488  gzprintf(fp_," scale %f %f %f\n",invertMax,invertMax,invertMax);
489  gzprintf(fp_," children [\n");
490  gzprintf(fp_," Shape {\n");
491  gzprintf(fp_," appearance Appearance {\n");
492  gzprintf(fp_," material DEF region_%iMaterial Material {\n",curMdlRegID_);
493  gzprintf(fp_," emissiveColor %f %f %f\n",colorme,colorme,1.0);
494  gzprintf(fp_," shininess 0.8\n");
495  gzprintf(fp_," }\n");
496  gzprintf(fp_," }\n");
497 
498  // since in general we don't have the ability to display wireframes
499  // (e.g. in CosmoPlayer), we will output faces as line segments instead
500  gzprintf(fp_," geometry IndexedLineSet {\n");
501 
502  if (i == 0) {
503  gzprintf(fp_," coord DEF all_nodes Coordinate {\n");
504  gzprintf(fp_," point [\n");
505  initNodeTraversal();
506  while(getNextNode() == 1) {
507  gzprintf(fp_," %f %f %f, \n", nodeX_, nodeY_, nodeZ_);
508  }
509  gzprintf(fp_," ]\n");
510  gzprintf(fp_," } \n");
511  } else {
512  gzprintf(fp_," coord USE all_nodes\n");
513  }
514  gzprintf(fp_," coordIndex [\n");
515  while (getNextElement() == 1) {
516  gzprintf(fp_," %d, %d, %d, %d, -1,\n",connID_[0]-1,connID_[2]-1,connID_[1]-1,connID_[0]-1);
517  gzprintf(fp_," %d, %d, %d, %d, -1,\n",connID_[0]-1,connID_[1]-1,connID_[3]-1,connID_[0]-1);
518  gzprintf(fp_," %d, %d, %d, %d, -1,\n",connID_[1]-1,connID_[2]-1,connID_[3]-1,connID_[1]-1);
519  gzprintf(fp_," %d, %d, %d, %d, -1,\n",connID_[0]-1,connID_[3]-1,connID_[2]-1,connID_[0]-1);
520  }
521  gzprintf(fp_," -1,\n");
522 
523  gzprintf(fp_," ]\n");
524  gzprintf(fp_," }\n");
525  gzprintf(fp_," }\n");
526  gzprintf(fp_," ]\n");
527  gzprintf(fp_,"}\n");
528 
529  i++;
530  }
531 
532  return closeOutputFile();
533 
534 }
