105 double cvCPUTimer::getGranularity()
106 {
107  if( !granularityKnown_ ) {
108  cvTimeInBaseTimer t1;
109  cvTimeInBaseTimer t2;
110 
111  t1 = currentTime();
112  do {
113  t2 = currentTime();
114  } while( t1 == t2 );
115 
116  granularity_ = t2 - t1;
117  granularityKnown_ = 1;
118  }
119  return granularity_;
120 }
