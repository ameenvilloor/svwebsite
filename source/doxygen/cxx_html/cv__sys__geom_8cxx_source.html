277 int sys_geom_Pick( cvPolyData *src, double pos[], cvPolyData **dst )
278 {
279  cvPolyData *merged_pd = NULL;
280  vtkPolyData *pd = NULL;
281  vtkPolyData *tmp = NULL;
282  cvPolyData *tmpPd = NULL;
283  int numPts;
284  double *pts = NULL;
285  vtkIdType *lines = NULL;
286  int numLines;
287  int *startIxs = NULL;
288  int numRegions;
289  int *regionLines = NULL;
290  int numRegionLines;
291  int i, classification;
292  int status = CV_ERROR;
293  cvSolidModel *solid = NULL;
294  int foundOne = 0;
295 
296  solid = cvSolidModel::DefaultInstantiateSolidModel();
297  if ( solid == NULL ) {
298  printf("ERR: default instantiate solid model failed\n");
299  return CV_ERROR;
300  }
301 
302  merged_pd = sys_geom_MergePts( src );
303  if ( merged_pd == NULL ) {
304  printf("ERR: merge points failed failed\n");
305  return CV_ERROR;
306  }
307  pd = merged_pd->GetVtkPolyData();
308 
309  if ( VtkUtils_GetPoints( pd, &pts, &numPts ) != CV_OK ) {
310  printf("ERR: VtkUtils_GetPoints failed\n");
311  return CV_ERROR;
312  }
313 
314  if ( VtkUtils_GetLines( pd, &lines, &numLines ) != CV_OK ) {
315  printf("ERR: VtkUtils_GetLines failed\n");
316  delete [] pts;
317  return CV_ERROR;
318  }
319 
320  if ( VtkUtils_FindClosedLineRegions( lines, numLines, numPts,
321  &startIxs, &numRegions ) != CV_OK ) {
322  printf("ERR: VtkUtils_FindClosedLineRegions failed\n");
323  delete [] pts;
324  delete [] lines;
325  return CV_ERROR;
326  }
327 
328  // Now, foreach closed region, get an ordered list of segments,
329  // create the corresponding solid, and check for point
330  // classification of the given pos.
331 
332  printf(" ------ sys_geom_Pick ------\n");
333  printf(" >>>>>> num closed regions [%d]\n", numRegions);
334 
335  for (i = 0; i < numRegions; i++) {
336 
337  if ( VtkUtils_GetClosedLineRegion( lines, numLines, startIxs[i],
338  &regionLines, &numRegionLines )
339  != CV_OK ) {
340  printf("ERR: VtkUtils_GetClosedLineRegion failed\n");
341  delete [] pts;
342  delete [] lines;
343  delete [] startIxs;
344  return CV_ERROR;
345  }
346 
347  if ( tmp != NULL ) {
348  tmp->Delete();
349  tmp = NULL;
350  }
351 
352  if ( VtkUtils_MakePolyDataFromLineIds( pts, numPts, lines, regionLines,
353  numRegionLines, &tmp ) != CV_OK ) {
354  printf("ERR: VtkUtils_MakePolyDataFromLineIds failed\n");
355  delete [] pts;
356  delete [] lines;
357  delete [] startIxs;
358  delete [] regionLines;
359  return CV_ERROR;
360  } else {
361  solid->Clear();
362  tmpPd = new cvPolyData( tmp );
363  if ( solid->MakePoly2d( tmpPd ) == CV_OK ) {
364  if ( solid->ClassifyPt( pos[0], pos[1], 0, &classification )
365  == CV_OK ) {
366  if ( classification >= 0 ) {
367  *dst = tmpPd;
368  status = CV_OK;
369  printf(" >>>>>> picked region lines [%d]\n", numRegionLines);
370  foundOne = 1;
371  break;
372  }
373  } else {
374  printf("ERR: cvSolidModel::ClassifyPt failed\n");
375  }
376  } else {
377  printf("ERR: cvSolidModel::MakePoly2d failed\n");
378  }
379  delete tmpPd;
380  }
381  }
382 
383  if ( ! foundOne ) {
384  printf(" >>>>>> no closed region selected\n");
385  }
386 
387  delete [] pts;
388  delete [] lines;
389  delete [] startIxs;
390  // be careful here since these objects will not exist if numRegions == 0
391  if (numRegions != NULL) delete [] regionLines;
392  if ( tmp != NULL ) tmp->Delete();
393  return status;
394 }
