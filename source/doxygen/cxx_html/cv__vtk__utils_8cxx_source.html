162 // -----------
163 // UpdateCells
164 // -----------
165 // This is a helper for FixTopology. We take in a vtk-style cell list
166 // (i.e. each cell descriptor is of size n+1, if there are n points in
167 // the cell, and the descriptor is of the format: {numIds, id0, id1,
168 // ...}). We generate on output an "updated" cell list which reflects
169 // any duplicate point id's (as generated by the point-merging
170 // process) which lead to degenerate cells (i.e. line to point,
171 // triangle to line).
172 //
173 // NOTE that only lines and triangles are currently handled!
174 //
175 // The pointer returned as *newCells needs to be free'd by the caller.
176 
177 static int UpdateCells( int *cells, int numCells,
178  vtkIdType **newCells, int *numNewCells )
179 {
180  vtkIdType *tmpNewCells;
181  int numTmpNewCells;
182  int cellIter, newCellIter;
183  int i, j, k;
184  int degen;
185 
186  // Check for newly-created duplications in point ID lists. If
187  // such a duplication is found, then this is a degenerate cell, so
188  // exclude it from the newCells list. Otherwise, enter this cell
189  // into the newCells list.
190 
191  tmpNewCells = new vtkIdType [numCells*4];
192  numTmpNewCells = 0;
193  cellIter = newCellIter = 0;
194 
195  // For each cell:
196  for (i = 0; i < numCells; i++) {
197  degen = 0;
198 
199  // For each point ID in this cell:
200  for (j = 0; j < cells[cellIter]; j++) {
201 
202  // Compare to every other point ID in this same cell, avoiding
203  // self comparisions and redundant comparisons:
204  for (k = 0; k < cells[cellIter]; k++) {
205  if (j == k) continue;
206 
207  // If the two indices being compared are the same, then this
208  // is a degenerate cell (e.g. a line has collapsed to a
209  // point, triangle to a line). Note that this does NOT hold
210  // for other cell types (e.g a four-noded polygon which has
211  // two coicident points is still a polygon). For now, we are
212  // NOT handling such cases.
213  if ( cells[ cellIter + j + 1 ] == cells[ cellIter + k + 1 ] ) {
214  degen = 1;
215  break;
216  }
217  }
218  if (degen) break;
219  }
220 
221  // If no points in the current cell were merged, then copy the
222  // cell to the newCells list:
223  if (!degen) {
224  tmpNewCells[newCellIter] = cells[cellIter];
225  for (j = 0; j < tmpNewCells[newCellIter]; j++) {
226  tmpNewCells[ newCellIter + j + 1 ] = cells[ cellIter + j + 1 ];
227  }
228  newCellIter = newCellIter + tmpNewCells[newCellIter] + 1;
229  numTmpNewCells++;
230  } else {
231  /*
232  printf("Cell[%d] degenerate: (%d,%d)\n", i,
233  cells[cellIter+1], cells[cellIter+2]);
234  */
235  }
236 
237  // Advance the "iterator" to the next cell position:
238  cellIter = cellIter + cells[cellIter] + 1;
239  }
240 
241  *newCells = tmpNewCells;
242  *numNewCells = numTmpNewCells;
243  return CV_OK;
244 }

247 // --------------------
248 // VtkUtils_FixTopology
249 // --------------------
250 // The basic strategy we will take has two main steps:
251 // 1. Rebuild point list.
252 // 2. Rebuild connectivities.
253 
254 // NOTE that the only elements we fix are line segments and triangles.
255 
256 int VtkUtils_FixTopology( vtkPolyData *pd, double tol )
257 {
258  PrePt_T *prePtList;
259  PostPt_T *postPtList;
260  int preNumPts, preNumElems, preNumLines, preNumTris;
261  int postNumPts = 0, postNumElems = 0;
262  vtkFloatingPointType pt[3];
263  int found;
264  vtkFloatingPointType curr[3];
265  vtkCellArray *lines, *tris, *newLines, *newTris;
266  int numCells, *cells, cellIter;
267  int numNewCells, newCellIter;
268  vtkIdType *newCells;
269  vtkIdType npts;
270  vtkIdType *pts;
271  int oldId, mappedId;
272  vtkPoints *compressedPts;
273 
274  if ( pd == NULL ) return CV_ERROR;
275 
276  preNumPts = pd->GetNumberOfPoints();
277  preNumElems = pd->GetNumberOfLines() + pd->GetNumberOfPolys();
278 
279  prePtList = new PrePt_T [preNumPts];
280  postPtList = new PostPt_T [preNumPts];
281 
282  // First, we want to iterate over all points to clean up "close"
283  // points and to build the post-processed pt list in this process.
284  int j;
285  for (int i = 0; i < preNumPts; i++) {
286  found = 0;
287  pd->GetPoint( i, pt );
288  for (j = 0; j < postNumPts; j++) {
289  curr[0] = postPtList[j].x;
290  curr[1] = postPtList[j].y;
291  curr[2] = postPtList[j].z;
292  if ( ( fabs(pt[0]-curr[0]) < tol ) &&
293  ( fabs(pt[1]-curr[1]) < tol ) &&
294  ( fabs(pt[2]-curr[2]) < tol ) ) {
295  found = 1;
296  /*
297  printf("Merging pt[%d] and pt[%d].\n",
298  i, postPtList[j].uncompressedId);
299  */
300  break;
301  }
302  }
303 
304  // Make a new entry in the non-compressed list:
305  prePtList[i].x = pt[0];
306  prePtList[i].y = pt[1];
307  prePtList[i].z = pt[2];
308 
309  // If this was a distinct point, add it to the compressed list:
310  if (!found) {
311  postPtList[postNumPts].x = pt[0];
312  postPtList[postNumPts].y = pt[1];
313  postPtList[postNumPts].z = pt[2];
314  postPtList[postNumPts].uncompressedId = i;
315 
316  // Make the non-compressed entry point to this compressed
317  // entry. We could set postId to postNumPts, in which case the
318  // new index is in the compressed list. Or, we could simply set
319  // postId to i, which is the uncompressed index. The latter is
320  // what we actually want, since we are going to simply put a new
321  // vtkCellArray into pd (for Lines and/or Polys) which we want
322  // to reference the ORIGINAL point list.
323 
324  prePtList[i].postId = postNumPts; // use compressed pt list
325 
326  //prePtList[i].postId = i; // use orig pt list
327 
328  postNumPts++;
329 
330  } else {
331  // Make the non-compressed entry point to the point in the
332  // compressed list that matched:
333 
334  prePtList[i].postId = j; // use compressed pt list
335 
336  //prePtList[i].postId = postPtList[j].uncompressedId; // use orig pt list
337  }
338  }
339 
340  // Build compressed point list:
341  compressedPts = vtkPoints::New();
342  for (int i = 0; i < postNumPts; i++) {
343  compressedPts->InsertNextPoint( (vtkFloatingPointType)postPtList[i].x,
344  (vtkFloatingPointType)postPtList[i].y,
345  (vtkFloatingPointType)postPtList[i].z );
346  }
347  pd->SetPoints( compressedPts );
348  compressedPts->Delete();
349 
350  // Now we have a compressed point list and a mapping of old id's to
351  // new ones (via the postId fields in prePtList). What we want to
352  // do now is propagate changes in the id's to the element
353  // connectivity descriptions.
354 
355  // Use vtkPolyData::SetLines and SetPolys. And perhaps
356  // vtkCellArray::InsertNextCell.
357 
358  // First make a working copy of the cell list:
359  // cells = new int [preNumElems*4];
360  cells = new int [pd->GetNumberOfLines() * 3];
361  lines = pd->GetLines();
362  preNumLines = lines->GetNumberOfCells();
363  lines->InitTraversal();
364  numCells = 0;
365  while ( lines->GetNextCell( npts, pts ) ) {
366  assert( npts == 2 );
367  cells[numCells*3] = npts;
368  cells[numCells*3+1] = pts[0];
369  cells[numCells*3+2] = pts[1];
370  numCells++;
371  }
372 
373  // Now update point ID's used in our working copy. Note that this
374  // update is done IN-PLACE in the "cells" list. New connectivities
375  // are written over the old ones in this working copy.
376  cellIter = 0;
377  for (int i = 0; i < numCells; i++) {
378  for (j = 0; j < cells[cellIter]; j++) {
379  oldId = cells[ cellIter + j + 1 ];
380  mappedId = prePtList[oldId].postId;
381  cells[ cellIter + j + 1 ] = mappedId;
382  }
383  // Advance to next cell description (i.e. size followed by list of
384  // point ID's):
385  cellIter = cellIter + cells[cellIter] + 1;
386  }
387 
388  UpdateCells( cells, numCells, &newCells, &numNewCells );
389 
390  newLines = vtkCellArray::New();
391 
392  // We want the indices used in newLines to actually point into the
393  // old (uncompressed) point list.
394  newCellIter = 0;
395  newLines->InitTraversal();
396  for (int i = 0; i < numNewCells; i++) {
397  newLines->InsertNextCell( newCells[newCellIter],
398  &(newCells[newCellIter+1]) );
399  newCellIter = newCellIter + newCells[newCellIter] + 1;
400  }
401  pd->SetLines( newLines );
402  newLines->Delete();
403 
404  delete [] cells;
405  delete [] newCells;
406 
407  // Now handle triangles:
408  cells = new int [pd->GetNumberOfPolys() * 4];
409  tris = pd->GetPolys();
410  preNumTris = tris->GetNumberOfCells();
411  tris->InitTraversal();
412  numCells = 0;
413  while ( tris->GetNextCell( npts, pts ) ) {
414  assert( npts == 3 );
415  cells[numCells*4] = npts;
416  cells[numCells*4+1] = pts[0];
417  cells[numCells*4+2] = pts[1];
418  cells[numCells*4+3] = pts[2];
419  numCells++;
420  }
421 
422  // Now update point ID's used in our working copy. Note that this
423  // update is done IN-PLACE in the "cells" list. New connectivities
424  // are written over the old ones in this working copy.
425  cellIter = 0;
426  for (int i = 0; i < numCells; i++) {
427  for (j = 0; j < cells[cellIter]; j++) {
428  oldId = cells[ cellIter + j + 1 ];
429  mappedId = prePtList[oldId].postId;
430  cells[ cellIter + j + 1 ] = mappedId;
431  }
432  // Advance to next cell description (i.e. size followed by list of
433  // point ID's):
434  cellIter = cellIter + cells[cellIter] + 1;
435  }
436 
437  UpdateCells( cells, numCells, &newCells, &numNewCells );
438 
439  newTris = vtkCellArray::New();
440 
441  // We want the indices used in newTris to actually point into the
442  // old (uncompressed) point list.
443  newCellIter = 0;
444  newTris->InitTraversal();
445  for (int i = 0; i < numNewCells; i++) {
446  newTris->InsertNextCell( newCells[newCellIter],
447  &(newCells[newCellIter+1]) );
448  newCellIter = newCellIter + newCells[newCellIter] + 1;
449  }
450  pd->SetPolys( newTris );
451  newTris->Delete();
452 
453 
454  printf(" \n\n------ VtkUtils_FixTopology ------\n");
455  printf(" >>>>>> orig. num pts [%d]\n", preNumPts);
456  printf(" >>>>>> new num pts [%d]\n", pd->GetNumberOfPoints());
457  printf(" >>>>>> orig. num lines [%d]\n", preNumLines);
458  printf(" >>>>>> new num lines [%d]\n", pd->GetNumberOfLines());
459  printf(" >>>>>> orig. num tris [%d]\n", preNumTris);
460  printf(" >>>>>> new num tris [%d]\n", pd->GetNumberOfPolys());
461  printf("\n");
462 
463  delete [] cells;
464  delete [] newCells;
465 
466  delete [] prePtList;
467  delete [] postPtList;
468 
469  return CV_OK;
470 }
