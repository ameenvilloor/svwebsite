55  subroutine drvlesprepdiag ( flowDiag, ilwork,
56  & ibc, bc, iper,
57  & rowp, colm,
58  & lhsk, lhsp)
59 c
60  use pointer_data
61  use pvsqbi
62  use convolimpflow !brings in the current part of convol coef for imp BC
63  use convolrcrflow !brings in the current part of convol coef for RCR BC
64  use lagrangemultipliers
65 
66  include "global.h"
67  include "mpif.h"
68  include "common_blocks/conpar.h"
69  include "common_blocks/fronts.h"
70  include "common_blocks/genpar.h"
71  include "common_blocks/inpdat.h"
72  include "common_blocks/nomodule.h"
73  include "common_blocks/timdat.h"
74  include "common_blocks/workfc.h"
75 C
76 C Argument variables
77 C
78  INTEGER ibc, ilwork, iper
79 C
80  REAL*8 bc, flowdiag
81 C
82 C Local variables
83 C
84  INTEGER i, irankcoupled, j
85 C
86  REAL*8 tfact
87 c
88  dimension flowdiag(nshg,4), ilwork(nlwork)
89  dimension ibc(nshg), iper(nshg), bc(nshg,ndofbc)
90  real*8 lhsk(9,nnz_tot), lhsp(4,nnz_tot)
91  integer rowp(nnz_tot), colm(nshg+1)
92  integer n, k
93 c
94  integer sparseloc
95 c
96 c
97 c.... Clear the flowdiag
98 c
99  if((flmpl.eq.1).or.(ipord.gt.1)) then
100  do n = 1, nshg
101  k = sparseloc( rowp(colm(n)), colm(n+1)-colm(n), n )
102  & + colm(n)-1
103 c
104  flowdiag(n,1) = lhsk(1,k)
105  flowdiag(n,2) = lhsk(5,k)
106  flowdiag(n,3) = lhsk(9,k)
107 c
108  flowdiag(n,4) = lhsp(4,k)
109  enddo
110  else
111  flowdiag = zero
112  do n = 1, nshg ! rowsum put on the diagonal instead of diag entry
113  do k=colm(n),colm(n+1)-1
114 
115 c
116  flowdiag(n,1) = flowdiag(n,1) + abs(lhsk(1,k))
117 c & + lhsK(2,k) + lhsK(3,k)
118  flowdiag(n,2) = flowdiag(n,2) + abs(lhsk(5,k))
119 c & + lhsK(4,k) + lhsK(6,k)
120  flowdiag(n,3) = flowdiag(n,3) + abs(lhsk(9,k))
121 c & + lhsK(7,k) + lhsK(8,k)
122 c
123  flowdiag(n,4) = flowdiag(n,4) + abs(lhsp(4,k))
124  enddo
125  flowdiag(n,:)=flowdiag(n,:)*pt33
126  enddo
127  endif
128  if(ipvsq.ge.3) then ! for first cut only do diagonal extraction
129  ! this is not yet correct for multi procs I suspect if partition
130  ! boundary cuts a p=QR face
131  tfact=alfi * gami * delt(1)
132  do n=1,nshg
133  if(numresistsrfs.gt.zero) then
134  do k = 1,numresistsrfs
135  if (nsrflistresist(k).eq.ndsurf(n)) then
136  irankcoupled=k
137  flowdiag(n,1:3) = flowdiag(n,1:3)
138  & + tfact*valuelistresist(irankcoupled)*
139  & nabi(n,:)*nabi(n,:)
140  endif
141  enddo
142  elseif(numimpsrfs.gt.zero) then
143  do k = 1,numimpsrfs
144  if (nsrflistimp(k).eq.ndsurf(n)) then
145  irankcoupled=k
146  flowdiag(n,1:3) = flowdiag(n,1:3)
147  & + tfact*impconvcoef(ntimeptpt+2,irankcoupled)*
148  & nabi(n,:)*nabi(n,:)
149  endif
150  enddo
151  elseif(numrcrsrfs.gt.zero) then
152  do k = 1,numrcrsrfs
153  if (nsrflistrcr(k).eq.ndsurf(n)) then
154  irankcoupled=k
155  flowdiag(n,1:3) = flowdiag(n,1:3)
156  & + tfact*rcrconvcoef(lstep+2,irankcoupled)* !check lstep+2 if restart from t.ne.0
157  & nabi(n,:)*nabi(n,:)
158  endif
159  enddo
160  endif
161  enddo
162  endif
163 c
164 c.... Now I am adding contributions from the Lagrange multipliers to preconditioning
165 c
166 c
167  if(lagrange .gt. zero) then
168  tfact=alfi * gami * delt(1)
169  call lagadddiag(flowdiag, tfact)
170  endif
171 
172 c
173  if(iabc==1) !are there any axisym bc's
174  & call rotabc(flowdiag, ibc, 'in ')
175 c
176 
177 c
178 c.... communicate : add the slaves part to the master's part of flowDiag
179 c
180  if (numpe > 1) then
181  call commu(flowdiag, ilwork, nflow, 'in ')
182  endif
183 c
184 c.... satisfy the boundary conditions on the diagonal
185 c
186  call bc3diag(ibc, bc, flowdiag)
187 c
188 c
189 c.... on processor periodicity was not taken care of in the setting of the
190 c boundary conditions on the matrix. Take care of it now.
191 c
192  call bc3per(ibc, flowdiag, iper, ilwork, 4)
193 c
194 c... slaves and masters have the correct values
195 c
196 c
197 c.... Calculate square root
198 c
199  do i = 1, nshg
200  do j = 1, nflow
201  if (flowdiag(i,j).ne.0)
202  & flowdiag(i,j) = 1. / sqrt(abs(flowdiag(i,j)))
203  enddo
204  enddo
205 c
206  return
207  end
