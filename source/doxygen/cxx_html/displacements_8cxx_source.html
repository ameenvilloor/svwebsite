67 void Cross( double ax, double ay, double az,
68  double bx, double by, double bz,
69  double *prodx, double *prody, double *prodz )
70 {
71  (*prodx) = ay * bz - az * by;

520 void siftDownKentries(Kentry Kentries[], int root, int bottom, int array_size)
521 {
522  int done, maxChild;
523  int row, col;
524  double value;
525 
526  done = 0;
527  while ((root*2 <= bottom) && (!done))
528  {
529  if (root*2 == bottom)
530  maxChild = root * 2;
531  else if (Kentries[root * 2].row > Kentries[root * 2 + 1].row)
532  maxChild = root * 2;
533  else
534  maxChild = root * 2 + 1;
535 
536  // hack since this seems to constantly end
537  if (root < 0 || root >= array_size) break;
538  if (maxChild < 0 || maxChild >= array_size) {
539  fprintf(stdout,"warning: illegal maxChild (%i)\n",maxChild);
540  fprintf(stdout,"root= %d bottom=%d array_size=%d\n",maxChild,root,bottom,array_size);
541  break;
542  }
543 
544  if (Kentries[root].row < Kentries[maxChild].row)
545  {
546  row = Kentries[root].row;
547  col = Kentries[root].col;
548  value = Kentries[root].value;
549  Kentries[root].row = Kentries[maxChild].row;
550  Kentries[root].col = Kentries[maxChild].col;
551  Kentries[root].value = Kentries[maxChild].value;
552  Kentries[maxChild].row = row;
553  Kentries[maxChild].col = col;
554  Kentries[maxChild].value = value;
555  root = maxChild;
556  }
557  else
558  done = 1;
559  }
